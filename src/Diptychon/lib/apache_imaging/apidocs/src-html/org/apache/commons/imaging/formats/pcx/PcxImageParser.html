049    public class PcxImageParser extends ImageParser implements PcxConstants {
050        // ZSoft's official spec is at http://www.qzx.com/pc-gpe/pcx.txt
051        // (among other places) but it's pretty thin. The fileformat.info document
052        // at http://www.fileformat.info/format/pcx/egff.htm is a little better
053        // but their gray sample image seems corrupt. PCX files themselves are
054        // the ultimate test but pretty hard to find nowdays, so the best
055        // test is against other image viewers (Irfanview is pretty good).
056        //
057        // Open source projects are generally poor at parsing PCX,
058        // SDL_Image/gdk-pixbuf/Eye of Gnome/GIMP/F-Spot all only do some formats,
059        // don't support uncompressed PCX, and/or don't handle black and white
060        // images properly.

101        @Override
102        public ImageInfo getImageInfo(final ByteSource byteSource, final Map<String,Object> params)
103                throws ImageReadException, IOException {
104            final PcxHeader pcxHeader = readPcxHeader(byteSource);
105            final Dimension size = getImageSize(byteSource, params);
106            return new ImageInfo(
107                    "PCX",
108                    pcxHeader.nPlanes * pcxHeader.bitsPerPixel,
109                    new ArrayList<String>(),
110                    ImageFormat.IMAGE_FORMAT_PCX,
111                    "ZSoft PCX Image",
112                    size.height,
113                    "image/x-pcx",
114                    1,
115                    pcxHeader.vDpi,
116                    Math.round(size.getHeight() / pcxHeader.vDpi),
117                    pcxHeader.hDpi,
118                    Math.round(size.getWidth() / pcxHeader.hDpi),
119                    size.width,
120                    false,
121                    false,
122                    !(pcxHeader.nPlanes == 3 && pcxHeader.bitsPerPixel == 8),
123                    ImageInfo.COLOR_TYPE_RGB,
124                    pcxHeader.encoding == PcxHeader.ENCODING_RLE ? ImageInfo.COMPRESSION_ALGORITHM_RLE
125                            : ImageInfo.COMPRESSION_ALGORITHM_NONE);
126        }

128        @Override
129        public Dimension getImageSize(final ByteSource byteSource, final Map<String,Object> params)
130                throws ImageReadException, IOException {
131            final PcxHeader pcxHeader = readPcxHeader(byteSource);
132            final int xSize = pcxHeader.xMax - pcxHeader.xMin + 1;
133            if (xSize < 0) {
134                throw new ImageReadException("Image width is negative");
135            }
136            final int ySize = pcxHeader.yMax - pcxHeader.yMin + 1;
137            if (ySize < 0) {
138                throw new ImageReadException("Image height is negative");
139            }
140            return new Dimension(xSize, ySize);
141        }

150    
151            public static final int ENCODING_UNCOMPRESSED = 0;
152            public static final int ENCODING_RLE = 1;
153            public static final int PALETTE_INFO_COLOR = 1;
154            public static final int PALETTE_INFO_GRAYSCALE = 2;
155            public final int manufacturer; // Always 10 = ZSoft .pcx
156            public final int version; // 0 = PC Paintbrush 2.5
157                                      // 2 = PC Paintbrush 2.8 with palette
158                                      // 3 = PC Paintbrush 2.8 w/o palette
159                                      // 4 = PC Paintbrush for Windows
160                                      // 5 = PC Paintbrush >= 3.0
161            public final int encoding; // 0 = very old uncompressed format, 1 = .pcx
162                                       // run length encoding
163            public final int bitsPerPixel; // Bits ***PER PLANE*** for each pixel
164            public final int xMin; // window
165            public final int yMin;
166            public final int xMax;
167            public final int yMax;
168            public final int hDpi; // horizontal dpi
169            public final int vDpi; // vertical dpi
170            public final int[] colormap; // palette for <= 16 colors
171            public final int reserved; // Always 0
172            public final int nPlanes; // Number of color planes
173            public final int bytesPerLine; // Number of bytes per scanline plane,
174                                           // must be an even number.
175            public final int paletteInfo; // 1 = Color/BW, 2 = Grayscale, ignored in
176                                          // Paintbrush IV/IV+
177            public final int hScreenSize; // horizontal screen size, in pixels.
178                                          // PaintBrush >= IV only.
179            public final int vScreenSize; // vertical screen size, in pixels.
180                                          // PaintBrush >= IV only.

209                pw.println("PcxHeader");
210                pw.println("Manufacturer: " + manufacturer);
211                pw.println("Version: " + version);
212                pw.println("Encoding: " + encoding);
213                pw.println("BitsPerPixel: " + bitsPerPixel);
214                pw.println("xMin: " + xMin);
215                pw.println("yMin: " + yMin);
216                pw.println("xMax: " + xMax);
217                pw.println("yMax: " + yMax);
218                pw.println("hDpi: " + hDpi);
219                pw.println("vDpi: " + vDpi);
220                pw.print("ColorMap: ");
221                for (int i = 0; i < colormap.length; i++) {
222                    if (i > 0) {
223                        pw.print(",");
224                    }
225                    pw.print("(" + (0xff & (colormap[i] >> 16)) + ","
226                            + (0xff & (colormap[i] >> 8)) + ","
227                            + (0xff & colormap[i]) + ")");
228                }
229                pw.println();
230                pw.println("Reserved: " + reserved);
231                pw.println("nPlanes: " + nPlanes);
232                pw.println("BytesPerLine: " + bytesPerLine);
233                pw.println("PaletteInfo: " + paletteInfo);
234                pw.println("hScreenSize: " + hScreenSize);
235                pw.println("vScreenSize: " + vScreenSize);
236                pw.println();
237            }

240        private PcxHeader readPcxHeader(final ByteSource byteSource)
241                throws ImageReadException, IOException {
242            InputStream is = null;
243            try {
244                is = byteSource.getInputStream();
245                return readPcxHeader(is, false);
246            } finally {
247                try {
248                    if (is != null) {
249                        is.close();
250                    }
251                } catch (final IOException ignored) {
252                }
253            }
254        }

256        private PcxHeader readPcxHeader(final InputStream is, final boolean isStrict)
257                throws ImageReadException, IOException {
258            final byte[] pcxHeaderBytes = readByteArray("PcxHeader", 128, is,
259                    "Not a Valid PCX File");
260            final int manufacturer = 0xff & pcxHeaderBytes[0];
261            final int version = 0xff & pcxHeaderBytes[1];
262            final int encoding = 0xff & pcxHeaderBytes[2];
263            final int bitsPerPixel = 0xff & pcxHeaderBytes[3];
264            final int xMin = convertByteArrayToShort("xMin", 4, pcxHeaderBytes,
265                    ByteOrder.LITTLE_ENDIAN);
266            final int yMin = convertByteArrayToShort("yMin", 6, pcxHeaderBytes,
267                    ByteOrder.LITTLE_ENDIAN);
268            final int xMax = convertByteArrayToShort("xMax", 8, pcxHeaderBytes,
269                    ByteOrder.LITTLE_ENDIAN);
270            final int yMax = convertByteArrayToShort("yMax", 10, pcxHeaderBytes,
271                    ByteOrder.LITTLE_ENDIAN);
272            final int hDpi = convertByteArrayToShort("hDpi", 12, pcxHeaderBytes,
273                    ByteOrder.LITTLE_ENDIAN);
274            final int vDpi = convertByteArrayToShort("vDpi", 14, pcxHeaderBytes,
275                    ByteOrder.LITTLE_ENDIAN);
276            final int[] colormap = new int[16];
277            for (int i = 0; i < 16; i++) {
278                colormap[i] = 0xff000000
279                        | ((0xff & pcxHeaderBytes[16 + 3 * i]) << 16)
280                        | ((0xff & pcxHeaderBytes[16 + 3 * i + 1]) << 8)
281                        | (0xff & pcxHeaderBytes[16 + 3 * i + 2]);
282            }
283            final int reserved = 0xff & pcxHeaderBytes[64];
284            final int nPlanes = 0xff & pcxHeaderBytes[65];
285            final int bytesPerLine = convertByteArrayToShort("BytesPerLine", 66,
286                    pcxHeaderBytes, ByteOrder.LITTLE_ENDIAN);
287            final int paletteInfo = convertByteArrayToShort("PaletteInfo", 68,
288                    pcxHeaderBytes, ByteOrder.LITTLE_ENDIAN);
289            final int hScreenSize = convertByteArrayToShort("hScreenSize", 70,
290                    pcxHeaderBytes, ByteOrder.LITTLE_ENDIAN);
291            final int vScreenSize = convertByteArrayToShort("vScreenSize", 72,
292                    pcxHeaderBytes, ByteOrder.LITTLE_ENDIAN);
293    
294            if (manufacturer != 10) {
295                throw new ImageReadException(
296                        "Not a Valid PCX File: manufacturer is " + manufacturer);
297            }
298            if (isStrict) {
299                // Note that reserved is sometimes set to a non-zero value
300                // by Paintbrush itself, so it shouldn't be enforced.
301                if (bytesPerLine % 2 != 0) {
302                    throw new ImageReadException(
303                            "Not a Valid PCX File: bytesPerLine is odd");
304                }
305            }
306    
307            return new PcxHeader(manufacturer, version, encoding, bitsPerPixel,
308                    xMin, yMin, xMax, yMax, hDpi, vDpi, colormap, reserved,
309                    nPlanes, bytesPerLine, paletteInfo, hScreenSize, vScreenSize);
310        }

319        private void readScanLine(final PcxHeader pcxHeader, final InputStream is,
320                final byte[] samples) throws IOException, ImageReadException {
321            if (pcxHeader.encoding == PcxHeader.ENCODING_UNCOMPRESSED) {
322                int r;
323                for (int bytesRead = 0; bytesRead < samples.length; bytesRead += r) {
324                    r = is.read(samples, bytesRead, samples.length - bytesRead);
325                    if (r < 0) {
326                        throw new ImageReadException(
327                                "Premature end of file reading image data");
328                    }
329                }
330            } else {
331                if (pcxHeader.encoding == PcxHeader.ENCODING_RLE) {
332                    for (int bytesRead = 0; bytesRead < samples.length;) {
333                        final byte b = readByte("Pixel", is, "Error reading image data");
334                        int count;
335                        byte sample;
336                        if ((b & 0xc0) == 0xc0) {
337                            count = b & 0x3f;
338                            sample = readByte("Pixel", is,
339                                    "Error reading image data");
340                        } else {
341                            count = 1;
342                            sample = b;
343                        }
344                        for (int i = 0; i < count && bytesRead + i < samples.length; i++) {
345                            samples[bytesRead + i] = sample;
346                        }
347                        bytesRead += count;
348                    }
349                } else {
350                    throw new ImageReadException("Invalid PCX encoding "
351                            + pcxHeader.encoding);
352                }
353            }
354        }

356        private int[] read256ColorPalette(final InputStream stream) throws IOException {
357            final byte[] paletteBytes = readByteArray("Palette", 769, stream,
358                    "Error reading palette");
359            if (paletteBytes[0] != 12) {
360                return null;
361            }
362            final int[] palette = new int[256];
363            for (int i = 0; i < palette.length; i++) {
364                palette[i] = ((0xff & paletteBytes[1 + 3 * i]) << 16)
365                        | ((0xff & paletteBytes[1 + 3 * i + 1]) << 8)
366                        | (0xff & paletteBytes[1 + 3 * i + 2]);
367            }
368            return palette;
369        }

371        private int[] read256ColorPaletteFromEndOfFile(final ByteSource byteSource)
372                throws IOException {
373            InputStream stream = null;
374            try {
375                stream = byteSource.getInputStream();
376                final long toSkip = byteSource.getLength() - 769;
377                skipBytes(stream, (int) toSkip);
378                return read256ColorPalette(stream);
379            } finally {
380                try {
381                    if (stream != null) {
382                        stream.close();
383                    }
384                } catch (final IOException closeException) {
385                }
386            }
387        }

389        private BufferedImage readImage(final PcxHeader pcxHeader, final InputStream is,
390                final ByteSource byteSource) throws ImageReadException, IOException {
391            final int xSize = pcxHeader.xMax - pcxHeader.xMin + 1;
392            if (xSize < 0) {
393                throw new ImageReadException("Image width is negative");
394            }
395            final int ySize = pcxHeader.yMax - pcxHeader.yMin + 1;
396            if (ySize < 0) {
397                throw new ImageReadException("Image height is negative");
398            }
399    
400            final int scanlineLength = pcxHeader.bytesPerLine * pcxHeader.nPlanes;
401            final byte[] scanline = new byte[scanlineLength];
402            if ((pcxHeader.bitsPerPixel == 1 || pcxHeader.bitsPerPixel == 2
403                    || pcxHeader.bitsPerPixel == 4 || pcxHeader.bitsPerPixel == 8)
404                    && pcxHeader.nPlanes == 1) {
405                final int bytesPerImageRow = (xSize * pcxHeader.bitsPerPixel + 7) / 8;
406                final byte[] image = new byte[ySize * bytesPerImageRow];
407                for (int y = 0; y < ySize; y++) {
408                    readScanLine(pcxHeader, is, scanline);
409                    System.arraycopy(scanline, 0, image, y * bytesPerImageRow,
410                            bytesPerImageRow);
411                }
412                final DataBufferByte dataBuffer = new DataBufferByte(image, image.length);
413                int[] palette;
414                if (pcxHeader.bitsPerPixel == 1) {
415                    palette = new int[] { 0x000000, 0xffffff };
416                } else if (pcxHeader.bitsPerPixel == 8) {
417                    // Normally the palette is read 769 bytes from the end of the
418                    // file.
419                    // However DCX files have multiple PCX images in one file, so
420                    // there could be extra data before the end! So try look for the
421                    // palette
422                    // immediately after the image data first.
423                    palette = read256ColorPalette(is);
424                    if (palette == null) {
425                        palette = read256ColorPaletteFromEndOfFile(byteSource);
426                    }
427                    if (palette == null) {
428                        throw new ImageReadException(
429                                "No 256 color palette found in image that needs it");
430                    }
431                } else {
432                    palette = pcxHeader.colormap;
433                }
434                WritableRaster raster;
435                if (pcxHeader.bitsPerPixel == 8) {
436                    raster = WritableRaster.createInterleavedRaster(dataBuffer,
437                            xSize, ySize, bytesPerImageRow, 1, new int[] { 0 },
438                            null);
439                } else {
440                    raster = WritableRaster.createPackedRaster(dataBuffer, xSize,
441                            ySize, pcxHeader.bitsPerPixel, null);
442                }
443                final IndexColorModel colorModel = new IndexColorModel(
444                        pcxHeader.bitsPerPixel, 1 << pcxHeader.bitsPerPixel,
445                        palette, 0, false, -1, DataBuffer.TYPE_BYTE);
446                return new BufferedImage(colorModel, raster,
447                        colorModel.isAlphaPremultiplied(), new Properties());
448            } else if (pcxHeader.bitsPerPixel == 1 && 2 <= pcxHeader.nPlanes
449                    && pcxHeader.nPlanes <= 4) {
450                final IndexColorModel colorModel = new IndexColorModel(pcxHeader.nPlanes,
451                        1 << pcxHeader.nPlanes, pcxHeader.colormap, 0, false, -1,
452                        DataBuffer.TYPE_BYTE);
453                final BufferedImage image = new BufferedImage(xSize, ySize,
454                        BufferedImage.TYPE_BYTE_BINARY, colorModel);
455                final byte[] unpacked = new byte[xSize];
456                for (int y = 0; y < ySize; y++) {
457                    readScanLine(pcxHeader, is, scanline);
458                    int nextByte = 0;
459                    Arrays.fill(unpacked, (byte) 0);
460                    for (int plane = 0; plane < pcxHeader.nPlanes; plane++) {
461                        for (int i = 0; i < pcxHeader.bytesPerLine; i++) {
462                            final int b = 0xff & scanline[nextByte++];
463                            for (int j = 0; j < 8 && 8 * i + j < unpacked.length; j++) {
464                                unpacked[8 * i + j] |= (byte) (((b >> (7 - j)) & 0x1) << plane);
465                            }
466                        }
467                    }
468                    image.getRaster().setDataElements(0, y, xSize, 1, unpacked);
469                }
470                return image;
471            } else if (pcxHeader.bitsPerPixel == 8 && pcxHeader.nPlanes == 3) {
472                final byte[][] image = new byte[3][];
473                image[0] = new byte[xSize * ySize];
474                image[1] = new byte[xSize * ySize];
475                image[2] = new byte[xSize * ySize];
476                for (int y = 0; y < ySize; y++) {
477                    readScanLine(pcxHeader, is, scanline);
478                    System.arraycopy(scanline, 0, image[0], y * xSize, xSize);
479                    System.arraycopy(scanline, pcxHeader.bytesPerLine, image[1], y
480                            * xSize, xSize);
481                    System.arraycopy(scanline, 2 * pcxHeader.bytesPerLine,
482                            image[2], y * xSize, xSize);
483                }
484                final DataBufferByte dataBuffer = new DataBufferByte(image,
485                        image[0].length);
486                final WritableRaster raster = WritableRaster.createBandedRaster(
487                        dataBuffer, xSize, ySize, xSize, new int[] { 0, 1, 2 },
488                        new int[] { 0, 0, 0 }, null);
489                final ColorModel colorModel = new ComponentColorModel(
490                        ColorSpace.getInstance(ColorSpace.CS_sRGB), false, false,
491                        Transparency.OPAQUE, DataBuffer.TYPE_BYTE);
492                return new BufferedImage(colorModel, raster,
493                        colorModel.isAlphaPremultiplied(), new Properties());
494            } else if ((pcxHeader.bitsPerPixel == 24 && pcxHeader.nPlanes == 1)
495                    || (pcxHeader.bitsPerPixel == 32 && pcxHeader.nPlanes == 1)) {
496                final int rowLength = 3 * xSize;
497                final byte[] image = new byte[rowLength * ySize];
498                for (int y = 0; y < ySize; y++) {
499                    readScanLine(pcxHeader, is, scanline);
500                    if (pcxHeader.bitsPerPixel == 24) {
501                        System.arraycopy(scanline, 0, image, y * rowLength,
502                                rowLength);
503                    } else {
504                        for (int x = 0; x < xSize; x++) {
505                            image[y * rowLength + 3 * x] = scanline[4 * x];
506                            image[y * rowLength + 3 * x + 1] = scanline[4 * x + 1];
507                            image[y * rowLength + 3 * x + 2] = scanline[4 * x + 2];
508                        }
509                    }
510                }
511                final DataBufferByte dataBuffer = new DataBufferByte(image, image.length);
512                final WritableRaster raster = WritableRaster.createInterleavedRaster(
513                        dataBuffer, xSize, ySize, rowLength, 3,
514                        new int[] { 2, 1, 0 }, null);
515                final ColorModel colorModel = new ComponentColorModel(
516                        ColorSpace.getInstance(ColorSpace.CS_sRGB), false, false,
517                        Transparency.OPAQUE, DataBuffer.TYPE_BYTE);
518                return new BufferedImage(colorModel, raster,
519                        colorModel.isAlphaPremultiplied(), new Properties());
520            } else {
521                throw new ImageReadException(
522                        "Invalid/unsupported image with bitsPerPixel "
523                                + pcxHeader.bitsPerPixel + " and planes "
524                                + pcxHeader.nPlanes);
525            }
526        }

528        @Override
529        public final BufferedImage getBufferedImage(final ByteSource byteSource,
530                Map<String,Object> params) throws ImageReadException, IOException {
531            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
532            boolean isStrict = false;
533            final Object strictness = params.get(PARAM_KEY_STRICT);
534            if (strictness != null) {
535                isStrict = ((Boolean) strictness).booleanValue();
536            }
537    
538            InputStream is = null;
539            try {
540                is = byteSource.getInputStream();
541                final PcxHeader pcxHeader = readPcxHeader(is, isStrict);
542                return readImage(pcxHeader, is, byteSource);
543            } finally {
544                try {
545                    if (is != null) {
546                        is.close();
547                    }
548                } catch (final IOException ignored) {
549                }
550            }
551        }
