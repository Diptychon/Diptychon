108        public final void debugNumber(final String msg, final int data, final int bytes) {
109            System.out.print(msg + ": " + data + " (");
110            int byteData = data;
111            for (int i = 0; i < bytes; i++) {
112                if (i > 0) {
113                    System.out.print(",");
114                }
115                final int singleByte = 0xff & byteData;
116                System.out.print((char) singleByte + " [" + singleByte + "]");
117                byteData >>= 8;
118            }
119            System.out.println(") [0x" + Integer.toHexString(data) + ", "
120                    + Integer.toBinaryString(data) + "]");
121        }

123        public final void readAndVerifyBytes(final byte expected[], final String exception)
124                throws ImageReadException, IOException {
125            for (int i = 0; i < expected.length; i++) {
126                final int data = is.read();
127                final byte b = (byte) (0xff & data);
128    
129                if ((data < 0) || (b != expected[i])) {
130                    System.out.println("i" + ": " + i);
131    
132                    this.debugByteArray("expected", expected);
133                    debugNumber("data[" + i + "]", b);
134                    // debugNumber("expected[" + i + "]", expected[i]);
135    
136                    throw new ImageReadException(exception);
137                }
138            }
139        }

156        public final void skipBytes(final int length, final String exception)
157                throws IOException {
158            long total = 0;
159            while (length != total) {
160                final long skipped = is.skip(length - total);
161                if (skipped < 1) {
162                    throw new IOException(exception + " (" + skipped + ")");
163                }
164                total += skipped;
165            }
166        }

168        protected final void scanForByte(final byte value) throws IOException {
169            int count = 0;
170            for (int i = 0; count < 3; i++) {
171                final int b = is.read();
172                if (b < 0) {
173                    return;
174                }
175                if ((0xff & b) == value) {
176                    System.out.println("\t" + i + ": match.");
177                    count++;
178                }
179            }
180        }

182        public final byte readByte(final String name, final String exception)
183                throws IOException {
184            final int result = is.read();
185    
186            if ((result < 0)) {
187                System.out.println(name + ": " + result);
188                throw new IOException(exception);
189            }
190    
191            if (debug) {
192                debugNumber(name, result);
193            }
194    
195            return (byte) (0xff & result);

198        protected final RationalNumber[] convertByteArrayToRationalArray(
199                final String name, final byte bytes[], final int start, final int length, final ByteOrder byteOrder) {
200            final int expectedLength = start + length * 8;
201    
202            if (bytes.length < expectedLength) {
203                System.out.println(name + ": expected length: " + expectedLength
204                        + ", actual length: " + bytes.length);
205                return null;
206            }
207    
208            final RationalNumber result[] = new RationalNumber[length];
209    
210            for (int i = 0; i < length; i++) {
211                result[i] = convertByteArrayToRational(name, bytes, start + i * 8,
212                        byteOrder);
213            }
214    
215            return result;
216        }

223        protected final RationalNumber convertByteArrayToRational(final String name,
224                final byte bytes[], final int start, final ByteOrder byteOrder) {
225            final int numerator = convertByteArrayToInt(name, bytes, start + 0, 4,
226                    byteOrder);
227            final int divisor = convertByteArrayToInt(name, bytes, start + 4, 4,

238        protected final int convertByteArrayToInt(final String name, final byte bytes[],
239                final int start, final int length, final ByteOrder byteOrder) {
240            final byte byte0 = bytes[start + 0];
241            final byte byte1 = bytes[start + 1];
242            final byte byte2 = bytes[start + 2];
243            byte byte3 = 0;
244            if (length == 4) {
245                byte3 = bytes[start + 3];
246            }
247    
248            int result;
249    
250            if (byteOrder == ByteOrder.MOTOROLA) {
251                result = ((0xff & byte0) << 24) + ((0xff & byte1) << 16)
252                        + ((0xff & byte2) << 8) + ((0xff & byte3) << 0);
253            } else {
254                result = ((0xff & byte3) << 24) + ((0xff & byte2) << 16)
255                        + ((0xff & byte1) << 8) + ((0xff & byte0) << 0);
256            }
257    
258            if (debug) {
259                debugNumber(name, result, 4);
260            }
261    
262            return result;
263        }

334        public final byte[] readByteArray(final String name, final int length, final String exception)
335                throws IOException {
336            final byte result[] = new byte[length];
337    
338            int read = 0;
339            while (read < length) {
340                final int count = is.read(result, read, length - read);
341                if (count < 1) {
342                    throw new IOException(exception);
343                }
344    
345                read += count;
346            }
347    
348            if (debug) {
349                for (int i = 0; ((i < length) && (i < 150)); i++) {
350                    debugNumber(name + " (" + i + ")", 0xff & result[i]);
351                }
352            }
353            return result;
354        }

372        public final byte[] readBytearray(final String name, final byte bytes[], final int start,
373                final int count) {
374            if (bytes.length < (start + count)) {
375                return null;
376            }
377    
378            final byte result[] = new byte[count];
379            System.arraycopy(bytes, start, result, 0, count);
380    
381            if (debug) {
382                debugByteArray(name, result);
383            }
384    
385            return result;

395        public final byte[] readByteArray(final int length, final String error,
396                final boolean verbose, final boolean strict) throws ImageReadException,
397                IOException {
398            final byte bytes[] = new byte[length];
399            int total = 0;
400            int read;
401            while ((read = read(bytes, total, length - total)) > 0) {
402                total += read;
403            }
404            if (total < length) {
405                if (strict) {
406                    throw new ImageReadException(error);
407                } else if (verbose) {
408                    System.out.println(error);
409                }
410                return null;
411            }
412            return bytes;
413        }

423        public final boolean compareByteArrays(final byte a[], final int aStart, final byte b[],
424                final int bStart, final int length) {
425            if (a.length < (aStart + length)) {
426                return false;
427            }
428            if (b.length < (bStart + length)) {
429                return false;
430            }
431    
432            for (int i = 0; i < length; i++) {
433                if (a[aStart + i] != b[bStart + i]) {
434                    debugNumber("a[" + (aStart + i) + "]", a[aStart + i]);
435                    debugNumber("b[" + (bStart + i) + "]", b[bStart + i]);
436    
437                    return false;
438                }
439            }
440    
441            return true;
442        }

509        public final int read2ByteInteger(final String exception)
510                throws ImageReadException, IOException {
511            final int byte0 = is.read();
512            final int byte1 = is.read();
513            if (byte0 < 0 || byte1 < 0) {
514                throw new ImageReadException(exception);
515            }
516    
517            if (byteOrder == ByteOrder.MOTOROLA) {
518                return ((0xff & byte0) << 8) + ((0xff & byte1) << 0);
519            } else {
520                return ((0xff & byte1) << 8) + ((0xff & byte0) << 0);
521            }
522        }

524        public final int read4ByteInteger(final String exception)
525                throws ImageReadException, IOException {
526            final int byte0 = is.read();
527            final int byte1 = is.read();
528            final int byte2 = is.read();
529            final int byte3 = is.read();
530            if (byte0 < 0 || byte1 < 0 || byte2 < 0 || byte3 < 0) {
531                throw new ImageReadException(exception);
532            }
533    
534            if (byteOrder == ByteOrder.MOTOROLA) {
535                return ((0xff & byte0) << 24) + ((0xff & byte1) << 16)
536                        + ((0xff & byte2) << 8) + ((0xff & byte3) << 0);
537            } else {
538                return ((0xff & byte3) << 24) + ((0xff & byte2) << 16)
539                        + ((0xff & byte1) << 8) + ((0xff & byte0) << 0);
540            }
541        }

570        protected final byte[] getRAFBytes(final RandomAccessFile raf, final long pos,
571                final int length, final String exception) throws IOException {
572            final byte result[] = new byte[length];
573    
574            if (debug) {
575                System.out.println("getRAFBytes pos" + ": " + pos);
576                System.out.println("getRAFBytes length" + ": " + length);
577            }
578    
579            raf.seek(pos);
580    
581            int read = 0;
582            while (read < length) {
583                final int count = raf.read(result, read, length - read);
584                if (count < 1) {
585                    throw new IOException(exception);
586                }
587    
588                read += count;
589            }
590    
591            return result;
592    
593        }
