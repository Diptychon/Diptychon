017    package org.apache.commons.imaging.icc;
018    
019    import java.io.IOException;
020    import java.io.PrintWriter;
021    import java.io.StringWriter;
022    
023    import org.apache.commons.imaging.ImageReadException;
024    
025    public class IccProfileInfo implements IccConstants {
026    
027        public final byte data[];
028        public final int ProfileSize;
029        public final int CMMTypeSignature;
030        public final int ProfileVersion;
031        public final int ProfileDeviceClassSignature;
032        public final int ColorSpace;
033        public final int ProfileConnectionSpace;
034        public final int ProfileFileSignature;
035        public final int PrimaryPlatformSignature;
036        public final int VariousFlags;
037        public final int DeviceManufacturer;
038        public final int DeviceModel;
039        public final int RenderingIntent;
040        public final int ProfileCreatorSignature;
041        public final byte ProfileID[];
042        public final IccTag tags[];
043    
044        public IccProfileInfo(final byte data[], final int ProfileSize, final int CMMTypeSignature,
045                final int ProfileVersion, final int ProfileDeviceClassSignature,
046                final int ColorSpace, final int ProfileConnectionSpace,
047                final int ProfileFileSignature, final int PrimaryPlatformSignature,
048                final int VariousFlags, final int DeviceManufacturer, final int DeviceModel,
049                final int RenderingIntent, final int ProfileCreatorSignature, final byte ProfileID[],
050                final IccTag tags[]) {
051            this.data = data;
052    
053            this.ProfileSize = ProfileSize;
054            this.CMMTypeSignature = CMMTypeSignature;
055            this.ProfileVersion = ProfileVersion;
056            this.ProfileDeviceClassSignature = ProfileDeviceClassSignature;
057            this.ColorSpace = ColorSpace;
058            this.ProfileConnectionSpace = ProfileConnectionSpace;
059            this.ProfileFileSignature = ProfileFileSignature;
060            this.PrimaryPlatformSignature = PrimaryPlatformSignature;
061            this.VariousFlags = VariousFlags;
062            this.DeviceManufacturer = DeviceManufacturer;
063            this.DeviceModel = DeviceModel;
064            this.RenderingIntent = RenderingIntent;
065            this.ProfileCreatorSignature = ProfileCreatorSignature;
066            this.ProfileID = ProfileID;
067    
068            this.tags = tags;
069        }
070    
071        public boolean issRGB() {
072            final boolean result = ((DeviceManufacturer == IEC) && (DeviceModel == sRGB));
073            return result;
074        }
075    
076        private void printCharQuad(final PrintWriter pw, final String msg, final int i) {
077            pw.println(msg + ": '" + (char) (0xff & (i >> 24))
078                    + (char) (0xff & (i >> 16)) + (char) (0xff & (i >> 8))
079                    + (char) (0xff & (i >> 0)) + "'");
080        }
081    
082        public void dump(final String prefix) {
083            System.out.print(toString());
084        }
085    
086        @Override
087        public String toString() {
088            try {
089                return toString("");
090            } catch (final Exception e) {
091                return "IccProfileInfo: Error";
092            }
093        }
094    
095        public String toString(final String prefix) throws ImageReadException,
096                IOException {
097            final StringWriter sw = new StringWriter();
098            final PrintWriter pw = new PrintWriter(sw);
099    
100            // StringBuffer result = new StringBuffer();
101            pw.println(prefix + ": " + "data length: " + data.length);
102    
103            printCharQuad(pw, prefix + ": " + "ProfileDeviceClassSignature",
104                    ProfileDeviceClassSignature);
105    
106            printCharQuad(pw, prefix + ": " + "CMMTypeSignature", CMMTypeSignature);
107    
108            printCharQuad(pw, prefix + ": " + "ProfileDeviceClassSignature",
109                    ProfileDeviceClassSignature);
110            printCharQuad(pw, prefix + ": " + "ColorSpace", ColorSpace);
111            printCharQuad(pw, prefix + ": " + "ProfileConnectionSpace",
112                    ProfileConnectionSpace);
113    
114            printCharQuad(pw, prefix + ": " + "ProfileFileSignature",
115                    ProfileFileSignature);
116    
117            printCharQuad(pw, prefix + ": " + "PrimaryPlatformSignature",
118                    PrimaryPlatformSignature);
119    
120            printCharQuad(pw, prefix + ": " + "ProfileFileSignature",
121                    ProfileFileSignature);
122    
123            printCharQuad(pw, prefix + ": " + "DeviceManufacturer",
124                    DeviceManufacturer);
125    
126            printCharQuad(pw, prefix + ": " + "DeviceModel", DeviceModel);
127    
128            printCharQuad(pw, prefix + ": " + "RenderingIntent", RenderingIntent);
129    
130            printCharQuad(pw, prefix + ": " + "ProfileCreatorSignature",
131                    ProfileCreatorSignature);
132    
133            for (int i = 0; i < tags.length; i++) {
134                final IccTag tag = tags[i];
135                tag.dump(pw, "\t" + i + ": ");
136            }
137    
138            pw.println(prefix + ": " + "issRGB: " + issRGB());
139            pw.flush();
140    
141            return sw.getBuffer().toString();
142        }
143    
144    }
