060        private synchronized static boolean loadColorNames() {
061            if (colorNames != null) {
062                return true;
063            }
064    
065            BufferedReader reader = null;
066            try {
067                final InputStream rgbTxtStream = XpmImageParser.class
068                        .getResourceAsStream("rgb.txt");
069                if (rgbTxtStream == null) {
070                    return false;
071                }
072                reader = new BufferedReader(new InputStreamReader(rgbTxtStream,
073                        "US-ASCII"));
074                final Map<String, Integer> colors = new HashMap<String, Integer>();
075                String line;
076                while ((line = reader.readLine()) != null) {
077                    if (line.startsWith("!")) {
078                        continue;
079                    }
080                    try {
081                        final int red = Integer.parseInt(line.substring(0, 3).trim());
082                        final int green = Integer.parseInt(line.substring(4, 7).trim());
083                        final int blue = Integer.parseInt(line.substring(8, 11).trim());
084                        final String colorName = line.substring(11).trim();
085                        colors.put(colorName, 0xff000000 | (red << 16)
086                                | (green << 8) | blue);
087                    } catch (final NumberFormatException nfe) {
088                    }
089                }
090                colorNames = colors;
091                return true;
092            } catch (final IOException ioException) {
093                Debug.debug(ioException);
094                return false;
095            } finally {
096                try {
097                    if (reader != null) {
098                        reader.close();
099                    }
100                } catch (final IOException ignored) {
101                }
102            }
103        }

140        @Override
141        public ImageInfo getImageInfo(final ByteSource byteSource, final Map<String,Object> params)
142                throws ImageReadException, IOException {
143            final XpmHeader xpmHeader = readXpmHeader(byteSource);
144            boolean isTransparent = false;
145            int colorType = ImageInfo.COLOR_TYPE_BW;
146            for (final Entry<Object, PaletteEntry> entry : xpmHeader.palette
147                    .entrySet()) {
148             final PaletteEntry paletteEntry = entry.getValue();
149             if ((paletteEntry.getBestARGB() & 0xff000000) != 0xff000000) {
150            isTransparent = true;
151             }
152             if (paletteEntry.haveColor) {
153            colorType = ImageInfo.COLOR_TYPE_RGB;
154             } else if (colorType != ImageInfo.COLOR_TYPE_RGB
155                && (paletteEntry.haveGray || paletteEntry.haveGray4Level)) {
156            colorType = ImageInfo.COLOR_TYPE_GRAYSCALE;
157             }
158          }
159            return new ImageInfo("XPM version 3", xpmHeader.numCharsPerPixel * 8,
160                    new ArrayList<String>(), ImageFormat.IMAGE_FORMAT_XPM,
161                    "X PixMap", xpmHeader.height, "image/x-xpixmap", 1, 0, 0, 0, 0,
162                    xpmHeader.width, false, isTransparent, true, colorType,
163                    ImageInfo.COMPRESSION_ALGORITHM_NONE);
164        }

216            int index;
217            boolean haveColor = false;
218            int colorArgb;
219            boolean haveGray = false;
220            int grayArgb;
221            boolean haveGray4Level = false;
222            int gray4LevelArgb;
223            boolean haveMono = false;
224            int monoArgb;

226            int getBestARGB() {
227                if (haveColor) {
228                    return colorArgb;
229                } else if (haveGray) {
230                    return grayArgb;
231                } else if (haveGray4Level) {
232                    return gray4LevelArgb;
233                } else if (haveMono) {
234                    return monoArgb;
235                } else {
236                    return 0x00000000;
237                }
238            }

252        private XpmParseResult parseXpmHeader(final ByteSource byteSource)
253                throws ImageReadException, IOException {
254            InputStream is = null;
255            try {
256                is = byteSource.getInputStream();
257                final StringBuilder firstComment = new StringBuilder();
258                final ByteArrayOutputStream preprocessedFile = BasicCParser.preprocess(
259                        is, firstComment, null);
260                if (!firstComment.toString().trim().equals("XPM")) {
261                    throw new ImageReadException("Parsing XPM file failed, "
262                            + "signature isn't '/* XPM */'");
263                }
264    
265                final XpmParseResult xpmParseResult = new XpmParseResult();
266                xpmParseResult.cParser = new BasicCParser(new ByteArrayInputStream(
267                        preprocessedFile.toByteArray()));
268                xpmParseResult.xpmHeader = parseXpmHeader(xpmParseResult.cParser);
269                return xpmParseResult;
270            } finally {
271                try {
272                    if (is != null) {
273                        is.close();
274                    }
275                } catch (final IOException ignored) {
276                }
277            }
278        }

280        private boolean parseNextString(final BasicCParser cParser,
281                final StringBuilder stringBuilder) throws IOException, ImageReadException {
282            stringBuilder.setLength(0);
283            String token = cParser.nextToken();
284            if (token.charAt(0) != '"') {
285                throw new ImageReadException("Parsing XPM file failed, "
286                        + "no string found where expected");
287            }
288            BasicCParser.unescapeString(stringBuilder, token);
289            for (token = cParser.nextToken(); token.charAt(0) == '"'; token = cParser
290                    .nextToken()) {
291                BasicCParser.unescapeString(stringBuilder, token);
292            }
293            if (token.equals(",")) {
294                return true;
295            } else if (token.equals("}")) {
296                return false;
297            } else {
298                throw new ImageReadException("Parsing XPM file failed, "
299                        + "no ',' or '}' found where expected");
300            }
301        }

303        private XpmHeader parseXpmValuesSection(final String row)
304                throws ImageReadException {
305            final String[] tokens = BasicCParser.tokenizeRow(row);
306            if (tokens.length < 4 && tokens.length > 7) {
307                throw new ImageReadException("Parsing XPM file failed, "
308                        + "<Values> section has incorrect tokens");
309            }
310            try {
311                final int width = Integer.parseInt(tokens[0]);
312                final int height = Integer.parseInt(tokens[1]);
313                final int numColors = Integer.parseInt(tokens[2]);
314                final int numCharsPerPixel = Integer.parseInt(tokens[3]);
315                int xHotSpot = -1;
316                int yHotSpot = -1;
317                boolean xpmExt = false;
318                if (tokens.length >= 6) {
319                    xHotSpot = Integer.parseInt(tokens[4]);
320                    yHotSpot = Integer.parseInt(tokens[5]);
321                }
322                if (tokens.length == 5 || tokens.length == 7) {
323                    if (tokens[tokens.length - 1].equals("XPMEXT")) {
324                        xpmExt = true;
325                    } else {
326                        throw new ImageReadException("Parsing XPM file failed, "
327                                + "can't parse <Values> section XPMEXT");
328                    }
329                }
330                return new XpmHeader(width, height, numColors, numCharsPerPixel,
331                        xHotSpot, yHotSpot, xpmExt);
332            } catch (final NumberFormatException nfe) {
333                throw new ImageReadException("Parsing XPM file failed, "
334                        + "error parsing <Values> section", nfe);
335            }
336        }

338        private int parseColor(String color) throws ImageReadException {
339            if (color.charAt(0) == '#') {
340                color = color.substring(1);
341                if (color.length() == 3) {
342                    final int red = Integer.parseInt(color.substring(0, 1), 16);
343                    final int green = Integer.parseInt(color.substring(1, 2), 16);
344                    final int blue = Integer.parseInt(color.substring(2, 3), 16);
345                    return 0xff000000 | (red << 20) | (green << 12) | (blue << 4);
346                } else if (color.length() == 6) {
347                    return 0xff000000 | Integer.parseInt(color, 16);
348                } else if (color.length() == 9) {
349                    final int red = Integer.parseInt(color.substring(0, 1), 16);
350                    final int green = Integer.parseInt(color.substring(3, 4), 16);
351                    final int blue = Integer.parseInt(color.substring(6, 7), 16);
352                    return 0xff000000 | (red << 16) | (green << 8) | blue;
353                } else if (color.length() == 12) {
354                    final int red = Integer.parseInt(color.substring(0, 1), 16);
355                    final int green = Integer.parseInt(color.substring(4, 5), 16);
356                    final int blue = Integer.parseInt(color.substring(8, 9), 16);
357                    return 0xff000000 | (red << 16) | (green << 8) | blue;
358                } else {
359                    return 0x00000000;
360                }
361            } else if (color.charAt(0) == '%') {
362                throw new ImageReadException("HSV colors are not implemented "
363                        + "even in the XPM specification!");
364            } else if (color.equals("None")) {
365                return 0x00000000;
366            } else {
367                if (!loadColorNames()) {
368                    return 0x00000000;
369                }
370                if (colorNames.containsKey(color)) {
371                    return (colorNames.get(color)).intValue();
372                } else {
373                    return 0x00000000;
374                }
375            }
376        }

378        private void populatePaletteEntry(final PaletteEntry paletteEntry, final String key, final String color) throws ImageReadException {
379            if (key.equals("m")) {
380                paletteEntry.monoArgb = parseColor(color);
381                paletteEntry.haveMono = true;
382            } else if (key.equals("g4")) {
383                paletteEntry.gray4LevelArgb = parseColor(color);
384                paletteEntry.haveGray4Level = true;
385            } else if (key.equals("g")) {
386                paletteEntry.grayArgb = parseColor(color);
387                paletteEntry.haveGray = true;
388            } else if (key.equals("s")) {
389                paletteEntry.colorArgb = parseColor(color);
390                paletteEntry.haveColor = true;
391            } else if (key.equals("c")) {
392                paletteEntry.colorArgb = parseColor(color);
393                paletteEntry.haveColor = true;
394            }
395        }

397        private void parsePaletteEntries(final XpmHeader xpmHeader, final BasicCParser cParser)
398                throws IOException, ImageReadException {
399            final StringBuilder row = new StringBuilder();
400            for (int i = 0; i < xpmHeader.numColors; i++) {
401                row.setLength(0);
402                final boolean hasMore = parseNextString(cParser, row);
403                if (!hasMore) {
404                    throw new ImageReadException("Parsing XPM file failed, "
405                            + "file ended while reading palette");
406                }
407                final String name = row.substring(0, xpmHeader.numCharsPerPixel);
408                final String[] tokens = BasicCParser.tokenizeRow(
409                        row.substring(xpmHeader.numCharsPerPixel));
410                final PaletteEntry paletteEntry = new PaletteEntry();
411                paletteEntry.index = i;
412                int previousKeyIndex = Integer.MIN_VALUE;
413                final StringBuilder colorBuffer = new StringBuilder();
414                for (int j = 0; j < tokens.length; j++) {
415                    final String token = tokens[j];
416                    boolean isKey = false;
417                    if (previousKeyIndex < (j - 1) && 
418                        token.equals("m") || token.equals("g4") ||
419                        token.equals("g") || token.equals("c") ||
420                        token.equals("s")) {
421                        isKey = true;
422                    }
423                    if (isKey) {
424                        if (previousKeyIndex >= 0) {
425                            final String key = tokens[previousKeyIndex];
426                            final String color = colorBuffer.toString();
427                            colorBuffer.setLength(0);
428                            populatePaletteEntry(paletteEntry, key, color);
429                        }
430                        previousKeyIndex = j;
431                    } else {
432                        if (previousKeyIndex < 0) {
433                            break;
434                        }
435                        if (colorBuffer.length() > 0) {
436                            colorBuffer.append(' ');
437                        }
438                        colorBuffer.append(token);
439                    }
440                }
441                if (previousKeyIndex >= 0 && colorBuffer.length() > 0) {
442                    final String key = tokens[previousKeyIndex];
443                    final String color = colorBuffer.toString();
444                    colorBuffer.setLength(0);
445                    populatePaletteEntry(paletteEntry, key, color);
446                }
447                xpmHeader.palette.put(name, paletteEntry);
448            }
449        }

451        private XpmHeader parseXpmHeader(final BasicCParser cParser)
452                throws ImageReadException, IOException {
453            String name;
454            String token;
455            token = cParser.nextToken();
456            if (token == null || !token.equals("static")) {
457                throw new ImageReadException(
458                        "Parsing XPM file failed, no 'static' token");
459            }
460            token = cParser.nextToken();
461            if (token == null || !token.equals("char")) {
462                throw new ImageReadException(
463                        "Parsing XPM file failed, no 'char' token");
464            }
465            token = cParser.nextToken();
466            if (token == null || !token.equals("*")) {
467                throw new ImageReadException(
468                        "Parsing XPM file failed, no '*' token");
469            }
470            name = cParser.nextToken();
471            if (name == null) {
472                throw new ImageReadException(
473                        "Parsing XPM file failed, no variable name");
474            }
475            if (name.charAt(0) != '_' && !Character.isLetter(name.charAt(0))) {
476                throw new ImageReadException(
477                        "Parsing XPM file failed, variable name "
478                                + "doesn't start with letter or underscore");
479            }
480            for (int i = 0; i < name.length(); i++) {
481                final char c = name.charAt(i);
482                if (!Character.isLetterOrDigit(c) && c != '_') {
483                    throw new ImageReadException(
484                            "Parsing XPM file failed, variable name "
485                                    + "contains non-letter non-digit non-underscore");
486                }
487            }
488            token = cParser.nextToken();
489            if (token == null || !token.equals("[")) {
490                throw new ImageReadException(
491                        "Parsing XPM file failed, no '[' token");
492            }
493            token = cParser.nextToken();
494            if (token == null || !token.equals("]")) {
495                throw new ImageReadException(
496                        "Parsing XPM file failed, no ']' token");
497            }
498            token = cParser.nextToken();
499            if (token == null || !token.equals("=")) {
500                throw new ImageReadException(
501                        "Parsing XPM file failed, no '=' token");
502            }
503            token = cParser.nextToken();
504            if (token == null || !token.equals("{")) {
505                throw new ImageReadException(
506                        "Parsing XPM file failed, no '{' token");
507            }
508    
509            final StringBuilder row = new StringBuilder();
510            final boolean hasMore = parseNextString(cParser, row);
511            if (!hasMore) {
512                throw new ImageReadException("Parsing XPM file failed, "
513                        + "file too short");
514            }
515            final XpmHeader xpmHeader = parseXpmValuesSection(row.toString());
516            parsePaletteEntries(xpmHeader, cParser);
517            return xpmHeader;
518        }

520        private BufferedImage readXpmImage(final XpmHeader xpmHeader, final BasicCParser cParser)
521                throws ImageReadException, IOException {
522            ColorModel colorModel;
523            WritableRaster raster;
524            int bpp;
525            if (xpmHeader.palette.size() <= (1 << 8)) {
526                final int[] palette = new int[xpmHeader.palette.size()];
527                for (final Entry<Object, PaletteEntry> entry : xpmHeader.palette
528                        .entrySet()) {
529                final PaletteEntry paletteEntry = entry.getValue();
530                palette[paletteEntry.index] = paletteEntry.getBestARGB();
531             }
532                colorModel = new IndexColorModel(8, xpmHeader.palette.size(),
533                        palette, 0, true, -1, DataBuffer.TYPE_BYTE);
534                raster = WritableRaster.createInterleavedRaster(
535                        DataBuffer.TYPE_BYTE, xpmHeader.width, xpmHeader.height, 1,
536                        null);
537                bpp = 8;
538            } else if (xpmHeader.palette.size() <= (1 << 16)) {
539                final int[] palette = new int[xpmHeader.palette.size()];
540                for (final Entry<Object, PaletteEntry> entry : xpmHeader.palette
541                        .entrySet()) {
542                final PaletteEntry paletteEntry = entry.getValue();
543                palette[paletteEntry.index] = paletteEntry.getBestARGB();
544             }
545                colorModel = new IndexColorModel(16, xpmHeader.palette.size(),
546                        palette, 0, true, -1, DataBuffer.TYPE_USHORT);
547                raster = WritableRaster.createInterleavedRaster(
548                        DataBuffer.TYPE_USHORT, xpmHeader.width, xpmHeader.height,
549                        1, null);
550                bpp = 16;
551            } else {
552                colorModel = new DirectColorModel(32, 0x00ff0000, 0x0000ff00,
553                        0x000000ff, 0xff000000);
554                raster = WritableRaster.createPackedRaster(DataBuffer.TYPE_INT,
555                        xpmHeader.width, xpmHeader.height, new int[] { 0x00ff0000,
556                                0x0000ff00, 0x000000ff, 0xff000000 }, null);
557                bpp = 32;
558            }
559    
560            final BufferedImage image = new BufferedImage(colorModel, raster,
561                    colorModel.isAlphaPremultiplied(), new Properties());
562            final DataBuffer dataBuffer = raster.getDataBuffer();
563            final StringBuilder row = new StringBuilder();
564            boolean hasMore = true;
565            for (int y = 0; y < xpmHeader.height; y++) {
566                row.setLength(0);
567                hasMore = parseNextString(cParser, row);
568                if (y < (xpmHeader.height - 1) && !hasMore) {
569                    throw new ImageReadException("Parsing XPM file failed, "
570                            + "insufficient image rows in file");
571                }
572                final int rowOffset = y * xpmHeader.width;
573                for (int x = 0; x < xpmHeader.width; x++) {
574                    final String index = row.substring(x * xpmHeader.numCharsPerPixel,
575                            (x + 1) * xpmHeader.numCharsPerPixel);
576                    final PaletteEntry paletteEntry = xpmHeader.palette.get(index);
577                    if (paletteEntry == null) {
578                        throw new ImageReadException(
579                                "No palette entry was defined " + "for " + index);
580                    }
581                    if (bpp <= 16) {
582                        dataBuffer.setElem(rowOffset + x, paletteEntry.index);
583                    } else {
584                        dataBuffer.setElem(rowOffset + x,
585                                paletteEntry.getBestARGB());
586                    }
587                }
588            }
589    
590            while (hasMore) {
591                row.setLength(0);
592                hasMore = parseNextString(cParser, row);
593            }
594    
595            final String token = cParser.nextToken();
596            if (!token.equals(";")) {
597                throw new ImageReadException("Last token wasn't ';'");
598            }
599    
600            return image;
601        }

617        private String randomName() {
618            final UUID uuid = UUID.randomUUID();
619            final StringBuilder stringBuilder = new StringBuilder("a");
620            long bits = uuid.getMostSignificantBits();
621            // Long.toHexString() breaks for very big numbers
622            for (int i = 64 - 8; i >= 0; i -= 8) {
623                stringBuilder.append(Integer.toHexString((int) ((bits >> i) & 0xff)));
624            }
625            bits = uuid.getLeastSignificantBits();
626            for (int i = 64 - 8; i >= 0; i -= 8) {
627                stringBuilder.append(Integer.toHexString((int) ((bits >> i) & 0xff)));
628            }
629            return stringBuilder.toString();
630        }

632        private String pixelsForIndex(int index, final int charsPerPixel) {
633            final StringBuilder stringBuilder = new StringBuilder();
634            int highestPower = 1;
635            for (int i = 1; i < charsPerPixel; i++) {
636                highestPower *= writePalette.length;
637            }
638            for (int i = 0; i < charsPerPixel; i++) {
639                final int multiple = index / highestPower;
640                index -= (multiple * highestPower);
641                highestPower /= writePalette.length;
642                stringBuilder.append(writePalette[multiple]);
643            }
644            return stringBuilder.toString();
645        }

647        private String toColor(final int color) {
648            final String hex = Integer.toHexString(color);
649            if (hex.length() < 6) {
650                final char zeroes[] = new char[6 - hex.length()];
651                Arrays.fill(zeroes, '0');
652                return "#" + new String(zeroes) + hex;
653            } else {
654                return "#" + hex;
655            }
656        }

658        @Override
659        public void writeImage(final BufferedImage src, final OutputStream os, Map<String,Object> params)
660                throws ImageWriteException, IOException {
661            // make copy of params; we'll clear keys as we consume them.
662            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
663    
664            // clear format key.
665            if (params.containsKey(PARAM_KEY_FORMAT)) {
666                params.remove(PARAM_KEY_FORMAT);
667            }
668    
669            if (params.size() > 0) {
670                final Object firstKey = params.keySet().iterator().next();
671                throw new ImageWriteException("Unknown parameter: " + firstKey);
672            }
673    
674            final PaletteFactory paletteFactory = new PaletteFactory();
675            boolean hasTransparency = false;
676            if (paletteFactory.hasTransparency(src, 1)) {
677                hasTransparency = true;
678            }
679            SimplePalette palette = null;
680            int maxColors = writePalette.length;
681            int charsPerPixel = 1;
682            for (; palette == null;) {
683                palette = paletteFactory.makeExactRgbPaletteSimple(src,
684                        hasTransparency ? maxColors - 1 : maxColors);
685                if (palette == null) {
686                    maxColors *= writePalette.length;
687                    charsPerPixel++;
688                }
689            }
690            int colors = palette.length();
691            if (hasTransparency) {
692                ++colors;
693            }
694    
695            String line = "/* XPM */\n";
696            os.write(line.getBytes("US-ASCII"));
697            line = "static char *" + randomName() + "[] = {\n";
698            os.write(line.getBytes("US-ASCII"));
699            line = "\"" + src.getWidth() + " " + src.getHeight() + " " + colors
700                    + " " + charsPerPixel + "\",\n";
701            os.write(line.getBytes("US-ASCII"));
702    
703            for (int i = 0; i < colors; i++) {
704                String color;
705                if (i < palette.length()) {
706                    color = toColor(palette.getEntry(i));
707                } else {
708                    color = "None";
709                }
710                line = "\"" + pixelsForIndex(i, charsPerPixel) + " c " + color
711                        + "\",\n";
712                os.write(line.getBytes("US-ASCII"));
713            }
714    
715            String separator = "";
716            for (int y = 0; y < src.getHeight(); y++) {
717                os.write(separator.getBytes("US-ASCII"));
718                separator = ",\n";
719                line = "\"";
720                os.write(line.getBytes("US-ASCII"));
721                for (int x = 0; x < src.getWidth(); x++) {
722                    final int argb = src.getRGB(x, y);
723                    if ((argb & 0xff000000) == 0) {
724                        line = pixelsForIndex(palette.length(), charsPerPixel);
725                    } else {
726                        line = pixelsForIndex(
727                                palette.getPaletteIndex(0xffffff & argb),
728                                charsPerPixel);
729                    }
730                    os.write(line.getBytes("US-ASCII"));
731                }
732                line = "\"";
733                os.write(line.getBytes("US-ASCII"));
734            }
735    
736            line = "\n};\n";
737            os.write(line.getBytes("US-ASCII"));
738        }
