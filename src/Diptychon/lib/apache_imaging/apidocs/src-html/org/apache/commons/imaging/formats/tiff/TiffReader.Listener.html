046        private TiffHeader readTiffHeader(final ByteSource byteSource,
047                final FormatCompliance formatCompliance) throws ImageReadException,
048                IOException {
049            InputStream is = null;
050            try {
051                is = byteSource.getInputStream();
052                return readTiffHeader(is, formatCompliance);
053            } finally {
054                try {
055                    if (is != null) {
056                        is.close();
057                    }
058                } catch (final Exception e) {
059                    Debug.debug(e);
060                }
061            }
062        }

064        private ByteOrder getTiffByteOrder(final int byteOrderByte) throws ImageReadException {
065            if (byteOrderByte == 'I') {
066                return ByteOrder.INTEL;
067            } else if (byteOrderByte == 'M') {
068                return ByteOrder.MOTOROLA;
069            } else {
070                throw new ImageReadException("Invalid TIFF byte order " + (0xff & byteOrderByte));
071            }
072        }
073    
074        private TiffHeader readTiffHeader(final InputStream is,
075                final FormatCompliance formatCompliance) throws ImageReadException,
076                IOException {
077            final int BYTE_ORDER_1 = readByte("BYTE_ORDER_1", is, "Not a Valid TIFF File");
078            final int BYTE_ORDER_2 = readByte("BYTE_ORDER_2", is, "Not a Valid TIFF File");
079            if (BYTE_ORDER_1 != BYTE_ORDER_2) {
080                throw new ImageReadException("Byte Order bytes don't match (" + BYTE_ORDER_1
081                        + ", " + BYTE_ORDER_2 + ").");
082            }
083            final ByteOrder byteOrder = getTiffByteOrder(BYTE_ORDER_1);
084            setByteOrder(byteOrder);
085    
086            final int tiffVersion = read2Bytes("tiffVersion", is, "Not a Valid TIFF File");
087            if (tiffVersion != 42) {
088                throw new ImageReadException("Unknown Tiff Version: " + tiffVersion);
089            }
090    
091            final int offsetToFirstIFD = read4Bytes("offsetToFirstIFD", is,
092                    "Not a Valid TIFF File");
093    
094            skipBytes(is, offsetToFirstIFD - 8,
095                    "Not a Valid TIFF File: couldn't find IFDs");
096    
097            if (debug) {
098                System.out.println("");
099            }
100    
101            return new TiffHeader(byteOrder, tiffVersion, offsetToFirstIFD);
102        }

104        private void readDirectories(final ByteSource byteSource,
105                final FormatCompliance formatCompliance, final Listener listener)
106                throws ImageReadException, IOException {
107            final TiffHeader tiffHeader = readTiffHeader(byteSource, formatCompliance);
108            if (!listener.setTiffHeader(tiffHeader)) {
109                return;
110            }
111    
112            final int offset = tiffHeader.offsetToFirstIFD;
113            final int dirType = TiffDirectory.DIRECTORY_TYPE_ROOT;
114    
115            final List<Number> visited = new ArrayList<Number>();
116            readDirectory(byteSource, offset, dirType, formatCompliance, listener,
117                    visited);
118        }
119    
120        private boolean readDirectory(final ByteSource byteSource, final int offset,
121                final int dirType, final FormatCompliance formatCompliance, final Listener listener,
122                final List<Number> visited) throws ImageReadException, IOException {
123            final boolean ignoreNextDirectory = false;
124            return readDirectory(byteSource, offset, dirType, formatCompliance,
125                    listener, ignoreNextDirectory, visited);
126        }
127    
128        private boolean readDirectory(final ByteSource byteSource, final int offset,
129                final int dirType, final FormatCompliance formatCompliance, final Listener listener,
130                final boolean ignoreNextDirectory, final List<Number> visited)
131                throws ImageReadException, IOException {
132    
133            // Debug.debug();
134            // Debug.debug("dir offset", offset + " (0x" +
135            // Integer.toHexString(offset)
136            // + ")");
137            // Debug.debug("dir key", key);
138            // Debug.debug("dir visited", visited);
139            // Debug.debug("dirType", dirType);
140            // Debug.debug();
141    
142            if (visited.contains(offset)) {
143                return false;
144            }
145            visited.add(offset);
146    
147            InputStream is = null;
148            try {
149                if (offset >= byteSource.getLength()) {
150                    // Debug.debug("skipping invalid directory!");
151                    return true;
152                }
153    
154                is = byteSource.getInputStream();
155                skipBytes(is, offset);
156    
157                final List<TiffField> fields = new ArrayList<TiffField>();
158    
159                int entryCount;
160                try {
161                    entryCount = read2Bytes("DirectoryEntryCount", is,
162                            "Not a Valid TIFF File");
163                } catch (final IOException e) {
164                    if (strict) {
165                        throw e;
166                    } else {
167                        return true;
168                    }
169                }
170    
171                // Debug.debug("entryCount", entryCount);
172    
173                for (int i = 0; i < entryCount; i++) {
174                    final int tag = read2Bytes("Tag", is, "Not a Valid TIFF File");
175                    final int type = read2Bytes("Type", is, "Not a Valid TIFF File");
176                    final int length = read4Bytes("Length", is, "Not a Valid TIFF File");
177    
178                    // Debug.debug("tag*", tag + " (0x" + Integer.toHexString(tag)
179                    // + ")");
180    
181                    final byte valueOffsetBytes[] = readByteArray("ValueOffset", 4, is,
182                            "Not a Valid TIFF File");
183                    final int valueOffset = convertByteArrayToInt("ValueOffset",
184                            valueOffsetBytes);
185    
186                    if (tag == 0) {
187                        // skip invalid fields.
188                        // These are seen very rarely, but can have invalid value
189                        // lengths,
190                        // which can cause OOM problems.
191                        continue;
192                    }
193    
194                    // if (keepField(tag, tags))
195                    // {
196                    final TiffField field = new TiffField(tag, dirType, type, length,
197                            valueOffset, valueOffsetBytes, getByteOrder());
198                    field.setSortHint(i);
199    
200                    // Debug.debug("tagInfo", field.tagInfo);
201    
202                    try {
203                        field.fillInValue(byteSource);
204                    } catch (final TiffValueOutsideFileBoundsException valueOutsideFileException) {
205                        if (strict) {
206                            final IOException ioEx = new IOException();
207                            ioEx.initCause(valueOutsideFileException);
208                            throw ioEx;
209                        } else {
210                            // corrupt field, ignore it
211                            continue;
212                        }
213                    }
214    
215                    // Debug.debug("\t" + "value", field.getValueDescription());
216    
217                    fields.add(field);
218    
219                    if (!listener.addField(field)) {
220                        return true;
221                    }
222                }
223    
224                final int nextDirectoryOffset = read4Bytes("nextDirectoryOffset", is,
225                        "Not a Valid TIFF File");
226                // Debug.debug("nextDirectoryOffset", nextDirectoryOffset);
227    
228                final TiffDirectory directory = new TiffDirectory(dirType, fields,
229                        offset, nextDirectoryOffset);
230    
231                if (listener.readImageData()) {
232                    if (directory.hasTiffImageData()) {
233                        final TiffImageData rawImageData = getTiffRawImageData(
234                                byteSource, directory);
235                        directory.setTiffImageData(rawImageData);
236                    }
237                    if (directory.hasJpegImageData()) {
238                        final JpegImageData rawJpegImageData = getJpegRawImageData(
239                                byteSource, directory);
240                        directory.setJpegImageData(rawJpegImageData);
241                    }
242                }
243    
244                if (!listener.addDirectory(directory)) {
245                    return true;
246                }
247    
248                if (listener.readOffsetDirectories()) {
249                    final TagInfoLong[] offsetFields = {
250                            EXIF_TAG_EXIF_OFFSET,
251                            EXIF_TAG_GPSINFO,
252                            EXIF_TAG_INTEROP_OFFSET
253                    };
254                    final int[] directoryTypes = {
255                            TiffDirectory.DIRECTORY_TYPE_EXIF,
256                            TiffDirectory.DIRECTORY_TYPE_GPS,
257                            TiffDirectory.DIRECTORY_TYPE_INTEROPERABILITY
258                    };
259                    for (int i = 0; i < offsetFields.length; i++) {
260                        final TagInfoLong offsetField = offsetFields[i];
261                        if (directory.findField(offsetField) != null) {
262                            int subDirectoryOffset;
263                            int subDirectoryType;
264                            boolean subDirectoryRead = false;
265                            try {
266                                subDirectoryOffset = directory.getSingleFieldValue(offsetField);
267                                subDirectoryType = directoryTypes[i];
268                                subDirectoryRead = readDirectory(byteSource,
269                                        subDirectoryOffset, subDirectoryType,
270                                        formatCompliance, listener, true, visited);
271        
272                            } catch (final ImageReadException imageReadException) {
273                                if (strict) {
274                                    throw imageReadException;
275                                }
276                            }
277                            if (!subDirectoryRead) {
278                                fields.remove(offsetField);
279                            }
280                        }
281                    }
282                }
283    
284                if (!ignoreNextDirectory && directory.nextDirectoryOffset > 0) {
285                    // Debug.debug("next dir", directory.nextDirectoryOffset );
286                    readDirectory(byteSource, directory.nextDirectoryOffset,
287                            dirType + 1, formatCompliance, listener, visited);
288                }
289    
290                return true;
291            } finally {
292                try {
293                    if (is != null) {
294                        is.close();
295                    }
296                } catch (final Exception e) {
297                    Debug.debug(e);
298                }
299            }
300        }

314        private static class Collector implements Listener {
315            private TiffHeader tiffHeader = null;
316            private final List<TiffDirectory> directories = new ArrayList<TiffDirectory>();
317            private final List<TiffField> fields = new ArrayList<TiffField>();
318            private final boolean readThumbnails;
319    
320            public Collector() {
321                this(null);
322            }
323    
324            public Collector(final Map<String,Object> params) {
325                boolean readThumbnails = true;
326                if (params != null && params.containsKey(PARAM_KEY_READ_THUMBNAILS)) {
327                    readThumbnails = Boolean.TRUE.equals(params
328                            .get(PARAM_KEY_READ_THUMBNAILS));
329                }
330                this.readThumbnails = readThumbnails;
331            }
332    
333            public boolean setTiffHeader(final TiffHeader tiffHeader) {
334                this.tiffHeader = tiffHeader;
335                return true;
336            }
337    
338            public boolean addDirectory(final TiffDirectory directory) {
339                directories.add(directory);
340                return true;
341            }
342    
343            public boolean addField(final TiffField field) {
344                fields.add(field);
345                return true;
346            }
347    
348            public boolean readImageData() {
349                return readThumbnails;
350            }
351    
352            public boolean readOffsetDirectories() {
353                return true;
354            }
355    
356            public TiffContents getContents() {
357                return new TiffContents(tiffHeader, directories);
358            }
359        }

399        public TiffContents readFirstDirectory(final ByteSource byteSource, final Map<String,Object> params,
400                final boolean readImageData, final FormatCompliance formatCompliance)
401                throws ImageReadException, IOException {
402            final Collector collector = new FirstDirectoryCollector(readImageData);
403            read(byteSource, params, formatCompliance, collector);
404            final TiffContents contents = collector.getContents();
405            if (contents.directories.size() < 1) {
406                throw new ImageReadException(
407                        "Image did not contain any directories.");
408            }
409            return contents;
410        }

414                throws ImageReadException, IOException {

442        private TiffImageData getTiffRawImageData(final ByteSource byteSource,
443                final TiffDirectory directory) throws ImageReadException, IOException {
444    
445            final List<ImageDataElement> elements = directory
446                    .getTiffRawImageDataElements();
447            final TiffImageData.Data data[] = new TiffImageData.Data[elements.size()];
448    
449            if (byteSource instanceof ByteSourceFile) {
450                final ByteSourceFile bsf = (ByteSourceFile) byteSource;
451                for (int i = 0; i < elements.size(); i++) {
452                    final TiffDirectory.ImageDataElement element = elements.get(i);
453                    data[i] = new TiffImageData.ByteSourceData(element.offset,
454                            element.length, bsf);
455                }
456            } else {
457                for (int i = 0; i < elements.size(); i++) {
458                    final TiffDirectory.ImageDataElement element = elements.get(i);
459                    final byte bytes[] = byteSource.getBlock(element.offset,
460                            element.length);
461                    data[i] = new TiffImageData.Data(element.offset,
462                            element.length, bytes);
463                }
464            }
465    
466            if (directory.imageDataInStrips()) {
467                final TiffField rowsPerStripField = directory
468                        .findField(TiffTagConstants.TIFF_TAG_ROWS_PER_STRIP);
469                int rowsPerStrip;
470    
471                if (null != rowsPerStripField) {
472                    rowsPerStrip = rowsPerStripField.getIntValue();
473                } else {
474                    final TiffField imageHeight = directory
475                            .findField(TiffTagConstants.TIFF_TAG_IMAGE_LENGTH);
476                    /**
477                     * if rows per strip not present then rowsPerStrip is equal to
478                     * imageLength or an infinity value;
479                     */
480                    rowsPerStrip = imageHeight.getIntValue();
481                }
482    
483                return new TiffImageData.Strips(data, rowsPerStrip);
484            } else {
485                final TiffField tileWidthField = directory
486                        .findField(TiffTagConstants.TIFF_TAG_TILE_WIDTH);
487                if (null == tileWidthField) {
488                    throw new ImageReadException("Can't find tile width field.");
489                }
490                final int tileWidth = tileWidthField.getIntValue();
491    
492                final TiffField tileLengthField = directory
493                        .findField(TiffTagConstants.TIFF_TAG_TILE_LENGTH);
494                if (null == tileLengthField) {
495                    throw new ImageReadException("Can't find tile length field.");
496                }
497                final int tileLength = tileLengthField.getIntValue();
498    
499                return new TiffImageData.Tiles(data, tileWidth, tileLength);
500            }
501        }

503        private JpegImageData getJpegRawImageData(final ByteSource byteSource,
504                final TiffDirectory directory) throws ImageReadException, IOException {
505            final ImageDataElement element = directory.getJpegRawImageDataElement();
506            final int offset = element.offset;
507            int length = element.length;
508            // In case the length is not correct, adjust it and check if the last read byte actually is the end of the image
509            if (offset + length > byteSource.getLength()) {
510                length = (int) byteSource.getLength() - offset;
511            }
512            final byte data[] = byteSource.getBlock(offset, length);
513            // check if the last read byte is actually the end of the image data
514            if (length < 2 ||
515                    (((data[data.length - 2] & 0xff) << 8) | (data[data.length - 1] & 0xff)) != JpegConstants.EOIMarker) {
516                throw new ImageReadException("JPEG EOI marker could not be found at expected location");
517            }
518            return new JpegImageData(offset, length, data);
519        }
