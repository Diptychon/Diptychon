096        private IcnsType(final String type, final int width, final int height, final int bitsPerPixel,
097                final boolean hasMask) {
098            this.type = typeAsInt(type);
099            this.width = width;
100            this.height = height;
101            this.bitsPerPixel = bitsPerPixel;
102            this.hasMask = hasMask;
103        }
104    
105        public int getType() {
106            return type;
107        }
108    
109        public int getWidth() {
110            return width;
111        }
112    
113        public int getHeight() {
114            return height;
115        }
116    
117        public int getBitsPerPixel() {
118            return bitsPerPixel;
119        }
120    
121        public boolean hasMask() {
122            return hasMask;
123        }
124    
125        @Override
126        public String toString() {
127            return getClass().getName() + "[" + "width=" + width + "," + "height="
128                    + height + "," + "bpp=" + bitsPerPixel + "," + "hasMask="
129                    + hasMask + "]";
130        }
131    
132        public static IcnsType findAnyType(final int type) {
133            for (final IcnsType allImageType : allImageTypes) {
134                if (allImageType.getType() == type) {
135                    return allImageType;
136                }
137            }
138            for (final IcnsType allMaskType : allMaskTypes) {
139                if (allMaskType.getType() == type) {
140                    return allMaskType;
141                }
142            }
143            return null;
144        }
145    
146        public static IcnsType findImageType(final int type) {
147            for (final IcnsType allImageType : allImageTypes) {
148                if (allImageType.getType() == type) {
149                    return allImageType;
150                }
151            }
152            return null;
153        }
154    
155        public static IcnsType find8BPPMaskType(final IcnsType imageType) {
156            for (final IcnsType allMaskType : allMaskTypes) {
157                if (allMaskType.getBitsPerPixel() == 8
158                        && allMaskType.getWidth() == imageType.getWidth()
159                        && allMaskType.getHeight() == imageType.getHeight()) {
160                    return allMaskType;
161                }
162            }
163            return null;
164        }
165    
166        public static IcnsType find1BPPMaskType(final IcnsType imageType) {
167            for (final IcnsType allMaskType : allMaskTypes) {
168                if (allMaskType.getBitsPerPixel() == 1
169                        && allMaskType.getWidth() == imageType.getWidth()
170                        && allMaskType.getHeight() == imageType.getHeight()) {
171                    return allMaskType;
172                }
173            }
174            return null;
175        }
176    
177        public static int typeAsInt(final String type) {
178            byte[] bytes = null;
179            try {
180                bytes = type.getBytes("US-ASCII");
181            } catch (final UnsupportedEncodingException cannotHappen) {
182                throw new IllegalArgumentException("Your Java doesn't support US-ASCII");
183            }
184            if (bytes.length != 4) {
185                throw new IllegalArgumentException("Invalid ICNS type");
186            }
187            return ((0xff & bytes[0]) << 24) | ((0xff & bytes[1]) << 16)
188                    | ((0xff & bytes[2]) << 8) | (0xff & bytes[3]);
189        }
190    
191        public static String describeType(final int type) {
192            final byte[] bytes = new byte[4];
193            bytes[0] = (byte) (0xff & (type >> 24));
194            bytes[1] = (byte) (0xff & (type >> 16));
195            bytes[2] = (byte) (0xff & (type >> 8));
196            bytes[3] = (byte) (0xff & type);
197            try {
198                return new String(bytes, "US-ASCII");
199            } catch (final UnsupportedEncodingException cannotHappen) {
200            }
201            return null;
202        }
