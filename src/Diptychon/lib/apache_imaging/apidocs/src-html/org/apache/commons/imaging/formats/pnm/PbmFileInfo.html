017    package org.apache.commons.imaging.formats.pnm;
018    
019    import java.io.IOException;
020    import java.io.InputStream;
021    
022    import org.apache.commons.imaging.ImageFormat;
023    import org.apache.commons.imaging.ImageInfo;
024    
025    public class PbmFileInfo extends FileInfo {
026        public PbmFileInfo(final int width, final int height, final boolean RAWBITS) {
027            super(width, height, RAWBITS);
028        }
029        
030        @Override
031        public boolean hasAlpha() {
032            return false;
033        }
034    
035        @Override
036        public int getNumComponents() {
037            return 1;
038        }
039    
040        @Override
041        public int getBitDepth() {
042            return 1;
043        }
044    
045        @Override
046        public ImageFormat getImageType() {
047            return ImageFormat.IMAGE_FORMAT_PBM;
048        }
049    
050        @Override
051        public int getColorType() {
052            return ImageInfo.COLOR_TYPE_BW;
053        }
054    
055        @Override
056        public String getImageTypeDescription() {
057            return "PBM: portable bitmap fileformat";
058        }
059    
060        @Override
061        public String getMIMEType() {
062            return "image/x-portable-bitmap";
063        }
064    
065        @Override
066        protected void newline() {
067            bitcache = 0;
068            bits_in_cache = 0;
069        }
070    
071        private int bitcache = 0;
072        private int bits_in_cache = 0;
073    
074        @Override
075        public int getRGB(final InputStream is) throws IOException {
076            if (bits_in_cache < 1) {
077                final int bits = is.read();
078                if (bits < 0) {
079                    throw new IOException("PBM: Unexpected EOF");
080                }
081                bitcache = 0xff & bits;
082                bits_in_cache += 8;
083            }
084    
085            final int bit = 0x1 & (bitcache >> 7);
086            bitcache <<= 1;
087            bits_in_cache--;
088    
089            if (bit == 0) {
090                return 0xffffffff;
091            }
092            if (bit == 1) {
093                return 0xff000000;
094            }
095            throw new IOException("PBM: bad bit: " + bit);
096        }
097    
098        @Override
099        public int getRGB(final WhiteSpaceReader wsr) throws IOException {
100            final int bit = Integer.parseInt(wsr.readtoWhiteSpace());
101            if (bit == 0) {
102                return 0xff000000;
103            }
104            if (bit == 1) {
105                return 0xffffffff;
106            }
107            throw new IOException("PBM: bad bit: " + bit);
108        }
109    
110    }
