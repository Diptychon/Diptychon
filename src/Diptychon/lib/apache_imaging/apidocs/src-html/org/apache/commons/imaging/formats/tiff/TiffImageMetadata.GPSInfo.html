063        private static final Map<Object, Integer> countTags(final List<TagInfo> tags) {
064            final Map<Object, Integer> map = new Hashtable<Object, Integer>();
065    
066            for (int i = 0; i < tags.size(); i++) {
067                final TagInfo tag = tags.get(i);
068    
069                final Integer count = map.get(tag.tag);
070                if (count == null) {
071                    map.put(tag.tag, 1);
072                } else {
073                    map.put(tag.tag, count + 1);
074                }
075            }
076    
077            return map;
078        }

129            public TiffOutputDirectory getOutputDirectory(final ByteOrder byteOrder)
130                    throws ImageWriteException {
131                try {
132                    final TiffOutputDirectory dstDir = new TiffOutputDirectory(type,
133                            byteOrder);
134    
135                    final List<? extends IImageMetadataItem> entries = getItems();
136                    for (int i = 0; i < entries.size(); i++) {
137                        final TiffImageMetadata.Item item = (TiffImageMetadata.Item) entries
138                                .get(i);
139                        final TiffField srcField = item.getTiffField();
140    
141                        if (null != dstDir.findField(srcField.tag)) {
142                            // ignore duplicate tags in a directory.
143                            continue;
144                        } else if (srcField.tagInfo.isOffset()) {
145                            // ignore offset fields.
146                            continue;
147                        }
148    
149                        final TagInfo tagInfo = srcField.tagInfo;
150                        final FieldType fieldType = srcField.fieldType;
151                        // byte bytes[] = srcField.fieldType.getRawBytes(srcField);
152    
153                        // Debug.debug("tagInfo", tagInfo);
154    
155                        final Object value = srcField.getValue();
156    
157                        // Debug.debug("value", Debug.getType(value));
158    
159                        final byte bytes[] = tagInfo.encodeValue(fieldType, value,
160                                byteOrder);
161    
162                        // if (tagInfo.isUnknown())
163                        // Debug.debug(
164                        // "\t" + "unknown tag(0x"
165                        // + Integer.toHexString(srcField.tag)
166                        // + ") bytes", bytes);
167    
168                        final int count = bytes.length / fieldType.length;
169                        final TiffOutputField dstField = new TiffOutputField(
170                                srcField.tag, tagInfo, fieldType, count, bytes);
171                        dstField.setSortHint(srcField.getSortHint());
172                        dstDir.add(dstField);
173                    }
174    
175                    dstDir.setTiffImageData(getTiffImageData());
176                    dstDir.setJpegImageData(getJpegImageData());
177    
178                    return dstDir;
179                } catch (final ImageReadException e) {
180                    throw new ImageWriteException(e.getMessage(), e);
181                }
182            }

218        public TiffOutputSet getOutputSet() throws ImageWriteException {
219            final ByteOrder byteOrder = contents.header.byteOrder;
220            final TiffOutputSet result = new TiffOutputSet(byteOrder);
221    
222            final List<? extends IImageMetadataItem> srcDirs = getDirectories();
223            for (int i = 0; i < srcDirs.size(); i++) {
224                final TiffImageMetadata.Directory srcDir = (TiffImageMetadata.Directory) srcDirs
225                        .get(i);
226    
227                if (null != result.findDirectory(srcDir.type)) {
228                    // Certain cameras right directories more than once.
229                    // This is a bug.
230                    // Ignore second directory of a given type.
231                    continue;
232                }
233    
234                final TiffOutputDirectory outputDirectory = srcDir
235                        .getOutputDirectory(byteOrder);
236                result.addDirectory(outputDirectory);
237            }
238    
239            return result;
240        }

246        public TiffField findField(final TagInfo tagInfo, final boolean exactDirectoryMatch)
247                throws ImageReadException {
248            // Please keep this method in sync with TiffField's getTag()
249            final Integer tagCount = tagCounts.get(tagInfo.tag);
250            final int tagsMatching = tagCount == null ? 0 : tagCount.intValue();
251    
252            final List<? extends IImageMetadataItem> directories = getDirectories();
253            if (exactDirectoryMatch
254                    || tagInfo.directoryType != TiffDirectoryType.EXIF_DIRECTORY_UNKNOWN) {
255                for (int i = 0; i < directories.size(); i++) {
256                    final Directory directory = (Directory) directories.get(i);
257                    if (directory.type == tagInfo.directoryType.directoryType) {
258                        final TiffField field = directory.findField(tagInfo);
259                        if (field != null) {
260                            return field;
261                        }
262                    }
263                }
264                if (exactDirectoryMatch || tagsMatching > 1) {
265                    return null;
266                }
267                for (int i = 0; i < directories.size(); i++) {
268                    final Directory directory = (Directory) directories.get(i);
269                    if (tagInfo.directoryType.isImageDirectory()
270                            && directory.type >= 0) {
271                        final TiffField field = directory.findField(tagInfo);
272                        if (field != null) {
273                            return field;
274                        }
275                    } else if (!tagInfo.directoryType.isImageDirectory()
276                            && directory.type < 0) {
277                        final TiffField field = directory.findField(tagInfo);
278                        if (field != null) {
279                            return field;
280                        }
281                    }
282                }
283            }
284    
285            for (int i = 0; i < directories.size(); i++) {
286                final Directory directory = (Directory) directories.get(i);
287                final TiffField field = directory.findField(tagInfo);
288                if (field != null) {
289                    return field;
290                }
291            }
292    
293            return null;
294        }

473        public GPSInfo getGPS() throws ImageReadException {
474            final TiffDirectory gpsDirectory = findDirectory(DIRECTORY_TYPE_GPS);
475            if (null == gpsDirectory) {
476                return null;
477            }
478    
479            // more specific example of how to access GPS values.
480            final TiffField latitudeRefField = gpsDirectory
481                    .findField(GpsTagConstants.GPS_TAG_GPS_LATITUDE_REF);
482            final TiffField latitudeField = gpsDirectory
483                    .findField(GpsTagConstants.GPS_TAG_GPS_LATITUDE);
484            final TiffField longitudeRefField = gpsDirectory
485                    .findField(GpsTagConstants.GPS_TAG_GPS_LONGITUDE_REF);
486            final TiffField longitudeField = gpsDirectory
487                    .findField(GpsTagConstants.GPS_TAG_GPS_LONGITUDE);
488    
489            if (latitudeRefField == null || latitudeField == null
490                    || longitudeRefField == null || longitudeField == null) {
491                return null;
492            }
493    
494            // all of these values are strings.
495            final String latitudeRef = latitudeRefField.getStringValue();
496            final RationalNumber latitude[] = (RationalNumber[]) latitudeField.getValue();
497            final String longitudeRef = longitudeRefField.getStringValue();
498            final RationalNumber longitude[] = (RationalNumber[]) longitudeField
499                    .getValue();
500    
501            if (latitude.length != 3 || longitude.length != 3) {
502                throw new ImageReadException(
503                        "Expected three values for latitude and longitude.");
504            }
505    
506            final RationalNumber latitudeDegrees = latitude[0];
507            final RationalNumber latitudeMinutes = latitude[1];
508            final RationalNumber latitudeSeconds = latitude[2];
509    
510            final RationalNumber longitudeDegrees = longitude[0];
511            final RationalNumber longitudeMinutes = longitude[1];
512            final RationalNumber longitudeSeconds = longitude[2];
513    
514            return new GPSInfo(latitudeRef, longitudeRef, latitudeDegrees,
515                    latitudeMinutes, latitudeSeconds, longitudeDegrees,
516                    longitudeMinutes, longitudeSeconds);
517        }

519        public static class GPSInfo {
520            public final String latitudeRef;
521            public final String longitudeRef;
522    
523            public final RationalNumber latitudeDegrees;
524            public final RationalNumber latitudeMinutes;
525            public final RationalNumber latitudeSeconds;
526            public final RationalNumber longitudeDegrees;
527            public final RationalNumber longitudeMinutes;
528            public final RationalNumber longitudeSeconds;
529    
530            public GPSInfo(final String latitudeRef, final String longitudeRef,
531                    final RationalNumber latitudeDegrees,
532                    final RationalNumber latitudeMinutes,
533                    final RationalNumber latitudeSeconds,
534                    final RationalNumber longitudeDegrees,
535                    final RationalNumber longitudeMinutes,
536                    final RationalNumber longitudeSeconds) {
537                this.latitudeRef = latitudeRef;
538                this.longitudeRef = longitudeRef;
539                this.latitudeDegrees = latitudeDegrees;
540                this.latitudeMinutes = latitudeMinutes;
541                this.latitudeSeconds = latitudeSeconds;
542                this.longitudeDegrees = longitudeDegrees;
543                this.longitudeMinutes = longitudeMinutes;
544                this.longitudeSeconds = longitudeSeconds;
545            }
546    
547            @Override
548            public String toString() {
549                // This will format the gps info like so:
550                //
551                // latitude: 8 degrees, 40 minutes, 42.2 seconds S
552                // longitude: 115 degrees, 26 minutes, 21.8 seconds E
553    
554                final StringBuilder result = new StringBuilder();
555                result.append("[GPS. ");
556                result.append("Latitude: " + latitudeDegrees.toDisplayString()
557                        + " degrees, " + latitudeMinutes.toDisplayString()
558                        + " minutes, " + latitudeSeconds.toDisplayString()
559                        + " seconds " + latitudeRef);
560                result.append(", Longitude: " + longitudeDegrees.toDisplayString()
561                        + " degrees, " + longitudeMinutes.toDisplayString()
562                        + " minutes, " + longitudeSeconds.toDisplayString()
563                        + " seconds " + longitudeRef);
564                result.append("]");
565    
566                return result.toString();
567            }
568    
569            public double getLongitudeAsDegreesEast() throws ImageReadException {
570                final double result = longitudeDegrees.doubleValue()
571                        + (longitudeMinutes.doubleValue() / 60.0)
572                        + (longitudeSeconds.doubleValue() / 3600.0);
573    
574                if (longitudeRef.trim().equalsIgnoreCase("e")) {
575                    return result;
576                } else if (longitudeRef.trim().equalsIgnoreCase("w")) {
577                    return -result;
578                } else {
579                    throw new ImageReadException("Unknown longitude ref: \""
580                            + longitudeRef + "\"");
581                }
582            }
583    
584            public double getLatitudeAsDegreesNorth() throws ImageReadException {
585                final double result = latitudeDegrees.doubleValue()
586                        + (latitudeMinutes.doubleValue() / 60.0)
587                        + (latitudeSeconds.doubleValue() / 3600.0);
588    
589                if (latitudeRef.trim().equalsIgnoreCase("n")) {
590                    return result;
591                } else if (latitudeRef.trim().equalsIgnoreCase("s")) {
592                    return -result;
593                } else {
594                    throw new ImageReadException("Unknown latitude ref: \""
595                            + latitudeRef + "\"");
596                }
597            }
