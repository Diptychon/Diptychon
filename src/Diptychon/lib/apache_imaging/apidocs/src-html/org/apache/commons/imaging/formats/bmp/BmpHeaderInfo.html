017    package org.apache.commons.imaging.formats.bmp;
018    
019    public class BmpHeaderInfo {
020        // ?BM? - Windows 3.1x, 95, NT, ?
021        // ?BA? - OS/2 Bitmap Array
022        // ?CI? - OS/2 Color Icon?
023        // CP? - OS/2 Color Pointer
024        // ?IC? - OS/2 Icon
025        // ?PT? - OS/2 Pointer
026        public final byte identifier1;
027        public final byte identifier2;
028    
029        public final int fileSize;
030        public final int reserved;
031        public final int bitmapDataOffset;
032    
033        public final int bitmapHeaderSize;
034        public final int width;
035        public final int height;
036        public final int planes;
037        public final int bitsPerPixel;
038        public final int compression;
039        public final int bitmapDataSize;
040        public final int hResolution;
041        public final int vResolution;
042        public final int colorsUsed;
043        public final int colorsImportant;
044    
045        public final int redMask;
046        public final int greenMask;
047        public final int blueMask;
048        public final int alphaMask;
049    
050        public final int colorSpaceType;
051        public final ColorSpace colorSpace;
052        public final int gammaRed;
053        public final int gammaGreen;
054        public final int gammaBlue;
055        public final int intent;
056        public final int profileData;
057        public final int profileSize;
058        public final int reservedV5;
059    
060        public static class ColorSpaceCoordinate {
061            public int x, y, z;
062        }
063    
064        public static class ColorSpace {
065            public ColorSpaceCoordinate red;
066            public ColorSpaceCoordinate green;
067            public ColorSpaceCoordinate blue;
068        }
069    
070        public BmpHeaderInfo(final byte identifier1, final byte identifier2, final int fileSize,
071                final int reserved, final int bitmapDataOffset, final int bitmapHeaderSize,
072                final int width, final int height, final int planes, final int bitsPerPixel,
073                final int compression, final int bitmapDataSize, final int hResolution,
074                final int vResolution, final int colorsUsed, final int colorsImportant, final int redMask,
075                final int greenMask, final int blueMask, final int alphaMask, final int colorSpaceType,
076                final ColorSpace colorSpace, final int gammaRed, final int gammaGreen, final int gammaBlue,
077                final int intent, final int profileData, final int profileSize, final int reservedV5) {
078            this.identifier1 = identifier1;
079            this.identifier2 = identifier2;
080            this.fileSize = fileSize;
081            this.reserved = reserved;
082            this.bitmapDataOffset = bitmapDataOffset;
083    
084            this.bitmapHeaderSize = bitmapHeaderSize;
085            this.width = width;
086            this.height = height;
087            this.planes = planes;
088            this.bitsPerPixel = bitsPerPixel;
089            this.compression = compression;
090            this.bitmapDataSize = bitmapDataSize;
091            this.hResolution = hResolution;
092            this.vResolution = vResolution;
093            this.colorsUsed = colorsUsed;
094            this.colorsImportant = colorsImportant;
095    
096            this.redMask = redMask;
097            this.greenMask = greenMask;
098            this.blueMask = blueMask;
099            this.alphaMask = alphaMask;
100            this.colorSpaceType = colorSpaceType;
101            this.colorSpace = colorSpace;
102            this.gammaRed = gammaRed;
103            this.gammaGreen = gammaGreen;
104            this.gammaBlue = gammaBlue;
105            this.intent = intent;
106            this.profileData = profileData;
107            this.profileSize = profileSize;
108            this.reservedV5 = reservedV5;
109        }
110    
111    }
