016    package org.apache.commons.imaging.formats.jpeg.decoder;
017    
018    import java.io.IOException;
019    import java.io.InputStream;
020    
021    import org.apache.commons.imaging.ImageReadException;
022    import org.apache.commons.imaging.formats.jpeg.JpegConstants;
023    
024    public class JpegInputStream {
025        // Figure F.18, F.2.2.5, page 111 of ITU-T T.81
026        private final InputStream is;
027        private int cnt = 0;
028        private int b;
029    
030        public JpegInputStream(final InputStream is) {
031            this.is = is;
032        }
033    
034        public int nextBit() throws IOException, ImageReadException {
035            if (cnt == 0) {
036                b = is.read();
037                if (b < 0) {
038                    throw new ImageReadException("Premature End of File");
039                }
040                cnt = 8;
041                if (b == 0xff) {
042                    final int b2 = is.read();
043                    if (b2 < 0) {
044                        throw new ImageReadException("Premature End of File");
045                    }
046                    if (b2 != 0) {
047                        if (b2 == (0xff & JpegConstants.DNLMarker)) {
048                            throw new ImageReadException("DNL not yet supported");
049                        } else {
050                            throw new ImageReadException("Invalid marker found "
051                                    + "in entropy data");
052                        }
053                    }
054                }
055            }
056            final int bit = (b >> 7) & 0x1;
057            cnt--;
058            b <<= 1;
059            return bit;
060        }
061    }
