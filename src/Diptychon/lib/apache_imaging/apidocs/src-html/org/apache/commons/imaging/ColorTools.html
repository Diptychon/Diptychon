043        public BufferedImage correctImage(final BufferedImage src, final File file)
044                throws ImageReadException, IOException {
045            final ICC_Profile icc = Imaging.getICCProfile(file);
046            if (icc == null) {
047                return src;
048            }
049    
050            final ICC_ColorSpace cs = new ICC_ColorSpace(icc);
051    
052            final BufferedImage dst = convertFromColorSpace(src, cs);
053            return dst;
054        }

056        public BufferedImage relabelColorSpace(final BufferedImage bi, final ICC_Profile profile)
057                throws ImagingOpException {
058            final ICC_ColorSpace cs = new ICC_ColorSpace(profile);
059    
060            return relabelColorSpace(bi, cs);
061        }

063        public BufferedImage relabelColorSpace(final BufferedImage bi, final ColorSpace cs)
064                throws ImagingOpException {
065            // This does not do the conversion. It tries to relabel the
066            // BufferedImage
067            // with its actual (presumably correct) Colorspace.
068            // use this when the image is mislabeled, presumably having been
069            // wrongly assumed to be sRGB
070    
071            final ColorModel cm = deriveColorModel(bi, cs);
072    
073            return relabelColorSpace(bi, cm);
074    
075        }
076    
077        public BufferedImage relabelColorSpace(final BufferedImage bi, final ColorModel cm)
078                throws ImagingOpException {
079            // This does not do the conversion. It tries to relabel the
080            // BufferedImage
081            // with its actual (presumably correct) Colorspace.
082            // use this when the image is mislabeled, presumably having been
083            // wrongly assumed to be sRGB
084    
085            final BufferedImage result = new BufferedImage(cm, bi.getRaster(), false,
086                    null);
087    
088            return result;
089        }

092                throws ImagingOpException {
093            // boolean hasAlpha = (bi.getAlphaRaster() != null);
094            return deriveColorModel(bi, cs, false);
095        }

097        public ColorModel deriveColorModel(final BufferedImage bi, final ColorSpace cs,
098                final boolean force_no_alpha) throws ImagingOpException {
099            return deriveColorModel(bi.getColorModel(), cs, force_no_alpha);

102        public ColorModel deriveColorModel(final ColorModel old_cm, final ColorSpace cs,
103                final boolean force_no_alpha) throws ImagingOpException {
104    
105            if (old_cm instanceof ComponentColorModel) {
106                final ComponentColorModel ccm = (ComponentColorModel) old_cm;
107                // ColorSpace cs = ColorSpace.getInstance(ColorSpace.CS_sRGB);
108                if (force_no_alpha) {
109                    return new ComponentColorModel(cs, false, false,
110                            ComponentColorModel.OPAQUE, ccm.getTransferType());
111                } else {
112                    return new ComponentColorModel(cs, ccm.hasAlpha(),
113                            ccm.isAlphaPremultiplied(), ccm.getTransparency(),
114                            ccm.getTransferType());
115                }
116            } else if (old_cm instanceof DirectColorModel) {
117                final DirectColorModel dcm = (DirectColorModel) old_cm;
118    
119                final int old_mask = dcm.getRedMask() | dcm.getGreenMask()
120                        | dcm.getBlueMask() | dcm.getAlphaMask();
121    
122                final int old_bits = count_bits_in_mask(old_mask);
123    
124                return new DirectColorModel(cs, old_bits, dcm.getRedMask(),
125                        dcm.getGreenMask(), dcm.getBlueMask(), dcm.getAlphaMask(),
126                        dcm.isAlphaPremultiplied(), dcm.getTransferType());
127            }
128            // else if (old_cm instanceof PackedColorModel)
129            // {
130            // PackedColorModel pcm = (PackedColorModel) old_cm;
131            //
132            // // int old_mask = dcm.getRedMask() | dcm.getGreenMask()
133            // // | dcm.getBlueMask() | dcm.getAlphaMask();
134            //
135            // int old_masks[] = pcm.getMasks();
136            // // System.out.println("old_mask: " + old_mask);
137            // int old_bits = count_bits_in_mask(old_masks);
138            // // System.out.println("old_bits: " + old_bits);
139            //
140            // // PackedColorModel(ColorSpace space, int bits, int rmask, int gmask,
141            // int bmask, int amask, boolean isAlphaPremultiplied, int trans, int
142            // transferType)
143            // cm = new PackedColorModel(cs, old_bits, pcm.getMasks(),
144            //
145            // pcm.isAlphaPremultiplied(), pcm.getTransparency(), pcm
146            // .getTransferType());
147            // }
148    
149            throw new ImagingOpException("Could not clone unknown ColorModel Type.");
150        }

152        private int count_bits_in_mask(int i) {
153            int count = 0;
154            while (i != 0) {
155                count += (i & 1);
156                // uses the unsigned version of java's right shift operator,
157                // so that left hand bits are zeroed.
158                i >>>= 1;
159            }
160            return count;
161        }

163        public BufferedImage convertToColorSpace(final BufferedImage bi, final ColorSpace to) {
164            final ColorSpace from = bi.getColorModel().getColorSpace();
165    
166            final RenderingHints hints = new RenderingHints(RenderingHints.KEY_RENDERING,
167                    RenderingHints.VALUE_RENDER_QUALITY);
168            hints.put(RenderingHints.KEY_COLOR_RENDERING,
169                    RenderingHints.VALUE_COLOR_RENDER_QUALITY);
170            hints.put(RenderingHints.KEY_DITHERING,
171                    RenderingHints.VALUE_DITHER_ENABLE);
172    
173            final ColorConvertOp op = new ColorConvertOp(from, to, hints);
174    
175            BufferedImage result = op.filter(bi, null);
176    
177            result = relabelColorSpace(result, to);
178    
179            return result;
180        }

182        public BufferedImage convertTosRGB(final BufferedImage bi) {
183            ColorSpace cs_sRGB = ColorSpace.getInstance(ColorSpace.CS_sRGB);
184    
185            final ColorModel srgbCM = ColorModel.getRGBdefault();
186            cs_sRGB = srgbCM.getColorSpace();
187    
188            return convertToColorSpace(bi, cs_sRGB);
189        }
190    
191        protected BufferedImage convertFromColorSpace(final BufferedImage bi,
192                final ColorSpace from) {
193            ColorSpace cs_sRGB;
194    
195            final ColorModel srgbCM = ColorModel.getRGBdefault();
196            cs_sRGB = srgbCM.getColorSpace();
197    
198            return convertBetweenColorSpaces(bi, from, cs_sRGB);
199    
200        }
201    
202        public BufferedImage convertBetweenICCProfiles(final BufferedImage bi,
203                final ICC_Profile from, final ICC_Profile to) {
204            final ICC_ColorSpace cs_from = new ICC_ColorSpace(from);
205            final ICC_ColorSpace cs_to = new ICC_ColorSpace(to);
206    
207            return convertBetweenColorSpaces(bi, cs_from, cs_to);
208        }
209    
210        public BufferedImage convertToICCProfile(final BufferedImage bi, final ICC_Profile to) {
211            final ICC_ColorSpace cs_to = new ICC_ColorSpace(to);
212    
213            return convertToColorSpace(bi, cs_to);
214        }

244        public BufferedImage convertBetweenColorSpaces(BufferedImage bi,
245                final ColorSpace from, final ColorSpace to) {
246            final RenderingHints hints = new RenderingHints(RenderingHints.KEY_RENDERING,
247                    RenderingHints.VALUE_RENDER_QUALITY);
248            hints.put(RenderingHints.KEY_COLOR_RENDERING,
249                    RenderingHints.VALUE_COLOR_RENDER_QUALITY);
250            hints.put(RenderingHints.KEY_DITHERING,
251                    RenderingHints.VALUE_DITHER_ENABLE);
252    
253            final ColorConvertOp op = new ColorConvertOp(from, to, hints);
254    
255            bi = relabelColorSpace(bi, from);
256    
257            BufferedImage result = op.filter(bi, null);
258    
259            result = relabelColorSpace(result, to);
260    
261            return result;
262        }
