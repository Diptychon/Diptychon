098        public static final String getChunkTypeName(final int chunkType) {
099            final StringBuilder result = new StringBuilder();
100            result.append((char) (0xff & (chunkType >> 24)));
101            result.append((char) (0xff & (chunkType >> 16)));
102            result.append((char) (0xff & (chunkType >> 8)));
103            result.append((char) (0xff & (chunkType >> 0)));
104            return result.toString();
105        }

110        public List<String> getChuckTypes(final InputStream is)
111                throws ImageReadException, IOException {
112            final List<PngChunk> chunks = readChunks(is, null, false);
113            final List<String> chunkTypes = new ArrayList<String>();
114            for (int i = 0; i < chunks.size(); i++) {
115                final PngChunk chunk = chunks.get(i);
116                chunkTypes.add(getChunkTypeName(chunk.chunkType));
117            }
118            return chunkTypes;

144        private boolean keepChunk(final int ChunkType, final int chunkTypes[]) {
145            // System.out.println("keepChunk: ");
146            if (chunkTypes == null) {
147                return true;
148            }
149    
150            for (final int chunkType2 : chunkTypes) {
151                if (chunkType2 == ChunkType) {
152                    return true;
153                }
154            }
155            return false;
156        }

158        private List<PngChunk> readChunks(final InputStream is, final int chunkTypes[],
159                final boolean returnAfterFirst) throws ImageReadException, IOException {
160            final List<PngChunk> result = new ArrayList<PngChunk>();
161    
162            while (true) {
163                if (debug) {
164                    System.out.println("");
165                }
166    
167                final int length = read4Bytes("Length", is, "Not a Valid PNG File");
168                final int chunkType = read4Bytes("ChunkType", is, "Not a Valid PNG File");
169    
170                if (debug) {
171                    printCharQuad("ChunkType", chunkType);
172                    debugNumber("Length", length, 4);
173                }
174                final boolean keep = keepChunk(chunkType, chunkTypes);
175    
176                byte bytes[] = null;
177                if (keep) {
178                    bytes = readByteArray("Chunk Data", length, is,
179                            "Not a Valid PNG File: Couldn't read Chunk Data.");
180                } else {
181                    skipBytes(is, length, "Not a Valid PNG File");
182                }
183    
184                if (debug) {
185                    if (bytes != null) {
186                        debugNumber("bytes", bytes.length, 4);
187                    }
188                }
189    
190                final int CRC = read4Bytes("CRC", is, "Not a Valid PNG File");
191    
192                if (keep) {
193                    if (chunkType == iCCP) {
194                        result.add(new PngChunkIccp(length, chunkType, CRC, bytes));
195                    } else if (chunkType == tEXt) {
196                        result.add(new PngChunkText(length, chunkType, CRC, bytes));
197                    } else if (chunkType == zTXt) {
198                        result.add(new PngChunkZtxt(length, chunkType, CRC, bytes));
199                    } else if (chunkType == IHDR) {
200                        result.add(new PngChunkIhdr(length, chunkType, CRC, bytes));
201                    } else if (chunkType == PLTE) {
202                        result.add(new PngChunkPlte(length, chunkType, CRC, bytes));
203                    } else if (chunkType == pHYs) {
204                        result.add(new PngChunkPhys(length, chunkType, CRC, bytes));
205                    } else if (chunkType == IDAT) {
206                        result.add(new PngChunkIdat(length, chunkType, CRC, bytes));
207                    } else if (chunkType == gAMA) {
208                        result.add(new PngChunkGama(length, chunkType, CRC, bytes));
209                    } else if (chunkType == iTXt) {
210                        result.add(new PngChunkItxt(length, chunkType, CRC, bytes));
211                    } else { 
212                        result.add(new PngChunk(length, chunkType, CRC, bytes));
213                    }
214    
215                    if (returnAfterFirst) {
216                        return result;
217                    }
218                }
219    
220                if (chunkType == IEND) {
221                    break;
222                }
223    
224            }
225    
226            return result;
227    
228        }

237        private List<PngChunk> readChunks(final ByteSource byteSource, final int chunkTypes[],
238                final boolean returnAfterFirst) throws ImageReadException, IOException {
239            InputStream is = null;
240    
241            try {
242                is = byteSource.getInputStream();
243    
244                readSignature(is);
245    
246                return readChunks(is, chunkTypes, returnAfterFirst);
247            } finally {
248                try {
249                    if (is != null) {
250                        is.close();
251                    }
252                } catch (final Exception e) {
253                    Debug.debug(e);
254                }
255            }
256        }

329        private boolean isGrayscale(final int colorType) throws ImageReadException {
330            // Color type is a single-byte integer that describes the interpretation
331            // of the
332            // image data. Color type codes represent sums of the following values:
333            // 1 (palette used), 2 (color used), and 4 (alpha channel used).
334            // Valid values are 0, 2, 3, 4, and 6.
335            //
336            // Bit depth restrictions for each color type are imposed to simplify
337            // implementations
338            // and to prohibit combinations that do not compress well. Decoders must
339            // support all
340            // valid combinations of bit depth and color type. The allowed
341            // combinations are:
342            //
343            // Color Allowed Interpretation
344            // Type Bit Depths
345            //
346            // 0 1,2,4,8,16 Each pixel is a grayscale sample.
347            //
348            // 2 8,16 Each pixel is an R,G,B triple.
349            //
350            // 3 1,2,4,8 Each pixel is a palette index;
351            // a PLTE chunk must appear.
352            //
353            // 4 8,16 Each pixel is a grayscale sample,
354            // followed by an alpha sample.
355            //
356            // 6 8,16 Each pixel is an R,G,B triple,
357            // followed by an alpha sample.
358            switch (colorType) {
359            case COLOR_TYPE_GREYSCALE:
360                return true;
361            case COLOR_TYPE_TRUE_COLOR:
362                return false;
363            case COLOR_TYPE_INDEXED_COLOR:
364                return false;
365            case COLOR_TYPE_GREYSCALE_WITH_ALPHA:
366                return true;
367            case COLOR_TYPE_TRUE_COLOR_WITH_ALPHA:
368                return false;
369            }
370    
371            // return -1;
372            throw new ImageReadException("PNG: unknown color type: " + colorType);
373        }

375        private int samplesPerPixel(final int colorType) throws ImageReadException {
376            // Color type is a single-byte integer that describes the interpretation
377            // of the
378            // image data. Color type codes represent sums of the following values:
379            // 1 (palette used), 2 (color used), and 4 (alpha channel used).
380            // Valid values are 0, 2, 3, 4, and 6.
381            //
382            // Bit depth restrictions for each color type are imposed to simplify
383            // implementations
384            // and to prohibit combinations that do not compress well. Decoders must
385            // support all
386            // valid combinations of bit depth and color type. The allowed
387            // combinations are:
388            //
389            // Color Allowed Interpretation
390            // Type Bit Depths
391            //
392            // 0 1,2,4,8,16 Each pixel is a grayscale sample.
393            //
394            // 2 8,16 Each pixel is an R,G,B triple.
395            //
396            // 3 1,2,4,8 Each pixel is a palette index;
397            // a PLTE chunk must appear.
398            //
399            // 4 8,16 Each pixel is a grayscale sample,
400            // followed by an alpha sample.
401            //
402            // 6 8,16 Each pixel is an R,G,B triple,
403            // followed by an alpha sample.
404            switch (colorType) {
405            case COLOR_TYPE_GREYSCALE:
406                return 1;
407            case COLOR_TYPE_TRUE_COLOR:
408                return 3;
409            case COLOR_TYPE_INDEXED_COLOR:
410                return 1; // is this accurate ? how may bits per index?
411            case COLOR_TYPE_GREYSCALE_WITH_ALPHA:
412                return 2;
413            case COLOR_TYPE_TRUE_COLOR_WITH_ALPHA:
414                return 4;
415            }
416    
417            // return -1;
418            throw new ImageReadException("PNG: unknown color type: " + colorType);
419        }

421        private List<PngChunk> filterChunks(final List<PngChunk> v, final int type) {
422            final List<PngChunk> result = new ArrayList<PngChunk>();
423    
424            for (int i = 0; i < v.size(); i++) {
425                final PngChunk chunk = v.get(i);
426                if (chunk.chunkType == type) {
427                    result.add(chunk);
428                }
429            }
430    
431            return result;
432        }

434        private boolean hasAlphaChannel(final int ColorType) throws ImageReadException {
435            switch (ColorType) {
436            case COLOR_TYPE_GREYSCALE: // 1,2,4,8,16 Each pixel is a grayscale
437                // sample.
438            case COLOR_TYPE_TRUE_COLOR: // 8,16 Each pixel is an R,G,B triple.
439            case COLOR_TYPE_INDEXED_COLOR: // 1,2,4,8 Each pixel is a palette index;
440                return false;
441            case COLOR_TYPE_GREYSCALE_WITH_ALPHA: // 8,16 Each pixel is a grayscale
442                // sample,
443                // followed by an alpha sample.
444            case COLOR_TYPE_TRUE_COLOR_WITH_ALPHA: // 8,16 Each pixel is an R,G,B
445                // triple,
446                // followed by an alpha sample.
447                return true;
448            default:
449                throw new ImageReadException("PNG: unknown color type: "
450                        + ColorType);
451            }
452        }

504        @Override
505        public ImageInfo getImageInfo(final ByteSource byteSource, final Map<String,Object> params)
506                throws ImageReadException, IOException {
507            final List<PngChunk> chunks = readChunks(byteSource, new int[] { IHDR, pHYs,
508                    tEXt, zTXt, tRNS, PLTE, iTXt, }, false);
509    
510            // if(chunks!=null)
511            // System.out.println("chunks: " + chunks.size());
512    
513            if ((chunks == null) || (chunks.size() < 1)) {
514                throw new ImageReadException("PNG: no chunks");
515            }
516    
517            final List<PngChunk> IHDRs = filterChunks(chunks, IHDR);
518            if (IHDRs.size() != 1) {
519                throw new ImageReadException("PNG contains more than one Header");
520            }
521    
522            final PngChunkIhdr pngChunkIHDR = (PngChunkIhdr) IHDRs.get(0);
523    
524            boolean isTransparent = false;
525    
526            final List<PngChunk> tRNSs = filterChunks(chunks, tRNS);
527            if (tRNSs.size() > 0) {
528                isTransparent = true;
529            } else {
530                // CE - Fix Alpha.
531                isTransparent = hasAlphaChannel(pngChunkIHDR.colorType);
532                // END FIX
533            }
534    
535            PngChunkPhys pngChunkpHYs = null;
536    
537            final List<PngChunk> pHYss = filterChunks(chunks, pHYs);
538            if (pHYss.size() > 1) {
539                throw new ImageReadException("PNG contains more than one pHYs: "
540                        + pHYss.size());
541            } else if (pHYss.size() == 1) {
542                pngChunkpHYs = (PngChunkPhys) pHYss.get(0);
543            }
544    
545            final List<PngChunk> tEXts = filterChunks(chunks, tEXt);
546            final List<PngChunk> zTXts = filterChunks(chunks, zTXt);
547            final List<PngChunk> iTXts = filterChunks(chunks, iTXt);
548    
549            final List<String> comments = new ArrayList<String>();
550            final List<PngText> textChunks = new ArrayList<PngText>();
551    
552            for (int i = 0; i < tEXts.size(); i++) {
553                final PngChunkText pngChunktEXt = (PngChunkText) tEXts.get(i);
554                comments.add(pngChunktEXt.keyword + ": " + pngChunktEXt.text);
555                textChunks.add(pngChunktEXt.getContents());
556            }
557            for (int i = 0; i < zTXts.size(); i++) {
558                final PngChunkZtxt pngChunkzTXt = (PngChunkZtxt) zTXts.get(i);
559                comments.add(pngChunkzTXt.keyword + ": " + pngChunkzTXt.text);
560                textChunks.add(pngChunkzTXt.getContents());
561            }
562            for (int i = 0; i < iTXts.size(); i++) {
563                final PngChunkItxt pngChunkiTXt = (PngChunkItxt) iTXts.get(i);
564                comments.add(pngChunkiTXt.keyword + ": " + pngChunkiTXt.text);
565                textChunks.add(pngChunkiTXt.getContents());
566            }
567    
568            final int BitsPerPixel = pngChunkIHDR.bitDepth
569                    * samplesPerPixel(pngChunkIHDR.colorType);
570            final ImageFormat Format = ImageFormat.IMAGE_FORMAT_PNG;
571            final String FormatName = "PNG Portable Network Graphics";
572            final int Height = pngChunkIHDR.height;
573            final String MimeType = "image/png";
574            final int NumberOfImages = 1;
575            final int Width = pngChunkIHDR.width;
576            final boolean isProgressive = (pngChunkIHDR.interlaceMethod != 0);
577    
578            int PhysicalHeightDpi = -1;
579            float PhysicalHeightInch = -1;
580            int PhysicalWidthDpi = -1;
581            float PhysicalWidthInch = -1;
582    
583            // if (pngChunkpHYs != null)
584            // {
585            // System.out.println("\t" + "pngChunkpHYs.UnitSpecifier: " +
586            // pngChunkpHYs.UnitSpecifier );
587            // System.out.println("\t" + "pngChunkpHYs.PixelsPerUnitYAxis: " +
588            // pngChunkpHYs.PixelsPerUnitYAxis );
589            // System.out.println("\t" + "pngChunkpHYs.PixelsPerUnitXAxis: " +
590            // pngChunkpHYs.PixelsPerUnitXAxis );
591            // }
592            if ((pngChunkpHYs != null) && (pngChunkpHYs.UnitSpecifier == 1)) { // meters
593                final double meters_per_inch = 0.0254;
594    
595                PhysicalWidthDpi = (int) Math
596                        .round(pngChunkpHYs.PixelsPerUnitXAxis
597                                * meters_per_inch);
598                PhysicalWidthInch = (float) (Width / (pngChunkpHYs.PixelsPerUnitXAxis * meters_per_inch));
599                PhysicalHeightDpi = (int) Math
600                        .round(pngChunkpHYs.PixelsPerUnitYAxis
601                                * meters_per_inch);
602                PhysicalHeightInch = (float) (Height / (pngChunkpHYs.PixelsPerUnitYAxis * meters_per_inch));
603            }
604    
605            final String FormatDetails = "Png";
606    
607            boolean usesPalette = false;
608    
609            final List<PngChunk> PLTEs = filterChunks(chunks, PLTE);
610            if (PLTEs.size() > 1) {
611                usesPalette = true;
612            }
613    
614            int ColorType;
615            switch (pngChunkIHDR.colorType) {
616            case COLOR_TYPE_GREYSCALE: // 1,2,4,8,16 Each pixel is a grayscale
617                // sample.
618            case COLOR_TYPE_GREYSCALE_WITH_ALPHA: // 8,16 Each pixel is a
619                // grayscale sample,
620                // followed by an alpha sample.
621                ColorType = ImageInfo.COLOR_TYPE_GRAYSCALE;
622                break;
623            case COLOR_TYPE_TRUE_COLOR: // 8,16 Each pixel is an R,G,B triple.
624            case COLOR_TYPE_INDEXED_COLOR: // 1,2,4,8 Each pixel is a palette
625                // index;
626            case COLOR_TYPE_TRUE_COLOR_WITH_ALPHA: // 8,16 Each pixel is an
627                // R,G,B triple,
628                // followed by an alpha sample.
629                ColorType = ImageInfo.COLOR_TYPE_RGB;
630                break;
631            default:
632                throw new ImageReadException("Png: Unknown ColorType: "
633                        + pngChunkIHDR.colorType);
634            }
635    
636            final String compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_PNG_FILTER;
637    
638            return new PngImageInfo(FormatDetails, BitsPerPixel, comments,
639                    Format, FormatName, Height, MimeType, NumberOfImages,
640                    PhysicalHeightDpi, PhysicalHeightInch, PhysicalWidthDpi,
641                    PhysicalWidthInch, Width, isProgressive, isTransparent,
642                    usesPalette, ColorType, compressionAlgorithm, textChunks);
643        }

645        @Override
646        public BufferedImage getBufferedImage(final ByteSource byteSource, final Map<String,Object> params)
647                throws ImageReadException, IOException {
648            ParamMap.getParamBoolean(params, PARAM_KEY_VERBOSE, false);
649    
650            if (params.containsKey(PARAM_KEY_VERBOSE)) {
651                params.remove(PARAM_KEY_VERBOSE);
652            }
653    
654            // if (params.size() > 0) {
655            // Object firstKey = params.keySet().iterator().next();
656            // throw new ImageWriteException("Unknown parameter: " + firstKey);
657            // }
658    
659            final List<PngChunk> chunks = readChunks(byteSource, new int[] { IHDR, PLTE,
660                    IDAT, tRNS, iCCP, gAMA, sRGB, }, false);
661    
662            if ((chunks == null) || (chunks.size() < 1)) {
663                throw new ImageReadException("PNG: no chunks");
664            }
665    
666            final List<PngChunk> IHDRs = filterChunks(chunks, IHDR);
667            if (IHDRs.size() != 1) {
668                throw new ImageReadException("PNG contains more than one Header");
669            }
670    
671            final PngChunkIhdr pngChunkIHDR = (PngChunkIhdr) IHDRs.get(0);
672    
673            final List<PngChunk> PLTEs = filterChunks(chunks, PLTE);
674            if (PLTEs.size() > 1) {
675                throw new ImageReadException("PNG contains more than one Palette");
676            }
677    
678            PngChunkPlte pngChunkPLTE = null;
679            if (PLTEs.size() == 1) {
680                pngChunkPLTE = (PngChunkPlte) PLTEs.get(0);
681            }
682    
683            // -----
684    
685            final List<PngChunk> IDATs = filterChunks(chunks, IDAT);
686            if (IDATs.size() < 1) {
687                throw new ImageReadException("PNG missing image data");
688            }
689    
690            ByteArrayOutputStream baos = new ByteArrayOutputStream();
691            for (int i = 0; i < IDATs.size(); i++) {
692                final PngChunkIdat pngChunkIDAT = (PngChunkIdat) IDATs.get(i);
693                final byte bytes[] = pngChunkIDAT.bytes;
694                // System.out.println(i + ": bytes: " + bytes.length);
695                baos.write(bytes);
696            }
697    
698            final byte compressed[] = baos.toByteArray();
699    
700            baos = null;
701    
702            TransparencyFilter transparencyFilter = null;
703    
704            final List<PngChunk> tRNSs = filterChunks(chunks, tRNS);
705            if (tRNSs.size() > 0) {
706                final PngChunk pngChunktRNS = tRNSs.get(0);
707                transparencyFilter = getTransparencyFilter(pngChunkIHDR.colorType,
708                        pngChunktRNS);
709            }
710    
711            ICC_Profile icc_profile = null;
712            GammaCorrection gammaCorrection = null;
713            {
714                final List<PngChunk> sRGBs = filterChunks(chunks, sRGB);
715                final List<PngChunk> gAMAs = filterChunks(chunks, gAMA);
716                final List<PngChunk> iCCPs = filterChunks(chunks, iCCP);
717                if (sRGBs.size() > 1) {
718                    throw new ImageReadException("PNG: unexpected sRGB chunk");
719                }
720                if (gAMAs.size() > 1) {
721                    throw new ImageReadException("PNG: unexpected gAMA chunk");
722                }
723                if (iCCPs.size() > 1) {
724                    throw new ImageReadException("PNG: unexpected iCCP chunk");
725                }
726    
727                if (sRGBs.size() == 1) {
728                    // no color management neccesary.
729                    if (debug) {
730                        System.out.println("sRGB, no color management neccesary.");
731                    }
732                } else if (iCCPs.size() == 1) {
733                    if (debug) {
734                        System.out.println("iCCP.");
735                    }
736    
737                    final PngChunkIccp pngChunkiCCP = (PngChunkIccp) iCCPs.get(0);
738                    final byte bytes[] = pngChunkiCCP.UncompressedProfile;
739    
740                    icc_profile = ICC_Profile.getInstance(bytes);
741                } else if (gAMAs.size() == 1) {
742                    final PngChunkGama pngChunkgAMA = (PngChunkGama) gAMAs.get(0);
743                    final double gamma = pngChunkgAMA.getGamma();
744    
745                    // charles: what is the correct target value here?
746                    // double targetGamma = 2.2;
747                    final double targetGamma = 1.0;
748                    final double diff = Math.abs(targetGamma - gamma);
749                    if (diff >= 0.5) {
750                        gammaCorrection = new GammaCorrection(gamma, targetGamma);
751                    }
752    
753                    if (gammaCorrection != null) {
754                        if (pngChunkPLTE != null) {
755                            pngChunkPLTE.correct(gammaCorrection);
756                        }
757                    }
758    
759                }
760            }
761    
762            {
763                final int width = pngChunkIHDR.width;
764                final int height = pngChunkIHDR.height;
765                final int colorType = pngChunkIHDR.colorType;
766                final int bitDepth = pngChunkIHDR.bitDepth;
767    
768                final int bitsPerSample = bitDepth;
769    
770                if (pngChunkIHDR.filterMethod != 0) {
771                    throw new ImageReadException("PNG: unknown FilterMethod: "
772                            + pngChunkIHDR.filterMethod);
773                }
774    
775                final int samplesPerPixel = samplesPerPixel(pngChunkIHDR.colorType);
776                final boolean isGrayscale = isGrayscale(pngChunkIHDR.colorType);
777    
778                final int bitsPerPixel = bitsPerSample * samplesPerPixel;
779    
780                final boolean hasAlpha = colorType == COLOR_TYPE_GREYSCALE_WITH_ALPHA
781                        || colorType == COLOR_TYPE_TRUE_COLOR_WITH_ALPHA;
782    
783                BufferedImage result;
784                if (isGrayscale) {
785                    result = getBufferedImageFactory(params)
786                            .getGrayscaleBufferedImage(width, height, hasAlpha);
787                } else {
788                    result = getBufferedImageFactory(params).getColorBufferedImage(
789                            width, height, hasAlpha);
790                }
791    
792                final ByteArrayInputStream bais = new ByteArrayInputStream(compressed);
793                final InflaterInputStream iis = new InflaterInputStream(bais);
794    
795                ScanExpediter scanExpediter;
796    
797                if (pngChunkIHDR.interlaceMethod == 0) {
798                    scanExpediter = new ScanExpediterSimple(width, height, iis,
799                            result, colorType, bitDepth, bitsPerPixel,
800                            pngChunkPLTE, gammaCorrection, transparencyFilter);
801                } else if (pngChunkIHDR.interlaceMethod == 1) {
802                    scanExpediter = new ScanExpediterInterlaced(width, height, iis,
803                            result, colorType, bitDepth, bitsPerPixel,
804                            pngChunkPLTE, gammaCorrection, transparencyFilter);
805                } else {
806                    throw new ImageReadException("Unknown InterlaceMethod: "
807                            + pngChunkIHDR.interlaceMethod);
808                }
809    
810                scanExpediter.drive();
811    
812                if (icc_profile != null) {
813                    final Boolean is_srgb = new IccProfileParser().issRGB(icc_profile);
814                    if (is_srgb == null || !is_srgb.booleanValue()) {
815                        final ICC_ColorSpace cs = new ICC_ColorSpace(icc_profile);
816    
817                        final ColorModel srgbCM = ColorModel.getRGBdefault();
818                        final ColorSpace cs_sRGB = srgbCM.getColorSpace();
819    
820                        result = new ColorTools().convertBetweenColorSpaces(result,
821                                cs, cs_sRGB);
822                    }
823                }
824    
825                return result;
826    
827            }
828    
829        }

831        @Override
832        public boolean dumpImageFile(final PrintWriter pw, final ByteSource byteSource)
833                throws ImageReadException, IOException {
834            final ImageInfo imageInfo = getImageInfo(byteSource);
835            if (imageInfo == null) {
836                return false;
837            }
838    
839            imageInfo.toString(pw, "");
840    
841            final List<PngChunk> chunks = readChunks(byteSource, null, false);
842            final List<PngChunk> IHDRs = filterChunks(chunks, IHDR);
843            if (IHDRs.size() != 1) {
844                if (debug) {
845                    System.out.println("PNG contains more than one Header");
846                }
847                return false;
848            }
849            final PngChunkIhdr pngChunkIHDR = (PngChunkIhdr) IHDRs.get(0);
850            pw.println("Color: "
851                    + getColorTypeDescription(pngChunkIHDR.colorType));
852    
853            pw.println("chunks: " + chunks.size());
854    
855            if ((chunks.size() < 1)) {
856                return false;
857            }
858    
859            for (int i = 0; i < chunks.size(); i++) {
860                final PngChunk chunk = chunks.get(i);
861                printCharQuad(pw, "\t" + i + ": ", chunk.chunkType);
862            }
863    
864            pw.println("");
865    
866            pw.flush();
867    
868            return true;
869        }

877        /**
878         * Extracts embedded XML metadata as XML string.
879         * <p>
880         * 
881         * @param byteSource
882         *            File containing image data.
883         * @param params
884         *            Map of optional parameters, defined in SanselanConstants.
885         * @return Xmp Xml as String, if present. Otherwise, returns null.
886         */
887        @Override
888        public String getXmpXml(final ByteSource byteSource, final Map<String,Object> params)
889                throws ImageReadException, IOException {
890    
891            final List<PngChunk> chunks = readChunks(byteSource, new int[] { iTXt, },
892                    false);
893    
894            if ((chunks == null) || (chunks.size() < 1)) {
895                return null;
896            }
897    
898            final List<PngChunkItxt> xmpChunks = new ArrayList<PngChunkItxt>();
899            for (int i = 0; i < chunks.size(); i++) {
900                final PngChunkItxt chunk = (PngChunkItxt) chunks.get(i);
901                if (!chunk.getKeyword().equals(XMP_KEYWORD)) {
902                    continue;
903                }
904                xmpChunks.add(chunk);
905            }
906    
907            if (xmpChunks.size() < 1) {
908                return null;
909            }
910            if (xmpChunks.size() > 1) {
911                throw new ImageReadException(
912                        "PNG contains more than one XMP chunk.");
913            }
914    
915            final PngChunkItxt chunk = xmpChunks.get(0);
916            return chunk.getText();
917        }
