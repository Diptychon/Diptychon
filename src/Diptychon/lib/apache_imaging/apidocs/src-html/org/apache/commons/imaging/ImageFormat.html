017    package org.apache.commons.imaging;
018    
019    /**
020     * Provides a definition for an image format.
021     */
022    public class ImageFormat {
023        private final String name;
024        private final String extension;
025        public final boolean actual;
026    
027        private ImageFormat(final String name, final boolean actual) {
028            this.name = name;
029            this.extension = name;
030            this.actual = actual;
031        }
032    
033        private ImageFormat(final String name) {
034            this.name = name;
035            this.extension = name;
036            actual = true;
037        }
038    
039        @Override
040        public boolean equals(final Object o) {
041            if (!(o instanceof ImageFormat)) {
042                return false;
043            }
044    
045            final ImageFormat other = (ImageFormat) o;
046    
047            return other.getName().equals(getName());
048    
049        }
050    
051        public String getName() {
052            return name;
053        }
054        
055        public String getExtension() {
056            return extension;
057        }
058    
059        @Override
060        public String toString() {
061            return "{" + getName() + ": " + getExtension() + "}";
062        }
063    
064        @Override
065        public int hashCode() {
066            return getName().hashCode();
067        }
068    
069        public static final ImageFormat IMAGE_FORMAT_UNKNOWN = new ImageFormat(
070                "UNKNOWN", false);
071        public static final ImageFormat IMAGE_FORMAT_BMP = new ImageFormat("BMP");
072        public static final ImageFormat IMAGE_FORMAT_DCX = new ImageFormat("DCX");
073        public static final ImageFormat IMAGE_FORMAT_GIF = new ImageFormat("GIF");
074        public static final ImageFormat IMAGE_FORMAT_ICNS = new ImageFormat("ICNS");
075        public static final ImageFormat IMAGE_FORMAT_ICO = new ImageFormat("ICO");
076        public static final ImageFormat IMAGE_FORMAT_JBIG2 = new ImageFormat("JBig2");
077        public static final ImageFormat IMAGE_FORMAT_JPEG = new ImageFormat("JPEG");
078        public static final ImageFormat IMAGE_FORMAT_PAM = new ImageFormat("PAM");
079        public static final ImageFormat IMAGE_FORMAT_PSD = new ImageFormat("PSD");
080        public static final ImageFormat IMAGE_FORMAT_PBM = new ImageFormat("PBM");
081        public static final ImageFormat IMAGE_FORMAT_PGM = new ImageFormat("PGM");
082        public static final ImageFormat IMAGE_FORMAT_PNM = new ImageFormat("PNM");
083        public static final ImageFormat IMAGE_FORMAT_PPM = new ImageFormat("PPM");
084        public static final ImageFormat IMAGE_FORMAT_PCX = new ImageFormat("PCX");
085        public static final ImageFormat IMAGE_FORMAT_PNG = new ImageFormat("PNG");
086        public static final ImageFormat IMAGE_FORMAT_RGBE = new ImageFormat("RGBE");
087        public static final ImageFormat IMAGE_FORMAT_TGA = new ImageFormat("TGA");
088        public static final ImageFormat IMAGE_FORMAT_TIFF = new ImageFormat("TIFF");
089        public static final ImageFormat IMAGE_FORMAT_WBMP = new ImageFormat("WBMP");
090        public static final ImageFormat IMAGE_FORMAT_XBM = new ImageFormat("XBM");
091        public static final ImageFormat IMAGE_FORMAT_XPM = new ImageFormat("XPM");
092        
093        
094    
095        /**
096         * Gets an array of ImageFormat objects for all supported image
097         * formats.
098         * @return A valid array of ImageFormat objects.
099         */
100        public static ImageFormat[] getAllFormats() {
101            final ImageFormat result[] = {
102                    IMAGE_FORMAT_UNKNOWN,
103                    IMAGE_FORMAT_BMP,
104                    IMAGE_FORMAT_DCX,
105                    IMAGE_FORMAT_GIF,
106                    IMAGE_FORMAT_ICNS,
107                    IMAGE_FORMAT_ICO,
108                    IMAGE_FORMAT_JBIG2,
109                    IMAGE_FORMAT_JPEG,
110                    IMAGE_FORMAT_PNG,
111                    IMAGE_FORMAT_PAM,
112                    IMAGE_FORMAT_PSD,
113                    IMAGE_FORMAT_PBM,
114                    IMAGE_FORMAT_PCX,
115                    IMAGE_FORMAT_PGM,
116                    IMAGE_FORMAT_PNM,
117                    IMAGE_FORMAT_PPM,
118                    IMAGE_FORMAT_RGBE,
119                    IMAGE_FORMAT_TGA,
120                    IMAGE_FORMAT_TIFF, 
121                    IMAGE_FORMAT_WBMP, 
122                    IMAGE_FORMAT_XBM,
123                    IMAGE_FORMAT_XPM,
124            };
125    
126            return result;
127        }
128    }
