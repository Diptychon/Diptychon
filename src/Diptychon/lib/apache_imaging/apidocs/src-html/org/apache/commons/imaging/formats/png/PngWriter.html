049        /*
050         1. IHDR: image header, which is the first chunk in a PNG datastream.
051         2. PLTE: palette table associated with indexed PNG images.
052         3. IDAT: image data chunks.
053         4. IEND: image trailer, which is the last chunk in a PNG datastream.
054    
055         The remaining 14 chunk types are termed ancillary chunk types, which encoders may generate and decoders may interpret.
056    
057         1. Transparency information: tRNS (see 11.3.2: Transparency information).
058         2. Colour space information: cHRM, gAMA, iCCP, sBIT, sRGB (see 11.3.3: Colour space information).
059         3. Textual information: iTXt, tEXt, zTXt (see 11.3.4: Textual information).
060         4. Miscellaneous information: bKGD, hIST, pHYs, sPLT (see 11.3.5: Miscellaneous information).
061         5. Time information: tIME (see 11.3.6: Time stamp information).
062        */
063        
064        private final void writeInt(final OutputStream os, final int value) throws IOException {
065            os.write(0xff & (value >> 24));
066            os.write(0xff & (value >> 16));
067            os.write(0xff & (value >> 8));
068            os.write(0xff & (value >> 0));
069        }
070    
071        private final void writeChunk(final OutputStream os, final byte chunkType[],
072                final byte data[]) throws IOException {
073            final int dataLength = data == null ? 0 : data.length;
074            writeInt(os, dataLength);
075            os.write(chunkType);
076            if (data != null) {
077                os.write(data);
078            }
079    
080            final PngCrc png_crc = new PngCrc();
081    
082            final long crc1 = png_crc.start_partial_crc(chunkType, chunkType.length);
083            final long crc2 = data == null ? crc1 : png_crc.continue_partial_crc(
084                    crc1, data, data.length);
085            final int crc = (int) png_crc.finish_partial_crc(crc2);
086    
087            writeInt(os, crc);
088        }

090        private static class ImageHeader {
091            public final int width;
092            public final int height;
093            public final byte bit_depth;
094            public final byte colorType;
095            public final byte compressionMethod;
096            public final byte filterMethod;
097            public final byte interlaceMethod;

099            public ImageHeader(final int width, final int height, final byte bit_depth,
100                    final byte colorType, final byte compressionMethod, final byte filterMethod,
101                    final byte interlaceMethod) {
102                this.width = width;
103                this.height = height;
104                this.bit_depth = bit_depth;
105                this.colorType = colorType;
106                this.compressionMethod = compressionMethod;
107                this.filterMethod = filterMethod;
108                this.interlaceMethod = interlaceMethod;
109            }

111        }

113        private void writeChunkIHDR(final OutputStream os, final ImageHeader value)
114                throws IOException {
115            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
116            writeInt(baos, value.width);
117            writeInt(baos, value.height);
118            baos.write(0xff & value.bit_depth);
119            baos.write(0xff & value.colorType);
120            baos.write(0xff & value.compressionMethod);
121            baos.write(0xff & value.filterMethod);
122            baos.write(0xff & value.interlaceMethod);
123    
124            // Debug.debug("baos", baos.toByteArray());
125    
126            writeChunk(os, IHDR_CHUNK_TYPE.toByteArray(), baos.toByteArray());
127        }

129        private void writeChunkiTXt(final OutputStream os, final PngText.Itxt text)
130                throws IOException, ImageWriteException {
131            if (!UnicodeUtils.isValidISO_8859_1(text.keyword)) {
132                throw new ImageWriteException(
133                        "Png tEXt chunk keyword is not ISO-8859-1: " + text.keyword);
134            }
135            if (!UnicodeUtils.isValidISO_8859_1(text.languageTag)) {
136                throw new ImageWriteException(
137                        "Png tEXt chunk language tag is not ISO-8859-1: "
138                                + text.languageTag);
139            }
140    
141            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
142    
143            // keyword
144            baos.write(text.keyword.getBytes("ISO-8859-1"));
145            baos.write(0);
146    
147            baos.write(1); // compressed flag, true
148            baos.write(COMPRESSION_DEFLATE_INFLATE); // compression method
149    
150            // language tag
151            baos.write(text.languageTag.getBytes("ISO-8859-1"));
152            baos.write(0);
153    
154            // translated keyword
155            baos.write(text.translatedKeyword.getBytes("utf-8"));
156            baos.write(0);
157    
158            baos.write(new ZLibUtils().deflate(text.text.getBytes("utf-8")));
159    
160            writeChunk(os, iTXt_CHUNK_TYPE.toByteArray(), baos.toByteArray());
161        }

163        private void writeChunkzTXt(final OutputStream os, final PngText.Ztxt text)
164                throws IOException, ImageWriteException {
165            if (!UnicodeUtils.isValidISO_8859_1(text.keyword)) {
166                throw new ImageWriteException(
167                        "Png zTXt chunk keyword is not ISO-8859-1: " + text.keyword);
168            }
169            if (!UnicodeUtils.isValidISO_8859_1(text.text)) {
170                throw new ImageWriteException(
171                        "Png zTXt chunk text is not ISO-8859-1: " + text.text);
172            }
173    
174            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
175    
176            // keyword
177            baos.write(text.keyword.getBytes("ISO-8859-1"));
178            baos.write(0);
179    
180            // compression method
181            baos.write(PngConstants.COMPRESSION_DEFLATE_INFLATE);
182    
183            // text
184            baos.write(new ZLibUtils().deflate(text.text.getBytes("ISO-8859-1")));
185    
186            writeChunk(os, zTXt_CHUNK_TYPE.toByteArray(), baos.toByteArray());
187        }

235        private void writeChunkPLTE(final OutputStream os, final Palette palette)
236                throws IOException {
237            final int length = palette.length();
238            final byte bytes[] = new byte[length * 3];
239    
240            // Debug.debug("length", length);
241            for (int i = 0; i < length; i++) {
242                final int rgb = palette.getEntry(i);
243                final int index = i * 3;
244                // Debug.debug("index", index);
245                bytes[index + 0] = (byte) (0xff & (rgb >> 16));
246                bytes[index + 1] = (byte) (0xff & (rgb >> 8));
247                bytes[index + 2] = (byte) (0xff & (rgb >> 0));
248            }
249    
250            writeChunk(os, PLTE_CHUNK_TYPE.toByteArray(), bytes);
251        }

253        private void writeChunkTRNS(final OutputStream os, final Palette palette) throws IOException {
254            final byte[] bytes = new byte[palette.length()];
255            
256            for (int i = 0; i < bytes.length; i++) {
257                bytes[i] = (byte) (0xff & (palette.getEntry(i) >> 24));
258            }
259            
260            writeChunk(os, TRNS_CHUNK_TYPE.toByteArray(), bytes);
261        }

272        private void writeChunkPHYS(final OutputStream os, final int xPPU, final int yPPU, final byte units)
273                throws IOException {
274            final byte[] bytes = new byte[9];
275            bytes[0] = (byte) (0xff & (xPPU >> 24));
276            bytes[1] = (byte) (0xff & (xPPU >> 16));
277            bytes[2] = (byte) (0xff & (xPPU >> 8));
278            bytes[3] = (byte) (0xff & (xPPU >> 0));
279            bytes[4] = (byte) (0xff & (yPPU >> 24));
280            bytes[5] = (byte) (0xff & (yPPU >> 16));
281            bytes[6] = (byte) (0xff & (yPPU >> 8));
282            bytes[7] = (byte) (0xff & (yPPU >> 0));
283            bytes[8] = units;
284            writeChunk(os, IPHYS_CHUNK_TYPE.toByteArray(), bytes);
285        }

287        private byte getColourType(final boolean hasAlpha, final boolean isGrayscale) {
288            byte result;
289    
290            final boolean index = false; // charles
291    
292            if (index) {
293                result = COLOR_TYPE_INDEXED_COLOR;
294            } else if (isGrayscale) {
295                if (hasAlpha) {
296                    result = COLOR_TYPE_GREYSCALE_WITH_ALPHA;
297                } else {
298                    result = COLOR_TYPE_GREYSCALE;
299                }
300            } else if (hasAlpha) {
301                result = COLOR_TYPE_TRUE_COLOR_WITH_ALPHA;
302            } else {
303                result = COLOR_TYPE_TRUE_COLOR;
304            }
305    
306            return result;
307        }

309        private byte getBitDepth(final byte colorType, final Map<String,Object> params) {
310            byte result = 8;
311    
312            final Object o = params.get(PARAM_KEY_PNG_BIT_DEPTH);
313            if (o != null && o instanceof Number) {
314                final int value = ((Number) o).intValue();
315                switch (value) {
316                case 1:
317                case 2:
318                case 4:
319                case 8:
320                case 16:
321                    result = (byte) value;
322                    break;
323                default:
324                    result = 8;
325                }
326                switch (colorType) {
327                case COLOR_TYPE_GREYSCALE:
328                    break;
329                case COLOR_TYPE_INDEXED_COLOR:
330                    result = (byte) Math.min(8, result);
331                    break;
332                case COLOR_TYPE_GREYSCALE_WITH_ALPHA:
333                case COLOR_TYPE_TRUE_COLOR:
334                case COLOR_TYPE_TRUE_COLOR_WITH_ALPHA:
335                    result = (byte) Math.max(8, result);
336                    break;
337                default:
338                    result = 8;
339                }
340            }
341    
342            return result;
343        }

345        /*
346         between two chunk types indicates alternatives.
347         Table 5.3 - Chunk ordering rules
348         Critical chunks
349         (shall appear in this order, except PLTE is optional)
350         Chunk name     Multiple allowed    Ordering constraints
351         IHDR   No  Shall be first
352         PLTE   No  Before first IDAT
353         IDAT   Yes Multiple IDAT chunks shall be consecutive
354         IEND   No  Shall be last
355         Ancillary chunks
356         (need not appear in this order)
357         Chunk name     Multiple allowed    Ordering constraints
358         cHRM   No  Before PLTE and IDAT
359         gAMA   No  Before PLTE and IDAT
360         iCCP   No  Before PLTE and IDAT. If the iCCP chunk is present, the sRGB chunk should not be present.
361         sBIT   No  Before PLTE and IDAT
362         sRGB   No  Before PLTE and IDAT. If the sRGB chunk is present, the iCCP chunk should not be present.
363         bKGD   No  After PLTE; before IDAT
364         hIST   No  After PLTE; before IDAT
365         tRNS   No  After PLTE; before IDAT
366         pHYs   No  Before IDAT
367         sPLT   Yes Before IDAT
368         tIME   No  None
369         iTXt   Yes None
370         tEXt   Yes None
371         zTXt   Yes None
372         */
373        public void writeImage(final BufferedImage src, final OutputStream os, Map<String,Object> params)
374                throws ImageWriteException, IOException {
375            // make copy of params; we'll clear keys as we consume them.
376            params = new HashMap<String,Object>(params);
377    
378            // clear format key.
379            if (params.containsKey(PARAM_KEY_FORMAT)) {
380                params.remove(PARAM_KEY_FORMAT);
381            }
382            // clear verbose key.
383            if (params.containsKey(PARAM_KEY_VERBOSE)) {
384                params.remove(PARAM_KEY_VERBOSE);
385            }
386    
387            final Map<String,Object> rawParams = new HashMap<String,Object>(params);
388            if (params.containsKey(PARAM_KEY_PNG_FORCE_TRUE_COLOR)) {
389                params.remove(PARAM_KEY_PNG_FORCE_TRUE_COLOR);
390            }
391            if (params.containsKey(PARAM_KEY_PNG_FORCE_INDEXED_COLOR)) {
392                params.remove(PARAM_KEY_PNG_FORCE_INDEXED_COLOR);
393            }
394            if (params.containsKey(PARAM_KEY_PNG_BIT_DEPTH)) {
395                params.remove(PARAM_KEY_PNG_BIT_DEPTH);
396            }
397            if (params.containsKey(PARAM_KEY_XMP_XML)) {
398                params.remove(PARAM_KEY_XMP_XML);
399            }
400            if (params.containsKey(PARAM_KEY_PNG_TEXT_CHUNKS)) {
401                params.remove(PARAM_KEY_PNG_TEXT_CHUNKS);
402            }
403            params.remove(PARAM_KEY_PIXEL_DENSITY);
404            if (params.size() > 0) {
405                final Object firstKey = params.keySet().iterator().next();
406                throw new ImageWriteException("Unknown parameter: " + firstKey);
407            }
408            params = rawParams;
409    
410            final int width = src.getWidth();
411            final int height = src.getHeight();
412    
413            final boolean hasAlpha = new PaletteFactory().hasTransparency(src);
414            if (verbose) {
415                Debug.debug("hasAlpha", hasAlpha);
416            }
417            // int transparency = new PaletteFactory().getTransparency(src);
418    
419            boolean isGrayscale = new PaletteFactory().isGrayscale(src);
420            if (verbose) {
421                Debug.debug("isGrayscale", isGrayscale);
422            }
423    
424            byte colorType;
425            {
426                final boolean forceIndexedColor = ParamMap.getParamBoolean(params,
427                        PARAM_KEY_PNG_FORCE_INDEXED_COLOR, false);
428                final boolean forceTrueColor = ParamMap.getParamBoolean(params,
429                        PARAM_KEY_PNG_FORCE_TRUE_COLOR, false);
430    
431                if (forceIndexedColor && forceTrueColor) {
432                    throw new ImageWriteException(
433                            "Params: Cannot force both indexed and true color modes");
434                } else if (forceIndexedColor) {
435                    colorType = COLOR_TYPE_INDEXED_COLOR;
436                } else if (forceTrueColor) {
437                    colorType = (byte) (hasAlpha ? COLOR_TYPE_TRUE_COLOR_WITH_ALPHA
438                            : COLOR_TYPE_TRUE_COLOR);
439                    isGrayscale = false;
440                } else {
441                    colorType = getColourType(hasAlpha, isGrayscale);
442                }
443                if (verbose) {
444                    Debug.debug("colorType", colorType);
445                }
446            }
447    
448            final byte bitDepth = getBitDepth(colorType, params);
449            if (verbose) {
450                Debug.debug("bit_depth", bitDepth);
451            }
452    
453            int sampleDepth;
454            if (colorType == COLOR_TYPE_INDEXED_COLOR) {
455                sampleDepth = 8;
456            } else {
457                sampleDepth = bitDepth;
458            }
459            if (verbose) {
460                Debug.debug("sample_depth", sampleDepth);
461            }
462    
463            {
464                PNG_Signature.writeTo(os);
465            }
466            {
467                // IHDR must be first
468    
469                final byte compressionMethod = COMPRESSION_TYPE_INFLATE_DEFLATE;
470                final byte filterMethod = FILTER_METHOD_ADAPTIVE;
471                final byte interlaceMethod = INTERLACE_METHOD_NONE;
472    
473                final ImageHeader imageHeader = new ImageHeader(width, height, bitDepth,
474                        colorType, compressionMethod, filterMethod, interlaceMethod);
475    
476                writeChunkIHDR(os, imageHeader);
477            }
478    
479            {
480                // sRGB No Before PLTE and IDAT. If the sRGB chunk is present, the
481                // iCCP chunk should not be present.
482    
483                // charles
484            }
485    
486            Palette palette = null;
487            if (colorType == COLOR_TYPE_INDEXED_COLOR) {
488                // PLTE No Before first IDAT
489    
490                final int max_colors = hasAlpha ? 255 : 256;
491    
492                palette = new PaletteFactory().makeQuantizedRgbPalette(src, max_colors);
493                // Palette palette2 = new PaletteFactory().makePaletteSimple(src,
494                // max_colors);
495    
496                // palette.dump();
497    
498                writeChunkPLTE(os, palette);
499            }
500    
501            final Object pixelDensityObj = params.get(PARAM_KEY_PIXEL_DENSITY);
502            if (pixelDensityObj instanceof PixelDensity) {
503                final PixelDensity pixelDensity = (PixelDensity) pixelDensityObj;
504                if (pixelDensity.isUnitless()) {
505                    writeChunkPHYS(os, (int) Math.round(pixelDensity
506                            .getRawHorizontalDensity()),
507                            (int) Math.round(pixelDensity.getRawVerticalDensity()),
508                            (byte) 0);
509                } else {
510                    writeChunkPHYS(os, (int) Math.round(pixelDensity
511                            .horizontalDensityMetres()),
512                            (int) Math.round(pixelDensity.verticalDensityMetres()),
513                            (byte) 1);
514                }
515            }
516    
517            if (params.containsKey(PARAM_KEY_XMP_XML)) {
518                final String xmpXml = (String) params.get(PARAM_KEY_XMP_XML);
519                writeChunkXmpiTXt(os, xmpXml);
520            }
521    
522            if (params.containsKey(PARAM_KEY_PNG_TEXT_CHUNKS)) {
523                final List<?> outputTexts = (List<?>) params.get(PARAM_KEY_PNG_TEXT_CHUNKS);
524                for (int i = 0; i < outputTexts.size(); i++) {
525                    final PngText text = (PngText) outputTexts.get(i);
526                    if (text instanceof PngText.Text) {
527                        writeChunktEXt(os, (PngText.Text) text);
528                    } else if (text instanceof PngText.Ztxt) {
529                        writeChunkzTXt(os, (PngText.Ztxt) text);
530                    } else if (text instanceof PngText.Itxt) {
531                        writeChunkiTXt(os, (PngText.Itxt) text);
532                    } else {
533                        throw new ImageWriteException(
534                                "Unknown text to embed in PNG: " + text);
535                    }
536                }
537            }
538    
539            {
540                // Debug.debug("writing IDAT");
541    
542                // IDAT Yes Multiple IDAT chunks shall be consecutive
543    
544                byte uncompressed[];
545                {
546                    final ByteArrayOutputStream baos = new ByteArrayOutputStream();
547    
548                    final boolean useAlpha = colorType == COLOR_TYPE_GREYSCALE_WITH_ALPHA
549                            || colorType == COLOR_TYPE_TRUE_COLOR_WITH_ALPHA;
550    
551                    final int row[] = new int[width];
552                    for (int y = 0; y < height; y++) {
553                        // Debug.debug("y", y + "/" + height);
554                        src.getRGB(0, y, width, 1, row, 0, width);
555    
556                        final byte filter_type = FILTER_TYPE_NONE;
557                        baos.write(filter_type);
558                        for (int x = 0; x < width; x++) {
559                            final int argb = row[x];
560    
561                            if (palette != null) {
562                                final int index = palette.getPaletteIndex(argb);
563                                baos.write(0xff & index);
564                            } else {
565                                final int alpha = 0xff & (argb >> 24);
566                                final int red = 0xff & (argb >> 16);
567                                final int green = 0xff & (argb >> 8);
568                                final int blue = 0xff & (argb >> 0);
569    
570                                if (isGrayscale) {
571                                    final int gray = (red + green + blue) / 3;
572                                    // if(y==0)
573                                    // {
574                                    // Debug.debug("gray: " + x + ", " + y +
575                                    // " argb: 0x"
576                                    // + Integer.toHexString(argb) + " gray: 0x"
577                                    // + Integer.toHexString(gray));
578                                    // // Debug.debug(x + ", " + y + " gray", gray);
579                                    // // Debug.debug(x + ", " + y + " gray", gray);
580                                    // Debug.debug(x + ", " + y + " gray", gray +
581                                    // " " + Integer.toHexString(gray));
582                                    // Debug.debug();
583                                    // }
584                                    baos.write(gray);
585                                } else {
586                                    baos.write(red);
587                                    baos.write(green);
588                                    baos.write(blue);
589                                }
590                                if (useAlpha) {
591                                    baos.write(alpha);
592                                }
593                            }
594                        }
595                    }
596                    uncompressed = baos.toByteArray();
597                }
598    
599                // Debug.debug("uncompressed", uncompressed.length);
600    
601                final ByteArrayOutputStream baos = new ByteArrayOutputStream();
602                final DeflaterOutputStream dos = new DeflaterOutputStream(baos);
603                final int chunk_size = 256 * 1024;
604                for (int index = 0; index < uncompressed.length; index += chunk_size) {
605                    final int end = Math.min(uncompressed.length, index + chunk_size);
606                    final int length = end - index;
607    
608                    dos.write(uncompressed, index, length);
609                    dos.flush();
610                    baos.flush();
611    
612                    final byte compressed[] = baos.toByteArray();
613                    baos.reset();
614                    if (compressed.length > 0) {
615                        // Debug.debug("compressed", compressed.length);
616                        writeChunkIDAT(os, compressed);
617                    }
618    
619                }
620                {
621                    dos.finish();
622                    final byte compressed[] = baos.toByteArray();
623                    if (compressed.length > 0) {
624                        // Debug.debug("compressed final", compressed.length);
625                        writeChunkIDAT(os, compressed);
626                    }
627                }
628            }
629    
630            {
631                // IEND No Shall be last
632    
633                writeChunkIEND(os);
634            }
635    
636            /*
637             Ancillary chunks
