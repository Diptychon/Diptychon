271        public void fillInValue(final ByteSource byteSource) throws IOException,
272                TiffValueOutsideFileBoundsException {
273            if (fieldType.isLocalValue(this)) {
274                return;
275            }
276    
277            final int valueLength = getValueLengthInBytes();
278            final long valueLengthLong = 0xffffffffL & valueLength;
279    
280            // Debug.debug("fillInValue tag", tag);
281            // Debug.debug("fillInValue tagInfo", tagInfo);
282            // Debug.debug("fillInValue valueOffset", valueOffset);
283            // Debug.debug("fillInValue valueLength", valueLength);
284    
285            if (valueOffset < 0
286                    || valueOffset + valueLengthLong > byteSource.getLength()) {
287                throw new TiffValueOutsideFileBoundsException(
288                        "Attempt to read byte range starting from " + valueOffset
289                                + " " + "of length " + valueLength + " "
290                                + "which is outside the file's size of "
291                                + byteSource.getLength());
292            }
293            final byte bytes[] = byteSource.getBlock(valueOffset, valueLength);
294            setOversizeValue(bytes);
295        }

633        public int[] getIntArrayValue() throws ImageReadException {
634            final Object o = getValue();
635            // if (o == null)
636            // return null;
637    
638            if (o instanceof Number) {
639                return new int[] { ((Number) o).intValue() };
640            } else if (o instanceof Number[]) {
641                final Number numbers[] = (Number[]) o;
642                final int result[] = new int[numbers.length];
643                for (int i = 0; i < numbers.length; i++) {
644                    result[i] = numbers[i].intValue();
645                }
646                return result;
647            } else if (o instanceof short[]) {
648                final short numbers[] = (short[]) o;
649                final int result[] = new int[numbers.length];
650                for (int i = 0; i < numbers.length; i++) {
651                    result[i] = 0xffff & numbers[i];
652                }
653                return result;
654            } else if (o instanceof int[]) {
655                final int numbers[] = (int[]) o;
656                final int result[] = new int[numbers.length];
657                for (int i = 0; i < numbers.length; i++) {
658                    result[i] = numbers[i];
659                }
660                return result;
661            }
662    
663            throw new ImageReadException("Unknown value: " + o + " for: "
664                    + tagInfo.getDescription());
665            // return null;
666        }
