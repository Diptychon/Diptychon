016    package org.apache.commons.imaging.formats.jpeg.segments;
017    
018    import java.io.ByteArrayInputStream;
019    import java.io.IOException;
020    import java.io.InputStream;
021    import java.util.ArrayList;
022    import java.util.List;
023    
024    import org.apache.commons.imaging.ImageReadException;
025    
026    public class DqtSegment extends Segment {
027        public final List<QuantizationTable> quantizationTables = new ArrayList<QuantizationTable>();
028    
029        public static class QuantizationTable {
030            public final int precision;
031            public final int destinationIdentifier;
032            public final int[] elements;
033    
034            public QuantizationTable(final int precision, final int destinationIdentifier,
035                    final int[] elements) {
036                this.precision = precision;
037                this.destinationIdentifier = destinationIdentifier;
038                this.elements = elements;
039            }
040        }
041    
042        public DqtSegment(final int marker, final byte[] segmentData)
043                throws ImageReadException, IOException {
044            this(marker, segmentData.length, new ByteArrayInputStream(segmentData));
045        }
046    
047        public DqtSegment(final int marker, int length, final InputStream is)
048                throws ImageReadException, IOException {
049            super(marker, length);
050    
051            while (length > 0) {
052                final int precisionAndDestination = readByte(
053                        "QuantizationTablePrecisionAndDestination", is,
054                        "Not a Valid JPEG File");
055                length--;
056                final int precision = (precisionAndDestination >> 4) & 0xf;
057                final int destinationIdentifier = precisionAndDestination & 0xf;
058    
059                final int[] elements = new int[64];
060                for (int i = 0; i < 64; i++) {
061                    if (precision == 0) {
062                        elements[i] = 0xff & readByte("QuantizationTableElement",
063                                is, "Not a Valid JPEG File");
064                        length--;
065                    } else if (precision == 1) {
066                        elements[i] = read2Bytes("QuantizationTableElement", is,
067                                "Not a Valid JPEG File");
068                        length -= 2;
069                    } else {
070                        throw new ImageReadException(
071                                "Quantization table precision '" + precision
072                                        + "' is invalid");
073                    }
074                }
075    
076                quantizationTables.add(new QuantizationTable(precision,
077                        destinationIdentifier, elements));
078            }
079        }
080    
081        @Override
082        public String getDescription() {
083            return "DQT (" + getSegmentType() + ")";
084        }
085    }
