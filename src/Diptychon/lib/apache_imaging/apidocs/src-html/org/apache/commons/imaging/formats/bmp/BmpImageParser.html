087        private static final byte BMP_HEADER_SIGNATURE[] = { 0x42, 0x4d, };
088    
089        private BmpHeaderInfo readBmpHeaderInfo(final InputStream is,
090                final FormatCompliance formatCompliance, final boolean verbose)
091                throws ImageReadException, IOException {
092            final byte identifier1 = readByte("Identifier1", is, "Not a Valid BMP File");
093            final byte identifier2 = readByte("Identifier2", is, "Not a Valid BMP File");
094    
095            if (formatCompliance != null) {
096                formatCompliance.compare_bytes("Signature", BMP_HEADER_SIGNATURE,
097                        new byte[] { identifier1, identifier2, });
098            }
099    
100            final int fileSize = read4Bytes("File Size", is, "Not a Valid BMP File");
101            final int reserved = read4Bytes("Reserved", is, "Not a Valid BMP File");
102            final int bitmapDataOffset = read4Bytes("Bitmap Data Offset", is,
103                    "Not a Valid BMP File");
104    
105            final int bitmapHeaderSize = read4Bytes("Bitmap Header Size", is,
106                    "Not a Valid BMP File");
107            int width = 0;
108            int height = 0;
109            int planes = 0;
110            int bitsPerPixel = 0;
111            int compression = 0;
112            int bitmapDataSize = 0;
113            int hResolution = 0;
114            int vResolution = 0;
115            int colorsUsed = 0;
116            int colorsImportant = 0;
117            int redMask = 0;
118            int greenMask = 0;
119            int blueMask = 0;
120            int alphaMask = 0;
121            int colorSpaceType = 0;
122            final BmpHeaderInfo.ColorSpace colorSpace = new BmpHeaderInfo.ColorSpace();
123            colorSpace.red = new BmpHeaderInfo.ColorSpaceCoordinate();
124            colorSpace.green = new BmpHeaderInfo.ColorSpaceCoordinate();
125            colorSpace.blue = new BmpHeaderInfo.ColorSpaceCoordinate();
126            int gammaRed = 0;
127            int gammaGreen = 0;
128            int gammaBlue = 0;
129            int intent = 0;
130            int profileData = 0;
131            int profileSize = 0;
132            int reservedV5 = 0;
133    
134            if (bitmapHeaderSize >= 40) {
135                // BITMAPINFOHEADER
136                width = read4Bytes("Width", is, "Not a Valid BMP File");
137                height = read4Bytes("Height", is, "Not a Valid BMP File");
138                planes = read2Bytes("Planes", is, "Not a Valid BMP File");
139                bitsPerPixel = read2Bytes("Bits Per Pixel", is,
140                        "Not a Valid BMP File");
141                compression = read4Bytes("Compression", is, "Not a Valid BMP File");
142                bitmapDataSize = read4Bytes("Bitmap Data Size", is,
143                        "Not a Valid BMP File");
144                hResolution = read4Bytes("HResolution", is, "Not a Valid BMP File");
145                vResolution = read4Bytes("VResolution", is, "Not a Valid BMP File");
146                colorsUsed = read4Bytes("ColorsUsed", is, "Not a Valid BMP File");
147                colorsImportant = read4Bytes("ColorsImportant", is,
148                        "Not a Valid BMP File");
149                if (bitmapHeaderSize >= 52 || compression == BI_BITFIELDS) {
150                    // 52 = BITMAPV2INFOHEADER, now undocumented
151                    // see http://en.wikipedia.org/wiki/BMP_file_format
152                    redMask = read4Bytes("RedMask", is, "Not a Valid BMP File");
153                    greenMask = read4Bytes("GreenMask", is, "Not a Valid BMP File");
154                    blueMask = read4Bytes("BlueMask", is, "Not a Valid BMP File");
155                }
156                if (bitmapHeaderSize >= 56) {
157                    // 56 = the now undocumented BITMAPV3HEADER sometimes used by
158                    // Photoshop
159                    // see http://forums.adobe.com/thread/751592?tstart=1
160                    alphaMask = read4Bytes("AlphaMask", is, "Not a Valid BMP File");
161                }
162                if (bitmapHeaderSize >= 108) {
163                    // BITMAPV4HEADER
164                    colorSpaceType = read4Bytes("ColorSpaceType", is,
165                            "Not a Valid BMP File");
166                    colorSpace.red.x = read4Bytes("ColorSpaceRedX", is,
167                            "Not a Valid BMP File");
168                    colorSpace.red.y = read4Bytes("ColorSpaceRedY", is,
169                            "Not a Valid BMP File");
170                    colorSpace.red.z = read4Bytes("ColorSpaceRedZ", is,
171                            "Not a Valid BMP File");
172                    colorSpace.green.x = read4Bytes("ColorSpaceGreenX", is,
173                            "Not a Valid BMP File");
174                    colorSpace.green.y = read4Bytes("ColorSpaceGreenY", is,
175                            "Not a Valid BMP File");
176                    colorSpace.green.z = read4Bytes("ColorSpaceGreenZ", is,
177                            "Not a Valid BMP File");
178                    colorSpace.blue.x = read4Bytes("ColorSpaceBlueX", is,
179                            "Not a Valid BMP File");
180                    colorSpace.blue.y = read4Bytes("ColorSpaceBlueY", is,
181                            "Not a Valid BMP File");
182                    colorSpace.blue.z = read4Bytes("ColorSpaceBlueZ", is,
183                            "Not a Valid BMP File");
184                    gammaRed = read4Bytes("GammaRed", is, "Not a Valid BMP File");
185                    gammaGreen = read4Bytes("GammaGreen", is,
186                            "Not a Valid BMP File");
187                    gammaBlue = read4Bytes("GammaBlue", is, "Not a Valid BMP File");
188                }
189                if (bitmapHeaderSize >= 124) {
190                    // BITMAPV5HEADER
191                    intent = read4Bytes("Intent", is, "Not a Valid BMP File");
192                    profileData = read4Bytes("ProfileData", is,
193                            "Not a Valid BMP File");
194                    profileSize = read4Bytes("ProfileSize", is,
195                            "Not a Valid BMP File");
196                    reservedV5 = read4Bytes("Reserved", is, "Not a Valid BMP File");
197                }
198            } else {
199                throw new ImageReadException("Invalid/unsupported BMP file");
200            }
201    
202            if (verbose) {
203                this.debugNumber("identifier1", identifier1, 1);
204                this.debugNumber("identifier2", identifier2, 1);
205                this.debugNumber("fileSize", fileSize, 4);
206                this.debugNumber("reserved", reserved, 4);
207                this.debugNumber("bitmapDataOffset", bitmapDataOffset, 4);
208                this.debugNumber("bitmapHeaderSize", bitmapHeaderSize, 4);
209                this.debugNumber("width", width, 4);
210                this.debugNumber("height", height, 4);
211                this.debugNumber("planes", planes, 2);
212                this.debugNumber("bitsPerPixel", bitsPerPixel, 2);
213                this.debugNumber("compression", compression, 4);
214                this.debugNumber("bitmapDataSize", bitmapDataSize, 4);
215                this.debugNumber("hResolution", hResolution, 4);
216                this.debugNumber("vResolution", vResolution, 4);
217                this.debugNumber("colorsUsed", colorsUsed, 4);
218                this.debugNumber("colorsImportant", colorsImportant, 4);
219                if (bitmapHeaderSize >= 52 || compression == BI_BITFIELDS) {
220                    this.debugNumber("redMask", redMask, 4);
221                    this.debugNumber("greenMask", greenMask, 4);
222                    this.debugNumber("blueMask", blueMask, 4);
223                }
224                if (bitmapHeaderSize >= 56) {
225                    this.debugNumber("alphaMask", alphaMask, 4);
226                }
227                if (bitmapHeaderSize >= 108) {
228                    this.debugNumber("colorSpaceType", colorSpaceType, 4);
229                    this.debugNumber("colorSpace.red.x", colorSpace.red.x);
230                    this.debugNumber("colorSpace.red.y", colorSpace.red.y);
231                    this.debugNumber("colorSpace.red.z", colorSpace.red.z);
232                    this.debugNumber("colorSpace.green.x", colorSpace.green.x);
233                    this.debugNumber("colorSpace.green.y", colorSpace.green.y);
234                    this.debugNumber("colorSpace.green.z", colorSpace.green.z);
235                    this.debugNumber("colorSpace.blue.x", colorSpace.blue.x);
236                    this.debugNumber("colorSpace.blue.y", colorSpace.blue.y);
237                    this.debugNumber("colorSpace.blue.z", colorSpace.blue.z);
238                    this.debugNumber("gammaRed", gammaRed, 4);
239                    this.debugNumber("gammaGreen", gammaGreen, 4);
240                    this.debugNumber("gammaBlue", gammaBlue, 4);
241                }
242                if (bitmapHeaderSize >= 124) {
243                    this.debugNumber("intent", intent, 4);
244                    this.debugNumber("profileData", profileData, 4);
245                    this.debugNumber("profileSize", profileSize, 4);
246                    this.debugNumber("reservedV5", reservedV5, 4);
247                }
248            }
249    
250            final BmpHeaderInfo result = new BmpHeaderInfo(identifier1, identifier2,
251                    fileSize, reserved, bitmapDataOffset, bitmapHeaderSize, width,
252                    height, planes, bitsPerPixel, compression, bitmapDataSize,
253                    hResolution, vResolution, colorsUsed, colorsImportant, redMask,
254                    greenMask, blueMask, alphaMask, colorSpaceType, colorSpace,
255                    gammaRed, gammaGreen, gammaBlue, intent, profileData,
256                    profileSize, reservedV5);
257            return result;
258        }

260        private static final int BI_RGB = 0;
261        private static final int BI_RLE4 = 2;
262        private static final int BI_RLE8 = 1;
263        private static final int BI_BITFIELDS = 3;

265        private byte[] getRLEBytes(final InputStream is, final int RLESamplesPerByte)
266                throws IOException {
267            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
268    
269            // this.setDebug(true);
270    
271            boolean done = false;
272            while (!done) {
273                final int a = 0xff & this.readByte("RLE a", is, "BMP: Bad RLE");
274                baos.write(a);
275                final int b = 0xff & this.readByte("RLE b", is, "BMP: Bad RLE");
276                baos.write(b);
277    
278                if (a == 0) {
279                    switch (b) {
280                    case 0: // EOL
281                        break;
282                    case 1: // EOF
283                        // System.out.println("xXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX"
284                        // );
285                        done = true;
286                        break;
287                    case 2: {
288                        // System.out.println("xXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX"
289                        // );
290                        final int c = 0xff & this.readByte("RLE c", is, "BMP: Bad RLE");
291                        baos.write(c);
292                        final int d = 0xff & this.readByte("RLE d", is, "BMP: Bad RLE");
293                        baos.write(d);
294    
295                    }
296                        break;
297                    default: {
298                        int size = b / RLESamplesPerByte;
299                        if ((b % RLESamplesPerByte) > 0) {
300                            size++;
301                        }
302                        if ((size % 2) != 0) {
303                            size++;
304                        }
305    
306                        // System.out.println("b: " + b);
307                        // System.out.println("size: " + size);
308                        // System.out.println("RLESamplesPerByte: " +
309                        // RLESamplesPerByte);
310                        // System.out.println("xXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX"
311                        // );
312                        final byte bytes[] = this.readByteArray("bytes", size, is,
313                                "RLE: Absolute Mode");
314                        baos.write(bytes);
315                    }
316                        break;
317                    }
318                }
319            }
320    
321            return baos.toByteArray();
322        }

324        private ImageContents readImageContents(final InputStream is,
325                final FormatCompliance formatCompliance, final boolean verbose)
326                throws ImageReadException, IOException {
327            final BmpHeaderInfo bhi = readBmpHeaderInfo(is, formatCompliance, verbose);
328    
329            int colorTableSize = bhi.colorsUsed;
330            if (colorTableSize == 0) {
331                colorTableSize = (1 << bhi.bitsPerPixel);
332            }
333    
334            if (verbose) {
335                this.debugNumber("ColorsUsed", bhi.colorsUsed, 4);
336                this.debugNumber("BitsPerPixel", bhi.bitsPerPixel, 4);
337                this.debugNumber("ColorTableSize", colorTableSize, 4);
338                this.debugNumber("bhi.colorsUsed", bhi.colorsUsed, 4);
339                this.debugNumber("Compression", bhi.compression, 4);
340            }
341    
342            // A palette is always valid, even for images that don't need it
343            // (like 32 bpp), it specifies the "optimal color palette" for
344            // when the image is displayed on a <= 256 color graphics card.
345            int paletteLength;
346            int rleSamplesPerByte = 0;
347            boolean rle = false;
348    
349            switch (bhi.compression) {
350            case BI_RGB:
351                if (verbose) {
352                    System.out.println("Compression: BI_RGB");
353                }
354                if (bhi.bitsPerPixel <= 8) {
355                    paletteLength = 4 * colorTableSize;
356                } else {
357                    paletteLength = 0;
358                }
359                // BytesPerPaletteEntry = 0;
360                // System.out.println("Compression: BI_RGBx2: " + bhi.BitsPerPixel);
361                // System.out.println("Compression: BI_RGBx2: " + (bhi.BitsPerPixel
362                // <= 16));
363                break;
364    
365            case BI_RLE4:
366                if (verbose) {
367                    System.out.println("Compression: BI_RLE4");
368                }
369                paletteLength = 4 * colorTableSize;
370                rleSamplesPerByte = 2;
371                // ExtraBitsPerPixel = 4;
372                rle = true;
373                // // BytesPerPixel = 2;
374                // // BytesPerPaletteEntry = 0;
375                break;
376            //
377            case BI_RLE8:
378                if (verbose) {
379                    System.out.println("Compression: BI_RLE8");
380                }
381                paletteLength = 4 * colorTableSize;
382                rleSamplesPerByte = 1;
383                // ExtraBitsPerPixel = 8;
384                rle = true;
385                // BytesPerPixel = 2;
386                // BytesPerPaletteEntry = 0;
387                break;
388            //
389            case BI_BITFIELDS:
390                if (verbose) {
391                    System.out.println("Compression: BI_BITFIELDS");
392                }
393                if (bhi.bitsPerPixel <= 8) {
394                    paletteLength = 4 * colorTableSize;
395                } else {
396                    paletteLength = 0;
397                }
398                // BytesPerPixel = 2;
399                // BytesPerPaletteEntry = 4;
400                break;
401    
402            default:
403                throw new ImageReadException("BMP: Unknown Compression: "
404                        + bhi.compression);
405            }
406    
407            byte colorTable[] = null;
408            if (paletteLength > 0) {
409                colorTable = this.readByteArray("ColorTable", paletteLength, is,
410                        "Not a Valid BMP File");
411            }
412    
413            if (verbose) {
414                this.debugNumber("paletteLength", paletteLength, 4);
415                System.out.println("ColorTable: "
416                        + ((colorTable == null) ? "null" : "" + colorTable.length));
417            }
418    
419            final int pixelCount = bhi.width * bhi.height;
420    
421            int imageLineLength = ((((bhi.bitsPerPixel) * bhi.width) + 7) / 8);
422    
423            if (verbose) {
424                // this.debugNumber("Total BitsPerPixel",
425                // (ExtraBitsPerPixel + bhi.BitsPerPixel), 4);
426                // this.debugNumber("Total Bit Per Line",
427                // ((ExtraBitsPerPixel + bhi.BitsPerPixel) * bhi.Width), 4);
428                // this.debugNumber("ExtraBitsPerPixel", ExtraBitsPerPixel, 4);
429                this.debugNumber("bhi.Width", bhi.width, 4);
430                this.debugNumber("bhi.Height", bhi.height, 4);
431                this.debugNumber("ImageLineLength", imageLineLength, 4);
432                // this.debugNumber("imageDataSize", imageDataSize, 4);
433                this.debugNumber("PixelCount", pixelCount, 4);
434            }
435            // int ImageLineLength = BytesPerPixel * bhi.Width;
436            while ((imageLineLength % 4) != 0) {
437                imageLineLength++;
438            }
439    
440            final int headerSize = BITMAP_FILE_HEADER_SIZE
441                    + bhi.bitmapHeaderSize
442                    + (bhi.bitmapHeaderSize == 40
443                            && bhi.compression == BI_BITFIELDS ? 3 * 4 : 0);
444            final int expectedDataOffset = headerSize + paletteLength;
445    
446            if (verbose) {
447                this.debugNumber("bhi.BitmapDataOffset", bhi.bitmapDataOffset, 4);
448                this.debugNumber("expectedDataOffset", expectedDataOffset, 4);
449            }
450            final int extraBytes = bhi.bitmapDataOffset - expectedDataOffset;
451            if (extraBytes < 0) {
452                throw new ImageReadException("BMP has invalid image data offset: "
453                        + bhi.bitmapDataOffset + " (expected: "
454                        + expectedDataOffset + ", paletteLength: " + paletteLength
455                        + ", headerSize: " + headerSize + ")");
456            } else if (extraBytes > 0) {
457                this.readByteArray("BitmapDataOffset", extraBytes, is,
458                        "Not a Valid BMP File");
459            }
460    
461            final int imageDataSize = bhi.height * imageLineLength;
462    
463            if (verbose) {
464                this.debugNumber("imageDataSize", imageDataSize, 4);
465            }
466    
467            byte imageData[];
468            if (rle) {
469                imageData = getRLEBytes(is, rleSamplesPerByte);
470            } else {
471                imageData = this.readByteArray("ImageData", imageDataSize, is,
472                        "Not a Valid BMP File");
473            }
474    
475            if (verbose) {
476                this.debugNumber("ImageData.length", imageData.length, 4);
477            }
478    
479            PixelParser pixelParser;
480    
481            switch (bhi.compression) {
482            case BI_RLE4:
483            case BI_RLE8:
484                pixelParser = new PixelParserRle(bhi, colorTable, imageData);
485                break;
486            case BI_RGB:
487                pixelParser = new PixelParserRgb(bhi, colorTable, imageData);
488                break;
489            case BI_BITFIELDS:
490                pixelParser = new PixelParserBitFields(bhi, colorTable, imageData);
491                break;
492            default:
493                throw new ImageReadException("BMP: Unknown Compression: "
494                        + bhi.compression);
495            }
496    
497            return new ImageContents(bhi, colorTable, imageData, pixelParser);
498        }

500        private BmpHeaderInfo readBmpHeaderInfo(final ByteSource byteSource,
501                final boolean verbose) throws ImageReadException, IOException {
502            InputStream is = null;
503            try {
504                is = byteSource.getInputStream();
505    
506                // readSignature(is);
507                return readBmpHeaderInfo(is, null, verbose);
508            } finally {
509                try {
510                    if (is != null) {
511                        is.close();
512                    }
513                } catch (final Exception e) {
514                    Debug.debug(e);
515                }
516    
517            }
518        }

526        @Override
527        public Dimension getImageSize(final ByteSource byteSource, Map<String,Object> params)
528                throws ImageReadException, IOException {
529            // make copy of params; we'll clear keys as we consume them.
530            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
531    
532            final boolean verbose = ParamMap.getParamBoolean(params, PARAM_KEY_VERBOSE,
533                    false);
534    
535            if (params.containsKey(PARAM_KEY_VERBOSE)) {
536                params.remove(PARAM_KEY_VERBOSE);
537            }
538    
539            if (params.size() > 0) {
540                final Object firstKey = params.keySet().iterator().next();
541                throw new ImageReadException("Unknown parameter: " + firstKey);
542            }
543    
544            final BmpHeaderInfo bhi = readBmpHeaderInfo(byteSource, verbose);
545    
546            if (bhi == null) {
547                throw new ImageReadException("BMP: couldn't read header");
548            }
549    
550            return new Dimension(bhi.width, bhi.height);
551    
552        }

569        private String getBmpTypeDescription(final int Identifier1, final int Identifier2) {
570            if ((Identifier1 == 'B') && (Identifier2 == 'M')) {
571                return "Windows 3.1x, 95, NT,";
572            }
573            if ((Identifier1 == 'B') && (Identifier2 == 'A')) {
574                return "OS/2 Bitmap Array";
575            }
576            if ((Identifier1 == 'C') && (Identifier2 == 'I')) {
577                return "OS/2 Color Icon";
578            }
579            if ((Identifier1 == 'C') && (Identifier2 == 'P')) {
580                return "OS/2 Color Pointer";
581            }
582            if ((Identifier1 == 'I') && (Identifier2 == 'C')) {
583                return "OS/2 Icon";
584            }
585            if ((Identifier1 == 'P') && (Identifier2 == 'T')) {
586                return "OS/2 Pointer";
587            }
588    
589            return "Unknown";
590        }

592        @Override
593        public ImageInfo getImageInfo(final ByteSource byteSource, Map<String,Object> params)
594                throws ImageReadException, IOException {
595            // make copy of params; we'll clear keys as we consume them.
596            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
597    
598            final boolean verbose = ParamMap.getParamBoolean(params, PARAM_KEY_VERBOSE,
599                    false);
600    
601            if (params.containsKey(PARAM_KEY_VERBOSE)) {
602                params.remove(PARAM_KEY_VERBOSE);
603            }
604    
605            if (params.size() > 0) {
606                final Object firstKey = params.keySet().iterator().next();
607                throw new ImageReadException("Unknown parameter: " + firstKey);
608            }
609    
610            InputStream is = null;
611            ImageContents ic = null;
612            try {
613                is = byteSource.getInputStream();
614                ic = readImageContents(is, FormatCompliance.getDefault(), verbose);
615            } finally {
616                if (is != null) {
617                    try {
618                        is.close();
619                    } catch (final IOException ignore) {
620                        Debug.debug(ignore);
621                    }
622                }
623            }
624    
625            if (ic == null) {
626                throw new ImageReadException("Couldn't read BMP Data");
627            }
628    
629            final BmpHeaderInfo bhi = ic.bhi;
630            final byte colorTable[] = ic.colorTable;
631    
632            if (bhi == null) {
633                throw new ImageReadException("BMP: couldn't read header");
634            }
635    
636            final int height = bhi.height;

735        public BufferedImage getBufferedImage(final InputStream inputStream, Map<String,Object> params)
736                throws ImageReadException, IOException {
737            // make copy of params; we'll clear keys as we consume them.
738            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
739    
740            final boolean verbose = ParamMap.getParamBoolean(params, PARAM_KEY_VERBOSE,
741                    false);
742    
743            if (params.containsKey(PARAM_KEY_VERBOSE)) {
744                params.remove(PARAM_KEY_VERBOSE);
745            }
746            if (params.containsKey(BUFFERED_IMAGE_FACTORY)) {
747                params.remove(BUFFERED_IMAGE_FACTORY);
748            }
749    
750            if (params.size() > 0) {
751                final Object firstKey = params.keySet().iterator().next();
752                throw new ImageReadException("Unknown parameter: " + firstKey);
753            }
754    
755            final ImageContents ic = readImageContents(inputStream,
756                    FormatCompliance.getDefault(), verbose);
757            if (ic == null) {
758                throw new ImageReadException("Couldn't read BMP Data");
759            }
760    
761            final BmpHeaderInfo bhi = ic.bhi;
762            // byte colorTable[] = ic.colorTable;
763            // byte imageData[] = ic.imageData;
764    
765            final int width = bhi.width;
766            final int height = bhi.height;
767    
768            if (verbose) {
769                System.out.println("width: " + width);
770                System.out.println("height: " + height);
771                System.out.println("width*height: " + width * height);
772                System.out.println("width*height*4: " + width * height * 4);
773            }
774    
775            final PixelParser pixelParser = ic.pixelParser;
776            final ImageBuilder imageBuilder = new ImageBuilder(width, height, true);
777            pixelParser.processImage(imageBuilder);
778    
779            return imageBuilder.getBufferedImage();
780

783        private static final int BITMAP_FILE_HEADER_SIZE = 14;
784        private static final int BITMAP_INFO_HEADER_SIZE = 40;
785    
786        @Override
787        public void writeImage(final BufferedImage src, final OutputStream os, Map<String,Object> params)
788                throws ImageWriteException, IOException {
789            // make copy of params; we'll clear keys as we consume them.
790            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
791    
792            PixelDensity pixelDensity = null;
793    
794            // clear format key.
795            if (params.containsKey(PARAM_KEY_FORMAT)) {
796                params.remove(PARAM_KEY_FORMAT);
797            }
798            if (params.containsKey(PARAM_KEY_PIXEL_DENSITY)) {
799                pixelDensity = (PixelDensity) params
800                        .remove(PARAM_KEY_PIXEL_DENSITY);
801            }
802            if (params.size() > 0) {
803                final Object firstKey = params.keySet().iterator().next();
804                throw new ImageWriteException("Unknown parameter: " + firstKey);
805            }
806    
807            final SimplePalette palette = new PaletteFactory().makeExactRgbPaletteSimple(
808                    src, 256);
809    
810            BmpWriter writer = null;
811            if (palette == null) {
812                writer = new BmpWriterRgb();
813            } else {
814                writer = new BmpWriterPalette(palette);
815            }
816    
817            final byte imagedata[] = writer.getImageData(src);
818            final BinaryOutputStream bos = new BinaryOutputStream(os, ByteOrder.INTEL);
819    
820            // write BitmapFileHeader
821            os.write(0x42); // B, Windows 3.1x, 95, NT, Bitmap
822            os.write(0x4d); // M
823    
824            final int filesize = BITMAP_FILE_HEADER_SIZE + BITMAP_INFO_HEADER_SIZE + // header
825                    // size
826                    4 * writer.getPaletteSize() + // palette size in bytes
827                    imagedata.length;
828            bos.write4Bytes(filesize);
829    
830            bos.write4Bytes(0); // reserved
831            bos.write4Bytes(BITMAP_FILE_HEADER_SIZE + BITMAP_INFO_HEADER_SIZE
832                    + 4 * writer.getPaletteSize()); // Bitmap Data Offset
833    
834            final int width = src.getWidth();
835            final int height = src.getHeight();
836    
837            // write BitmapInfoHeader
838            bos.write4Bytes(BITMAP_INFO_HEADER_SIZE); // Bitmap Info Header Size
839            bos.write4Bytes(width); // width
840            bos.write4Bytes(height); // height
841            bos.write2Bytes(1); // Number of Planes
842            bos.write2Bytes(writer.getBitsPerPixel()); // Bits Per Pixel
843    
844            bos.write4Bytes(BI_RGB); // Compression
845            bos.write4Bytes(imagedata.length); // Bitmap Data Size
846            bos.write4Bytes(pixelDensity != null ? (int) Math
847                    .round(pixelDensity.horizontalDensityMetres()) : 0); // HResolution
848            bos.write4Bytes(pixelDensity != null ? (int) Math
849                    .round(pixelDensity.verticalDensityMetres()) : 0); // VResolution
850            if (palette == null) {
851                bos.write4Bytes(0); // Colors
852            } else {
853                bos.write4Bytes(palette.length()); // Colors
854            }
855            bos.write4Bytes(0); // Important Colors
856            // bos.write_4_bytes(0); // Compression
857    
858            // write Palette
859            writer.writePalette(bos);
860            // write Image Data
861            bos.writeByteArray(imagedata);
862        }
