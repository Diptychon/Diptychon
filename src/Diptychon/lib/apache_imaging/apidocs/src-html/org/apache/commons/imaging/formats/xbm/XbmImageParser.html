111        private static class XbmHeader {
112            int width;
113            int height;
114            int xHot = -1;
115            int yHot = -1;
116    
117            public XbmHeader(final int width, final int height, final int xHot, final int yHot) {
118                this.width = width;
119                this.height = height;
120                this.xHot = xHot;
121                this.yHot = yHot;
122            }
123    
124            public void dump(final PrintWriter pw) {
125                pw.println("XbmHeader");
126                pw.println("Width: " + width);
127                pw.println("Height: " + height);
128                if (xHot != -1 && yHot != -1) {
129                    pw.println("X hot: " + xHot);
130                    pw.println("Y hot: " + yHot);
131                }
132            }
133        }

140        private XbmHeader readXbmHeader(final ByteSource byteSource)
141                throws ImageReadException, IOException {
142            final XbmParseResult result = parseXbmHeader(byteSource);
143            return result.xbmHeader;
144        }

146        private XbmParseResult parseXbmHeader(final ByteSource byteSource)
147                throws ImageReadException, IOException {
148            InputStream is = null;
149            try {
150                is = byteSource.getInputStream();
151                final Map<String, String> defines = new HashMap<String, String>();
152                final ByteArrayOutputStream preprocessedFile = BasicCParser.preprocess(
153                        is, null, defines);
154                int width = -1;
155                int height = -1;
156                int xHot = -1;
157                int yHot = -1;
158                for (final Entry<String, String> entry : defines.entrySet()) {
159                final String name = entry.getKey();
160                if (name.endsWith("_width")) {
161                width = Integer.parseInt(entry.getValue());
162                } else if (name.endsWith("_height")) {
163                height = Integer.parseInt(entry.getValue());
164                } else if (name.endsWith("_x_hot")) {
165                xHot = Integer.parseInt(entry.getValue());
166                } else if (name.endsWith("_y_hot")) {
167                yHot = Integer.parseInt(entry.getValue());
168                }
169             }
170                if (width == -1) {
171                    throw new ImageReadException("width not found");
172                }
173                if (height == -1) {
174                    throw new ImageReadException("height not found");
175                }
176    
177                final XbmParseResult xbmParseResult = new XbmParseResult();
178                xbmParseResult.cParser = new BasicCParser(new ByteArrayInputStream(
179                        preprocessedFile.toByteArray()));
180                xbmParseResult.xbmHeader = new XbmHeader(width, height, xHot, yHot);
181                return xbmParseResult;
182            } finally {
183                try {
184                    if (is != null) {
185                        is.close();
186                    }
187                } catch (final IOException ignored) {
188                }
189            }
190        }

192        private BufferedImage readXbmImage(final XbmHeader xbmHeader, final BasicCParser cParser)
193                throws ImageReadException, IOException {
194            String token;
195            token = cParser.nextToken();
196            if (token == null || !token.equals("static")) {
197                throw new ImageReadException(
198                        "Parsing XBM file failed, no 'static' token");
199            }
200            token = cParser.nextToken();
201            if (token == null) {
202                throw new ImageReadException(
203                        "Parsing XBM file failed, no 'unsigned' "
204                                + "or 'char' token");
205            }
206            if (token.equals("unsigned")) {
207                token = cParser.nextToken();
208            }
209            if (token == null || !token.equals("char")) {
210                throw new ImageReadException(
211                        "Parsing XBM file failed, no 'char' token");
212            }
213            final String name = cParser.nextToken();
214            if (name == null) {
215                throw new ImageReadException(
216                        "Parsing XBM file failed, no variable name");
217            }
218            if (name.charAt(0) != '_' && !Character.isLetter(name.charAt(0))) {
219                throw new ImageReadException(
220                        "Parsing XBM file failed, variable name "
221                                + "doesn't start with letter or underscore");
222            }
223            for (int i = 0; i < name.length(); i++) {
224                final char c = name.charAt(i);
225                if (!Character.isLetterOrDigit(c) && c != '_') {
226                    throw new ImageReadException(
227                            "Parsing XBM file failed, variable name "
228                                    + "contains non-letter non-digit non-underscore");
229                }
230            }
231            token = cParser.nextToken();
232            if (token == null || !token.equals("[")) {
233                throw new ImageReadException(
234                        "Parsing XBM file failed, no '[' token");
235            }
236            token = cParser.nextToken();
237            if (token == null || !token.equals("]")) {
238                throw new ImageReadException(
239                        "Parsing XBM file failed, no ']' token");
240            }
241            token = cParser.nextToken();
242            if (token == null || !token.equals("=")) {
243                throw new ImageReadException(
244                        "Parsing XBM file failed, no '=' token");
245            }
246            token = cParser.nextToken();
247            if (token == null || !token.equals("{")) {
248                throw new ImageReadException(
249                        "Parsing XBM file failed, no '{' token");
250            }
251    
252            final int rowLength = (xbmHeader.width + 7) / 8;
253            final byte[] imageData = new byte[rowLength * xbmHeader.height];
254            for (int i = 0; i < imageData.length; i++) {
255                token = cParser.nextToken();
256                if (token == null || !token.startsWith("0x")) {
257                    throw new ImageReadException("Parsing XBM file failed, "
258                            + "hex value missing");
259                }
260                if (token.length() > 4) {
261                    throw new ImageReadException("Parsing XBM file failed, "
262                            + "hex value too long");
263                }
264                final int value = Integer.parseInt(token.substring(2), 16);
265                int flipped = 0;
266                for (int j = 0; j < 8; j++) {
267                    if ((value & (1 << j)) != 0) {
268                        flipped |= (0x80 >>> j);
269                    }
270                }
271                imageData[i] = (byte) flipped;
272    
273                token = cParser.nextToken();
274                if (token == null) {
275                    throw new ImageReadException("Parsing XBM file failed, "
276                            + "premature end of file");
277                }
278                if (!token.equals(",")
279                        && (i < (imageData.length - 1) || !token.equals("}"))) {
280                    throw new ImageReadException("Parsing XBM file failed, "
281                            + "punctuation error");
282                }
283            }
284    
285            final int[] palette = { 0xffffff, 0x000000 };
286            final ColorModel colorModel = new IndexColorModel(1, 2, palette, 0, false,
287                    -1, DataBuffer.TYPE_BYTE);
288            final DataBufferByte dataBuffer = new DataBufferByte(imageData,
289                    imageData.length);
290            final WritableRaster raster = WritableRaster.createPackedRaster(dataBuffer,
291                    xbmHeader.width, xbmHeader.height, 1, null);
292            final BufferedImage image = new BufferedImage(colorModel, raster,
293                    colorModel.isAlphaPremultiplied(), new Properties());
294            return image;
295        }

311        private String randomName() {
312            final UUID uuid = UUID.randomUUID();
313            final StringBuilder stringBuilder = new StringBuilder("a");
314            long bits = uuid.getMostSignificantBits();
315            // Long.toHexString() breaks for very big numbers
316            for (int i = 64 - 8; i >= 0; i -= 8) {
317                stringBuilder.append(Integer
318                        .toHexString((int) ((bits >> i) & 0xff)));
319            }
320            bits = uuid.getLeastSignificantBits();
321            for (int i = 64 - 8; i >= 0; i -= 8) {
322                stringBuilder.append(Integer
323                        .toHexString((int) ((bits >> i) & 0xff)));
324            }
325            return stringBuilder.toString();
326        }

328        private String toPrettyHex(final int value) {
329            final String s = Integer.toHexString(0xff & value);
330            if (s.length() == 2) {
331                return "0x" + s;
332            }
333            return "0x0" + s;
334        }

336        @Override
337        public void writeImage(final BufferedImage src, final OutputStream os, Map<String,Object> params)
338                throws ImageWriteException, IOException {
339            // make copy of params; we'll clear keys as we consume them.
340            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
341    
342            // clear format key.
343            if (params.containsKey(PARAM_KEY_FORMAT)) {
344                params.remove(PARAM_KEY_FORMAT);
345            }
346    
347            if (params.size() > 0) {
348                final Object firstKey = params.keySet().iterator().next();
349                throw new ImageWriteException("Unknown parameter: " + firstKey);
350            }
351    
352            final String name = randomName();
353    
354            os.write(("#define " + name + "_width " + src.getWidth() + "\n")
355                    .getBytes("US-ASCII"));
356            os.write(("#define " + name + "_height " + src.getHeight() + "\n")
357                    .getBytes("US-ASCII"));
358            os.write(("static unsigned char " + name + "_bits[] = {")
359                    .getBytes("US-ASCII"));
360    
361            int bitcache = 0;
362            int bits_in_cache = 0;
363            String separator = "\n  ";
364            int written = 0;
365            for (int y = 0; y < src.getHeight(); y++) {
366                for (int x = 0; x < src.getWidth(); x++) {
367                    final int argb = src.getRGB(x, y);
368                    final int red = 0xff & (argb >> 16);
369                    final int green = 0xff & (argb >> 8);
370                    final int blue = 0xff & (argb >> 0);
371                    int sample = (red + green + blue) / 3;
372                    if (sample > 127) {
373                        sample = 0;
374                    } else {
375                        sample = 1;
376                    }
377                    bitcache |= (sample << bits_in_cache);
378                    ++bits_in_cache;
379                    if (bits_in_cache == 8) {
380                        os.write(separator.getBytes("US-ASCII"));
381                        separator = ",";
382                        if (written == 12) {
383                            os.write("\n  ".getBytes("US-ASCII"));
384                            written = 0;
385                        }
386                        os.write(toPrettyHex(bitcache).getBytes("US-ASCII"));
387                        bitcache = 0;
388                        bits_in_cache = 0;
389                        ++written;
390                    }
391                }
392                if (bits_in_cache != 0) {
393                    os.write(separator.getBytes("US-ASCII"));
394                    separator = ",";
395                    if (written == 12) {
396                        os.write("\n  ".getBytes("US-ASCII"));
397                        written = 0;
398                    }
399                    os.write(toPrettyHex(bitcache).getBytes("US-ASCII"));
400                    bitcache = 0;
401                    bits_in_cache = 0;
402                    ++written;
403                }
404            }
405    
406            os.write("\n};\n".getBytes("US-ASCII"));
407        }
