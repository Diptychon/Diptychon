083                throws ImageReadException, IOException {
084            InputStream is = null;
085    
086            try {
087                is = byteSource.getInputStream();
088    
089                return readHeader(is);
090            } finally {
091                try {
092                    if (is != null) {
093                        is.close();
094                    }
095                } catch (final Exception e) {
096                    Debug.debug(e);
097                }
098    
099            }

102        private PsdHeaderInfo readHeader(final InputStream is) throws ImageReadException,
103                IOException {
104            readAndVerifyBytes(is, new byte[] { 56, 66, 80, 83 },
105                    "Not a Valid PSD File");
106    
107            final int Version = read2Bytes("Version", is, "Not a Valid PSD File");
108    
109            final byte Reserved[] = readByteArray("Reserved", 6, is,
110                    "Not a Valid PSD File");
111    
112            final int Channels = read2Bytes("Channels", is, "Not a Valid PSD File");
113            final int Rows = read4Bytes("Rows", is, "Not a Valid PSD File");
114            final int Columns = read4Bytes("Columns", is, "Not a Valid PSD File");
115            final int Depth = read2Bytes("Depth", is, "Not a Valid PSD File");
116            final int Mode = read2Bytes("Mode", is, "Not a Valid PSD File");
117    
118            final PsdHeaderInfo result = new PsdHeaderInfo(Version, Reserved, Channels,
119                    Rows, Columns, Depth, Mode);
120    
121            return result;
122        }

175        private boolean keepImageResourceBlock(final int ID, final int imageResourceIDs[]) {
176            if (imageResourceIDs == null) {
177                return true;
178            }
179    
180            for (final int imageResourceID : imageResourceIDs) {
181                if (ID == imageResourceID) {
182                    return true;
183                }
184            }
185    
186            return false;
187        }

189        private List<ImageResourceBlock> readImageResourceBlocks(final InputStream is,
190                final int imageResourceIDs[], final int maxBlocksToRead, int available)
191                throws ImageReadException, IOException {
192            final List<ImageResourceBlock> result = new ArrayList<ImageResourceBlock>();
193    
194            while (available > 0) {
195                readAndVerifyBytes(is, new byte[] { 56, 66, 73, 77 },
196                        "Not a Valid PSD File");
197                available -= 4;
198    
199                final int id = read2Bytes("ID", is, "Not a Valid PSD File");
200                available -= 2;
201    
202                final int nameLength = readByte("NameLength", is, "Not a Valid PSD File");
203    
204                available -= 1;
205                final byte nameBytes[] = readByteArray("NameData", nameLength, is,
206                        "Not a Valid PSD File");
207                available -= nameLength;
208                if (((nameLength + 1) % 2) != 0) {
209                    final int NameDiscard = readByte("NameDiscard", is,
210                            "Not a Valid PSD File");
211                    available -= 1;
212                }
213                // String Name = readPString("Name", 6, is, "Not a Valid PSD File");
214                final int DataSize = read4Bytes("Size", is, "Not a Valid PSD File");
215                available -= 4;
216                // int ActualDataSize = ((DataSize % 2) == 0)
217                // ? DataSize
218                // : DataSize + 1; // pad to make even
219    
220                final byte Data[] = readByteArray("Data", DataSize, is,
221                        "Not a Valid PSD File");
222                available -= DataSize;
223    
224                if ((DataSize % 2) != 0) {
225                    final int DataDiscard = readByte("DataDiscard", is,
226                            "Not a Valid PSD File");
227                    available -= 1;
228                }
229    
230                if (keepImageResourceBlock(id, imageResourceIDs)) {
231                    result.add(new ImageResourceBlock(id, nameBytes, Data));
232    
233                    if ((maxBlocksToRead >= 0)
234                            && (result.size() >= maxBlocksToRead)) {
235                        return result;
236                    }
237                }
238                // debugNumber("ID", ID, 2);
239    
240            }
241    
242            return result;
243        }

245        private List<ImageResourceBlock> readImageResourceBlocks(
246                final ByteSource byteSource, final int imageResourceIDs[], final int maxBlocksToRead)
247                throws ImageReadException, IOException {
248            InputStream is = null;
249    
250            try {
251                is = byteSource.getInputStream();
252    
253                final ImageContents imageContents = readImageContents(is);
254    
255                is.close();
256    
257                is = this.getInputStream(byteSource, PSD_SECTION_IMAGE_RESOURCES);
258                final byte ImageResources[] = readByteArray("ImageResources",
259                        imageContents.ImageResourcesLength, is,
260                        "Not a Valid PSD File");
261    
262                return readImageResourceBlocks(ImageResources, imageResourceIDs,
263                        maxBlocksToRead);
264            } finally {
265                try {
266                    if (is != null) {
267                        is.close();
268                    }
269                } catch (final Exception e) {
270                    Debug.debug(e);
271                }
272    
273            }
274        }

277        private static final int PSD_SECTION_COLOR_MODE = 1;
278        private static final int PSD_SECTION_IMAGE_RESOURCES = 2;

284        private InputStream getInputStream(final ByteSource byteSource, final int section)
285                throws ImageReadException, IOException {
286            InputStream is = null;
287            boolean notFound = false;
288            try {
289                is = byteSource.getInputStream();
290    
291                if (section == PSD_SECTION_HEADER) {
292                    return is;
293                }
294    
295                skipBytes(is, PSD_HEADER_LENGTH);
296                // is.skip(kHeaderLength);
297    
298                final int ColorModeDataLength = read4Bytes("ColorModeDataLength", is,
299                        "Not a Valid PSD File");
300    
301                if (section == PSD_SECTION_COLOR_MODE) {
302                    return is;
303                }
304    
305                skipBytes(is, ColorModeDataLength);
306                // byte ColorModeData[] = readByteArray("ColorModeData",
307                // ColorModeDataLength, is, "Not a Valid PSD File");
308    
309                final int ImageResourcesLength = read4Bytes("ImageResourcesLength", is,
310                        "Not a Valid PSD File");
311    
312                if (section == PSD_SECTION_IMAGE_RESOURCES) {
313                    return is;
314                }
315    
316                skipBytes(is, ImageResourcesLength);
317                // byte ImageResources[] = readByteArray("ImageResources",
318                // ImageResourcesLength, is, "Not a Valid PSD File");
319    
320                final int LayerAndMaskDataLength = read4Bytes("LayerAndMaskDataLength",
321                        is, "Not a Valid PSD File");
322    
323                if (section == PSD_SECTION_LAYER_AND_MASK_DATA) {
324                    return is;
325                }
326    
327                skipBytes(is, LayerAndMaskDataLength);
328                // byte LayerAndMaskData[] = readByteArray("LayerAndMaskData",
329                // LayerAndMaskDataLength, is, "Not a Valid PSD File");
330    
331                read2Bytes("Compression", is, "Not a Valid PSD File");
332    
333                // byte ImageData[] = readByteArray("ImageData",
334                // LayerAndMaskDataLength, is, "Not a Valid PSD File");
335    
336                if (section == PSD_SECTION_IMAGE_DATA) {
337                    return is;
338                }
339                notFound = true;
340            } finally {
341                if (notFound && is != null) {
342                    try {
343                        is.close();
344                    } catch (final IOException ignore) {
345                    }
346                }
347            }
348            throw new ImageReadException("getInputStream: Unknown Section: "
349                    + section);
350        }

364                skipBytes(is, PSD_HEADER_LENGTH);
365

368    
369                if (section == PSD_SECTION_COLOR_MODE) {

372                }
373

376                // ColorModeDataLength, is, "Not a Valid PSD File");
377    
378                final int ImageResourcesLength = read4Bytes("ImageResourcesLength", is,
379                        "Not a Valid PSD File");
380    
381                if (section == PSD_SECTION_IMAGE_RESOURCES) {
382                    return readByteArray("ImageResources", ImageResourcesLength,
383                            is, "Not a Valid PSD File");
384                }
385    
386                skipBytes(is, ImageResourcesLength);

410                // "Not a Valid PSD File");

416                    }
417                } catch (final Exception e) {
418                    Debug.debug(e);

451        public byte[] getICCProfileBytes(final ByteSource byteSource, final Map<String,Object> params)
452                throws ImageReadException, IOException {
453            final List<ImageResourceBlock> blocks = readImageResourceBlocks(byteSource,
454                    new int[] { IMAGE_RESOURCE_ID_ICC_PROFILE, }, 1);
455    
456            if ((blocks == null) || (blocks.size() < 1)) {
457                return null;
458            }
459    
460            final ImageResourceBlock irb = blocks.get(0);
461            final byte bytes[] = irb.data;
462            if ((bytes == null) || (bytes.length < 1)) {
463                return null;
464            }
465            return bytes;
466        }

495        private int getChannelsPerMode(final int mode) {
496            switch (mode) {
497            case 0: // Bitmap
498                return 1;
499            case 1: // Grayscale
500                return 1;
501            case 2: // Indexed
502                return -1;
503            case 3: // RGB
504                return 3;
505            case 4: // CMYK
506                return 4;
507            case 7: // Multichannel
508                return -1;
509            case 8: // Duotone
510                return -1;
511            case 9: // Lab
512                return 4;
513            default:
514                return -1;
515    
516            }
517        }

519        @Override
520        public ImageInfo getImageInfo(final ByteSource byteSource, final Map<String,Object> params)
521                throws ImageReadException, IOException {
522            final ImageContents imageContents = readImageContents(byteSource);
523            // ImageContents imageContents = readImage(byteSource, false);
524    
525            if (imageContents == null) {
526                throw new ImageReadException("PSD: Couldn't read blocks");
527            }
528    
529            final PsdHeaderInfo header = imageContents.header;
530            if (header == null) {
531                throw new ImageReadException("PSD: Couldn't read Header");
532            }
533    
534            final int Width = header.Columns;
535            final int Height = header.Rows;
536    
537            final List<String> Comments = new ArrayList<String>();
538            // TODO: comments...
539    
540            int BitsPerPixel = header.Depth * getChannelsPerMode(header.Mode);
541            // System.out.println("header.Depth: " + header.Depth);
542            // System.out.println("header.Mode: " + header.Mode);
543            // System.out.println("getChannelsPerMode(header.Mode): " +
544            // getChannelsPerMode(header.Mode));
545            if (BitsPerPixel < 0) {
546                BitsPerPixel = 0;
547            }
548            final ImageFormat Format = ImageFormat.IMAGE_FORMAT_PSD;
549            final String FormatName = "Photoshop";
550            final String MimeType = "image/x-photoshop";
551            // we ought to count images, but don't yet.
552            final int NumberOfImages = -1;
553            // not accurate ... only reflects first
554            final boolean isProgressive = false;
555    
556            final int PhysicalWidthDpi = 72;
557            final float PhysicalWidthInch = (float) ((double) Width / (double) PhysicalWidthDpi);
558            final int PhysicalHeightDpi = 72;
559            final float PhysicalHeightInch = (float) ((double) Height / (double) PhysicalHeightDpi);
560    
561            final String FormatDetails = "Psd";
562    
563            final boolean isTransparent = false; // TODO: inaccurate.
564            final boolean usesPalette = header.Mode == COLOR_MODE_INDEXED;
565            final int ColorType = ImageInfo.COLOR_TYPE_UNKNOWN;
566    
567            String compressionAlgorithm;
568            switch (imageContents.Compression) {
569            case 0:
570                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_NONE;
571                break;
572            case 1:
573                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_PSD;
574                break;
575            default:
576                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_UNKNOWN;
577            }
578    
579            final ImageInfo result = new ImageInfo(FormatDetails, BitsPerPixel, Comments,
580                    Format, FormatName, Height, MimeType, NumberOfImages,
581                    PhysicalHeightDpi, PhysicalHeightInch, PhysicalWidthDpi,
582                    PhysicalWidthInch, Width, isProgressive, isTransparent,
583                    usesPalette, ColorType, compressionAlgorithm);
584    
585            return result;
586        }

650                throws ImageReadException, IOException {
651            final ImageContents imageContents = readImageContents(byteSource);
652            // ImageContents imageContents = readImage(byteSource, false);
653    
654            if (imageContents == null) {
655                throw new ImageReadException("PSD: Couldn't read blocks");
656            }
657    
658            final PsdHeaderInfo header = imageContents.header;
659            if (header == null) {
660                throw new ImageReadException("PSD: Couldn't read Header");
661            }
662    
663            // ImageDescriptor id = (ImageDescriptor)
664            // findBlock(fImageContents.blocks,
665            // kImageSeperator);
666            // if (id == null)
667            // throw new ImageReadException("PSD: Couldn't read Image Descriptor");
668            // GraphicControlExtension gce = (GraphicControlExtension) findBlock(
669            // fImageContents.blocks, kGraphicControlExtension);
670    
671            readImageResourceBlocks(byteSource,
672            // fImageContents.ImageResources,
673                    null, -1);
674    
675            final int width = header.Columns;
676            final int height = header.Rows;
677            // int height = header.Columns;
678    
679            // int transfer_type;
680    
681            // transfer_type = DataBuffer.TYPE_BYTE;
682    
683            final boolean hasAlpha = false;
684            final BufferedImage result = getBufferedImageFactory(params)
685                    .getColorBufferedImage(width, height, hasAlpha);
686    
687            DataParser dataParser;
688            switch (imageContents.header.Mode) {
689            case 0: // bitmap
690                dataParser = new DataParserBitmap();
691                break;
692            case 1:
693            case 8: // Duotone=8;
694                dataParser = new DataParserGrayscale();
695                break;
696            case 3:
697                dataParser = new DataParserRgb();
698                break;
699            case 4:
700                dataParser = new DataParserCmyk();
701                break;
702            case 9:
703                dataParser = new DataParserLab();
704                break;
705            case COLOR_MODE_INDEXED:
706            // case 2 : // Indexed=2;
707            {
708    
709                final byte ColorModeData[] = getData(byteSource, PSD_SECTION_COLOR_MODE);
710    
711                // ImageResourceBlock block = findImageResourceBlock(blocks,
712                // 0x03EB);
713                // if (block == null)
714                // throw new ImageReadException(
715                // "Missing: Indexed Color Image Resource Block");
716    
717                dataParser = new DataParserIndexed(ColorModeData);
718                break;
719            }
720            case 7: // Multichannel=7;
721                // fDataParser = new DataParserStub();
722                // break;
723    
724                // case 1 :
725                // fDataReader = new CompressedDataReader();
726                // break;
727            default:
728                throw new ImageReadException("Unknown Mode: "
729                        + imageContents.header.Mode);
730            }
731            DataReader fDataReader;
732            switch (imageContents.Compression) {
733            case 0:
734                fDataReader = new UncompressedDataReader(dataParser);
735                break;
736            case 1:
737                fDataReader = new CompressedDataReader(dataParser);
738                break;
739            default:
740                throw new ImageReadException("Unknown Compression: "
741                        + imageContents.Compression);
742            }
743            
744            InputStream is = null;
745    
746            try {
747                is = getInputStream(byteSource, PSD_SECTION_IMAGE_DATA);
748                fDataReader.readData(is, result, imageContents, this);
749    
750                fDataReader.dump();
751                // is.
752                // ImageContents imageContents = readImageContents(is);
753                // return imageContents;
754            } finally {
755                try {
756                    if (is != null) {
757                        is.close();
758                    }
759                } catch (final Exception e) {
760                    Debug.debug(e);
761                }
762    
763            }
764    
765            return result;

773        /**
774         * Extracts embedded XML metadata as XML string.
775         * <p>
776         * 
777         * @param byteSource
778         *            File containing image data.
779         * @param params
780         *            Map of optional parameters, defined in SanselanConstants.
781         * @return Xmp Xml as String, if present. Otherwise, returns null.
782         */
783        @Override
784        public String getXmpXml(final ByteSource byteSource, final Map<String,Object> params)
785                throws ImageReadException, IOException {
786    
787            final ImageContents imageContents = readImageContents(byteSource);
788    
789            if (imageContents == null) {
790                throw new ImageReadException("PSD: Couldn't read blocks");
791            }
792    
793            final PsdHeaderInfo header = imageContents.header;
794            if (header == null) {
795                throw new ImageReadException("PSD: Couldn't read Header");
796            }
797    
798            final List<ImageResourceBlock> blocks = readImageResourceBlocks(byteSource,
799                    new int[] { IMAGE_RESOURCE_ID_XMP, }, -1);
800    
801            if ((blocks == null) || (blocks.size() < 1)) {
802                return null;
803            }
804    
805            final List<ImageResourceBlock> xmpBlocks = new ArrayList<ImageResourceBlock>();
806            if (false) {
807                // TODO: for PSD 7 and later, verify "XMP" name.
808                for (int i = 0; i < blocks.size(); i++) {
809                    final ImageResourceBlock block = blocks.get(i);
810                    if (!block.getName().equals(BLOCK_NAME_XMP)) {
811                        continue;
812                    }
813                    xmpBlocks.add(block);
814                }
815            } else {
816                xmpBlocks.addAll(blocks);
817            }
818    
819            if (xmpBlocks.size() < 1) {
820                return null;
821            }
822            if (xmpBlocks.size() > 1) {
823                throw new ImageReadException(
824                        "PSD contains more than one XMP block.");
825            }
826    
827            final ImageResourceBlock block = xmpBlocks.get(0);
828    
829            try {
830                // segment data is UTF-8 encoded xml.
831                final String xml = new String(block.data, 0, block.data.length, "utf-8");
832                return xml;
833            } catch (final UnsupportedEncodingException e) {
834                throw new ImageReadException("Invalid JPEG XMP Segment.");
835            }
836        }
