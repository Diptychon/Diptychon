017    package org.apache.commons.imaging;
018    
019    /**
020     * Defines constants that may be used in passing options to 
021     * ImageParser read/write implementations, the utility routines
022     * implemented in the Imaging class, and throughout the
023     * Apache Commons Imaging package.  Individual ImageParser 
024     * implementations may define their own format-specific options.
025     */
026    public interface ImagingConstants {
027        /**
028         * Parameter key. Applies to read and write operations.
029         * <p>
030         * Valid values: Boolean.TRUE and Boolean.FALSE.
031         */
032        public static final String PARAM_KEY_VERBOSE = "VERBOSE";
033    
034        /**
035         * Parameter key. Used to hint the filename when reading from a byte array
036         * or InputStream. The filename hint can help disambiguate what file the
037         * image format.
038         * <p>
039         * Applies to read operations.
040         * <p>
041         * Valid values: filename as string
042         * <p>
043         * 
044         * @see java.io.InputStream
045         */
046        public static final String PARAM_KEY_FILENAME = "FILENAME";
047    
048        /**
049         * Parameter key. Used in write operations to indicate desired image format.
050         * <p>
051         * Valid values: Any format defined in ImageFormat, such as
052         * ImageFormat.IMAGE_FORMAT_PNG.
053         * <p>
054         * 
055         * @see org.apache.commons.imaging.ImageFormat
056         */
057        public static final String PARAM_KEY_FORMAT = "FORMAT";
058    
059        /**
060         * Parameter key. Used in write operations to indicate desired compression
061         * algorithm.
062         * <p>
063         * Currently only applies to writing TIFF image files.
064         * <p>
065         * Valid values: TiffConstants.TIFF_COMPRESSION_UNCOMPRESSED,
066         * TiffConstants.TIFF_COMPRESSION_CCITT_1D,
067         * TiffConstants.TIFF_COMPRESSION_LZW,
068         * TiffConstants.TIFF_COMPRESSION_PACKBITS.
069         * <p>
070         * 
071         * @see org.apache.commons.imaging.formats.tiff.constants.TiffConstants
072         */
073        public static final String PARAM_KEY_COMPRESSION = "COMPRESSION";
074    
075        public static final String BUFFERED_IMAGE_FACTORY = "BUFFERED_IMAGE_FACTORY";
076    
077        /**
078         * Parameter key. Indicates whether to read embedded thumbnails.
079         * <p>
080         * Only applies to read EXIF metadata from JPEG/JFIF files.
081         * <p>
082         * Valid values: Boolean.TRUE and Boolean.FALSE.
083         * <p>
084         * 
085         * @see org.apache.commons.imaging.formats.tiff.constants.TiffConstants
086         */
087        public static final String PARAM_KEY_READ_THUMBNAILS = "READ_THUMBNAILS";
088    
089        /**
090         * Parameter key. Indicates whether to throw exceptions when parsing invalid
091         * files, or whether to tolerate small problems.
092         * <p>
093         * Valid values: Boolean.TRUE and Boolean.FALSE. Default value:
094         * Boolean.FALSE.
095         * <p>
096         * 
097         * @see org.apache.commons.imaging.formats.tiff.constants.TiffConstants
098         */
099        public static final String PARAM_KEY_STRICT = "STRICT";
100    
101        /**
102         * Parameter key.
103         * 
104         * Only used when writing images.
105         * <p>
106         * Valid values: TiffOutputSet to write into the image's EXIF metadata.
107         * <p>
108         * 
109         * @see org.apache.commons.imaging.formats.tiff.write.TiffOutputSet
110         */
111        public static final String PARAM_KEY_EXIF = "EXIF";
112    
113        /**
114         * Parameter key.
115         * 
116         * Only used when writing images.
117         * <p>
118         * Valid values: String of XMP XML.
119         * <p>
120         */
121        public static final String PARAM_KEY_XMP_XML = "XMP_XML";
122    
123        /**
124         * Parameter key. Used in write operations to indicate the desired pixel
125         * density (DPI), and/or aspect ratio.
126         * <p>
127         * Valid values: PixelDensity
128         * <p>
129         * 
130         * @see org.apache.commons.imaging.PixelDensity
131         */
132        public static final String PARAM_KEY_PIXEL_DENSITY = "PIXEL_DENSITY";
133    }
