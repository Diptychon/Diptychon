062        protected TiffOutputSummary validateDirectories(final TiffOutputSet outputSet)
063                throws ImageWriteException {
064            final List<TiffOutputDirectory> directories = outputSet.getDirectories();
065    
066            if (1 > directories.size()) { 
067                throw new ImageWriteException("No directories.");
068            }
069    
070            TiffOutputDirectory exifDirectory = null;
071            TiffOutputDirectory gpsDirectory = null;
072            TiffOutputDirectory interoperabilityDirectory = null;
073            TiffOutputField exifDirectoryOffsetField = null;
074            TiffOutputField gpsDirectoryOffsetField = null;
075            TiffOutputField interoperabilityDirectoryOffsetField = null;
076    
077            final List<Integer> directoryIndices = new ArrayList<Integer>();
078            final Map<Integer, TiffOutputDirectory> directoryTypeMap = new HashMap<Integer, TiffOutputDirectory>();
079            for (int i = 0; i < directories.size(); i++) {
080                final TiffOutputDirectory directory = directories.get(i);
081                final int dirType = directory.type;
082                directoryTypeMap.put(dirType, directory);
083                // Debug.debug("validating dirType", dirType + " ("
084                // + directory.getFields().size() + " fields)");
085    
086                if (dirType < 0) {
087                    switch (dirType) {
088                    case DIRECTORY_TYPE_EXIF:
089                        if (exifDirectory != null) {
090                            throw new ImageWriteException(
091                                    "More than one EXIF directory.");
092                        }
093                        exifDirectory = directory;
094                        break;
095    
096                    case DIRECTORY_TYPE_GPS:
097                        if (gpsDirectory != null) {
098                            throw new ImageWriteException(
099                                    "More than one GPS directory.");
100                        }
101                        gpsDirectory = directory;
102                        break;
103    
104                    case DIRECTORY_TYPE_INTEROPERABILITY:
105                        if (interoperabilityDirectory != null) {
106                            throw new ImageWriteException(
107                                    "More than one Interoperability directory.");
108                        }
109                        interoperabilityDirectory = directory;
110                        break;
111                    default:
112                        throw new ImageWriteException("Unknown directory: "
113                                + dirType);
114                    }
115                } else {
116                    if (directoryIndices.contains(dirType)) {
117                        throw new ImageWriteException(
118                                "More than one directory with index: " + dirType
119                                        + ".");
120                    }
121                    directoryIndices.add(dirType);
122                    // dirMap.put(arg0, arg1)
123                }
124    
125                final HashSet<Integer> fieldTags = new HashSet<Integer>();
126                final List<TiffOutputField> fields = directory.getFields();
127                for (int j = 0; j < fields.size(); j++) {
128                    final TiffOutputField field = fields.get(j);
129    
130                    if (fieldTags.contains(field.tag)) {
131                        throw new ImageWriteException("Tag ("
132                                + field.tagInfo.getDescription()
133                                + ") appears twice in directory.");
134                    }
135                    fieldTags.add(field.tag);
136    
137                    if (field.tag == ExifTagConstants.EXIF_TAG_EXIF_OFFSET.tag) {
138                        if (exifDirectoryOffsetField != null) {
139                            throw new ImageWriteException(
140                                    "More than one Exif directory offset field.");
141                        }
142                        exifDirectoryOffsetField = field;
143                    } else if (field.tag == ExifTagConstants.EXIF_TAG_INTEROP_OFFSET.tag) {
144                        if (interoperabilityDirectoryOffsetField != null) {
145                            throw new ImageWriteException(
146                                    "More than one Interoperability directory offset field.");
147                        }
148                        interoperabilityDirectoryOffsetField = field;
149                    } else if (field.tag == ExifTagConstants.EXIF_TAG_GPSINFO.tag) {
150                        if (gpsDirectoryOffsetField != null) {
151                            throw new ImageWriteException(
152                                    "More than one GPS directory offset field.");
153                        }
154                        gpsDirectoryOffsetField = field;
155                    }
156                }
157                // directory.
158            }
159    
160            if (directoryIndices.size() < 1) {
161                throw new ImageWriteException("Missing root directory.");
162            }
163    
164            // "normal" TIFF directories should have continous indices starting with
165            // 0, ie. 0, 1, 2...
166            Collections.sort(directoryIndices);
167    
168            TiffOutputDirectory previousDirectory = null;
169            for (int i = 0; i < directoryIndices.size(); i++) {
170                final Integer index = directoryIndices.get(i);
171                if (index.intValue() != i) {
172                    throw new ImageWriteException("Missing directory: " + i + ".");
173                }
174    
175                // set up chain of directory references for "normal" directories.
176                final TiffOutputDirectory directory = directoryTypeMap.get(index);
177                if (null != previousDirectory) {
178                    previousDirectory.setNextDirectory(directory);
179                }
180                previousDirectory = directory;
181            }
182    
183            final TiffOutputDirectory rootDirectory = directoryTypeMap
184                    .get(DIRECTORY_TYPE_ROOT);
185    
186            // prepare results
187            final TiffOutputSummary result = new TiffOutputSummary(byteOrder,
188                    rootDirectory, directoryTypeMap);
189    
190            if (interoperabilityDirectory == null
191                    && interoperabilityDirectoryOffsetField != null) {
192                // perhaps we should just discard field?
193                throw new ImageWriteException(
194                        "Output set has Interoperability Directory Offset field, but no Interoperability Directory");
195            } else if (interoperabilityDirectory != null) {
196                if (exifDirectory == null) {
197                    exifDirectory = outputSet.addExifDirectory();
198                }
199    
200                if (interoperabilityDirectoryOffsetField == null) {
201                    interoperabilityDirectoryOffsetField =
202                            TiffOutputField.createOffsetField(
203                                    ExifTagConstants.EXIF_TAG_INTEROP_OFFSET,
204                                    byteOrder);
205                    exifDirectory.add(interoperabilityDirectoryOffsetField);
206                }
207    
208                result.add(interoperabilityDirectory,
209                        interoperabilityDirectoryOffsetField);
210            }
211    
212            // make sure offset fields and offset'd directories correspond.
213            if (exifDirectory == null && exifDirectoryOffsetField != null) {
214                // perhaps we should just discard field?
215                throw new ImageWriteException(
216                        "Output set has Exif Directory Offset field, but no Exif Directory");
217            } else if (exifDirectory != null) {
218                if (exifDirectoryOffsetField == null) {
219                    exifDirectoryOffsetField = TiffOutputField.createOffsetField(
220                            ExifTagConstants.EXIF_TAG_EXIF_OFFSET, byteOrder);
221                    rootDirectory.add(exifDirectoryOffsetField);
222                }
223    
224                result.add(exifDirectory, exifDirectoryOffsetField);
225            }
226    
227            if (gpsDirectory == null && gpsDirectoryOffsetField != null) {
228                // perhaps we should just discard field?
229                throw new ImageWriteException(
230                        "Output set has GPS Directory Offset field, but no GPS Directory");
231            } else if (gpsDirectory != null) {
232                if (gpsDirectoryOffsetField == null) {
233                    gpsDirectoryOffsetField = TiffOutputField.createOffsetField(
234                            ExifTagConstants.EXIF_TAG_GPSINFO, byteOrder);
235                    rootDirectory.add(gpsDirectoryOffsetField);
236                }
237    
238                result.add(gpsDirectory, gpsDirectoryOffsetField);
239            }
240    
241            return result;
242    
243            // Debug.debug();
244        }

246        public void writeImage(final BufferedImage src, final OutputStream os, Map<String,Object> params)
247                throws ImageWriteException, IOException {
248            // make copy of params; we'll clear keys as we consume them.
249            params = new HashMap<String,Object>(params);
250    
251            // clear format key.
252            if (params.containsKey(PARAM_KEY_FORMAT)) {
253                params.remove(PARAM_KEY_FORMAT);
254            }
255    
256            TiffOutputSet userExif = null;
257            if (params.containsKey(PARAM_KEY_EXIF)) {
258                userExif = (TiffOutputSet) params.remove(PARAM_KEY_EXIF);
259            }
260    
261            String xmpXml = null;
262            if (params.containsKey(PARAM_KEY_XMP_XML)) {
263                xmpXml = (String) params.get(PARAM_KEY_XMP_XML);
264                params.remove(PARAM_KEY_XMP_XML);
265            }
266    
267            PixelDensity pixelDensity = (PixelDensity) params
268                    .remove(PARAM_KEY_PIXEL_DENSITY);
269            if (pixelDensity == null) {
270                pixelDensity = PixelDensity.createFromPixelsPerInch(72, 72);
271            }
272    
273            final int width = src.getWidth();
274            final int height = src.getHeight();
275    
276            int compression = TIFF_COMPRESSION_LZW; // LZW is default
277            if (params.containsKey(PARAM_KEY_COMPRESSION)) {
278                final Object value = params.get(PARAM_KEY_COMPRESSION);
279                if (value != null) {
280                    if (!(value instanceof Number)) {
281                        throw new ImageWriteException(
282                                "Invalid compression parameter: " + value);
283                    }
284                    compression = ((Number) value).intValue();
285                }
286                params.remove(PARAM_KEY_COMPRESSION);
287            }
288            final HashMap<String,Object> rawParams = new HashMap<String,Object>(params);
289            params.remove(PARAM_KEY_T4_OPTIONS);
290            params.remove(PARAM_KEY_T6_OPTIONS);
291            if (params.size() > 0) {
292                final Object firstKey = params.keySet().iterator().next();
293                throw new ImageWriteException("Unknown parameter: " + firstKey);
294            }
295    
296            int samplesPerPixel;
297            int bitsPerSample;
298            int photometricInterpretation;
299            if (compression == TIFF_COMPRESSION_CCITT_1D
300                    || compression == TIFF_COMPRESSION_CCITT_GROUP_3
301                    || compression == TIFF_COMPRESSION_CCITT_GROUP_4) {
302                samplesPerPixel = 1;
303                bitsPerSample = 1;
304                photometricInterpretation = 0;
305            } else {
306                samplesPerPixel = 3;
307                bitsPerSample = 8;
308                photometricInterpretation = 2;
309            }
310    
311            int rowsPerStrip = 64000 / (width * bitsPerSample * samplesPerPixel); // TODO:
312            rowsPerStrip = Math.max(1, rowsPerStrip); // must have at least one.
313    
314            final byte strips[][] = getStrips(src, samplesPerPixel, bitsPerSample,
315                    rowsPerStrip);
316    
317            // System.out.println("width: " + width);
318            // System.out.println("height: " + height);
319            // System.out.println("fRowsPerStrip: " + fRowsPerStrip);
320            // System.out.println("fSamplesPerPixel: " + fSamplesPerPixel);
321            // System.out.println("stripCount: " + stripCount);
322    
323            int t4Options = 0;
324            int t6Options = 0;
325            if (compression == TIFF_COMPRESSION_CCITT_1D) {
326                for (int i = 0; i < strips.length; i++) {
327                    strips[i] = T4AndT6Compression.compressModifiedHuffman(
328                            strips[i], width, strips[i].length / ((width + 7) / 8));
329                }
330            } else if (compression == TIFF_COMPRESSION_CCITT_GROUP_3) {
331                final Integer t4Parameter = (Integer) rawParams.get(PARAM_KEY_T4_OPTIONS);
332                if (t4Parameter != null) {
333                    t4Options = t4Parameter.intValue();
334                }
335                t4Options &= 0x7;
336                final boolean is2D = (t4Options & 1) != 0;
337                final boolean usesUncompressedMode = (t4Options & 2) != 0;
338                if (usesUncompressedMode) {
339                    throw new ImageWriteException(
340                            "T.4 compression with the uncompressed mode extension is not yet supported");
341                }
342                final boolean hasFillBitsBeforeEOL = (t4Options & 4) != 0;
343                for (int i = 0; i < strips.length; i++) {
344                    if (is2D) {
345                        strips[i] = T4AndT6Compression.compressT4_2D(strips[i],
346                                width, strips[i].length / ((width + 7) / 8),
347                                hasFillBitsBeforeEOL, rowsPerStrip);
348                    } else {
349                        strips[i] = T4AndT6Compression.compressT4_1D(strips[i],
350                                width, strips[i].length / ((width + 7) / 8),
351                                hasFillBitsBeforeEOL);
352                    }
353                }
354            } else if (compression == TIFF_COMPRESSION_CCITT_GROUP_4) {
355                final Integer t6Parameter = (Integer) rawParams.get(PARAM_KEY_T6_OPTIONS);
356                if (t6Parameter != null) {
357                    t6Options = t6Parameter.intValue();
358                }
359                t6Options &= 0x4;
360                final boolean usesUncompressedMode = (t6Options & TIFF_FLAG_T6_OPTIONS_UNCOMPRESSED_MODE) != 0;
361                if (usesUncompressedMode) {
362                    throw new ImageWriteException(
363                            "T.6 compression with the uncompressed mode extension is not yet supported");
364                }
365                for (int i = 0; i < strips.length; i++) {
366                    strips[i] = T4AndT6Compression.compressT6(strips[i], width,
367                            strips[i].length / ((width + 7) / 8));
368                }
369            } else if (compression == TIFF_COMPRESSION_PACKBITS) {
370                for (int i = 0; i < strips.length; i++) {
371                    strips[i] = new PackBits().compress(strips[i]);
372                }
373            } else if (compression == TIFF_COMPRESSION_LZW) {
374                for (int i = 0; i < strips.length; i++) {
375                    final byte uncompressed[] = strips[i];
376    
377                    final int LZW_MINIMUM_CODE_SIZE = 8;
378    
379                    final MyLzwCompressor compressor = new MyLzwCompressor(
380                            LZW_MINIMUM_CODE_SIZE, ByteOrder.MOTOROLA, true);
381                    final byte compressed[] = compressor.compress(uncompressed);
382    
383                    strips[i] = compressed;
384                }
385            } else if (compression == TIFF_COMPRESSION_UNCOMPRESSED) {
386                // do nothing.
387            } else {
388                throw new ImageWriteException(
389                        "Invalid compression parameter (Only CCITT 1D/Group 3/Group 4, LZW, Packbits and uncompressed supported).");
390            }
391    
392            final TiffElement.DataElement imageData[] = new TiffElement.DataElement[strips.length];
393            for (int i = 0; i < strips.length; i++) {
394                imageData[i] = new TiffImageData.Data(0, strips[i].length,
395                        strips[i]);
396            }
397    
398            final TiffOutputSet outputSet = new TiffOutputSet(byteOrder);
399            final TiffOutputDirectory directory = outputSet.addRootDirectory();
400    
401            // WriteField stripOffsetsField;
402    
403            {
404    
405                directory.add(TiffTagConstants.TIFF_TAG_IMAGE_WIDTH, width);
406                directory.add(TiffTagConstants.TIFF_TAG_IMAGE_LENGTH, height);
407                directory.add(TiffTagConstants.TIFF_TAG_PHOTOMETRIC_INTERPRETATION,
408                        (short) photometricInterpretation);
409                directory.add(TiffTagConstants.TIFF_TAG_COMPRESSION,
410                        (short) compression);
411                directory.add(TiffTagConstants.TIFF_TAG_SAMPLES_PER_PIXEL,
412                        (short) samplesPerPixel);
413    
414                if (samplesPerPixel == 3) {
415                    directory.add(TiffTagConstants.TIFF_TAG_BITS_PER_SAMPLE,
416                            (short) bitsPerSample, (short) bitsPerSample,
417                            (short) bitsPerSample);
418                } else if (samplesPerPixel == 1) {
419                    directory.add(TiffTagConstants.TIFF_TAG_BITS_PER_SAMPLE,
420                            (short) bitsPerSample);
421                }
422                // {
423                // stripOffsetsField = new WriteField(TIFF_TAG_STRIP_OFFSETS,
424                // FIELD_TYPE_LONG, stripOffsets.length, FIELD_TYPE_LONG
425                // .writeData(stripOffsets, byteOrder));
426                // directory.add(stripOffsetsField);
427                // }
428                // {
429                // WriteField field = new WriteField(TIFF_TAG_STRIP_BYTE_COUNTS,
430                // FIELD_TYPE_LONG, stripByteCounts.length,
431                // FIELD_TYPE_LONG.writeData(stripByteCounts,
432                // WRITE_BYTE_ORDER));
433                // directory.add(field);
434                // }
435                directory.add(TiffTagConstants.TIFF_TAG_ROWS_PER_STRIP,
436                        rowsPerStrip);
437                if (pixelDensity.isUnitless()) {
438                    directory.add(TiffTagConstants.TIFF_TAG_RESOLUTION_UNIT,
439                            (short) 0);
440                    directory.add(TiffTagConstants.TIFF_TAG_XRESOLUTION,
441                            RationalNumberUtilities.getRationalNumber(pixelDensity
442                                    .getRawHorizontalDensity()));
443                    directory.add(TiffTagConstants.TIFF_TAG_YRESOLUTION,
444                            RationalNumberUtilities.getRationalNumber(pixelDensity
445                                    .getRawVerticalDensity()));
446                } else if (pixelDensity.isInInches()) {
447                    directory.add(TiffTagConstants.TIFF_TAG_RESOLUTION_UNIT,
448                            (short) 2);
449                    directory.add(TiffTagConstants.TIFF_TAG_XRESOLUTION,
450                            RationalNumberUtilities.getRationalNumber(pixelDensity
451                                    .horizontalDensityInches()));
452                    directory.add(TiffTagConstants.TIFF_TAG_YRESOLUTION,
453                            RationalNumberUtilities.getRationalNumber(pixelDensity
454                                    .verticalDensityInches()));
455                } else {
456                    directory.add(TiffTagConstants.TIFF_TAG_RESOLUTION_UNIT,
457                            (short) 1);
458                    directory.add(TiffTagConstants.TIFF_TAG_XRESOLUTION,
459                            RationalNumberUtilities.getRationalNumber(pixelDensity
460                                    .horizontalDensityCentimetres()));
461                    directory.add(TiffTagConstants.TIFF_TAG_YRESOLUTION,
462                            RationalNumberUtilities.getRationalNumber(pixelDensity
463                                    .verticalDensityCentimetres()));
464                }
465                if (t4Options != 0) {
466                    directory.add(TiffTagConstants.TIFF_TAG_T4_OPTIONS, t4Options);
467                }
468                if (t6Options != 0) {
469                    directory.add(TiffTagConstants.TIFF_TAG_T6_OPTIONS, t6Options);
470                }
471    
472                if (null != xmpXml) {
473                    final byte xmpXmlBytes[] = xmpXml.getBytes("utf-8");
474                    directory.add(TiffTagConstants.TIFF_TAG_XMP, xmpXmlBytes);
475                }
476    
477            }
478    
479            final TiffImageData tiffImageData = new TiffImageData.Strips(imageData,
480                    rowsPerStrip);
481            directory.setTiffImageData(tiffImageData);
482    
483            if (userExif != null) {
484                combineUserExifIntoFinalExif(userExif, outputSet);
485            }
486    
487            write(os, outputSet);
488        }

490        private void combineUserExifIntoFinalExif(final TiffOutputSet userExif,
491                final TiffOutputSet outputSet) throws ImageWriteException {
492            final List<TiffOutputDirectory> outputDirectories = outputSet
493                    .getDirectories();
494            Collections.sort(outputDirectories, TiffOutputDirectory.COMPARATOR);
495            for (final TiffOutputDirectory userDirectory : userExif.getDirectories()) {
496                final int location = Collections.binarySearch(outputDirectories,
497                        userDirectory, TiffOutputDirectory.COMPARATOR);
498                if (location < 0) {
499                    outputSet.addDirectory(userDirectory);
500                } else {
501                    final TiffOutputDirectory outputDirectory = outputDirectories
502                            .get(location);
503                    for (final TiffOutputField userField : userDirectory.getFields()) {
504                        if (outputDirectory.findField(userField.tagInfo) == null) {
505                            outputDirectory.add(userField);
506                        }
507                    }
508                }
509            }
510        }

512        private byte[][] getStrips(final BufferedImage src, final int samplesPerPixel,
513                final int bitsPerSample, final int rowsPerStrip) {
514            final int width = src.getWidth();
515            final int height = src.getHeight();
516    
517            final int stripCount = (height + rowsPerStrip - 1) / rowsPerStrip;
518    
519            byte result[][] = null;
520            { // Write Strips
521                result = new byte[stripCount][];
522    
523                int remaining_rows = height;
524    
525                for (int i = 0; i < stripCount; i++) {
526                    final int rowsInStrip = Math.min(rowsPerStrip, remaining_rows);
527                    remaining_rows -= rowsInStrip;
528    
529                    final int bitsInRow = bitsPerSample * samplesPerPixel * width;
530                    final int bytesPerRow = (bitsInRow + 7) / 8;
531                    final int bytesInStrip = rowsInStrip * bytesPerRow;
532    
533                    final byte uncompressed[] = new byte[bytesInStrip];
534    
535                    int counter = 0;
536                    int y = i * rowsPerStrip;
537                    final int stop = i * rowsPerStrip + rowsPerStrip;
538    
539                    for (; (y < height) && (y < stop); y++) {
540                        int bitCache = 0;
541                        int bitsInCache = 0;
542                        for (int x = 0; x < width; x++) {
543                            final int rgb = src.getRGB(x, y);
544                            final int red = 0xff & (rgb >> 16);
545                            final int green = 0xff & (rgb >> 8);
546                            final int blue = 0xff & (rgb >> 0);
547    
548                            if (bitsPerSample == 1) {
549                                int sample = (red + green + blue) / 3;
550                                if (sample > 127) {
551                                    sample = 0;
552                                } else {
553                                    sample = 1;
554                                }
555                                bitCache <<= 1;
556                                bitCache |= sample;
557                                bitsInCache++;
558                                if (bitsInCache == 8) {
559                                    uncompressed[counter++] = (byte) bitCache;
560                                    bitCache = 0;
561                                    bitsInCache = 0;
562                                }
563                            } else {
564                                uncompressed[counter++] = (byte) red;
565                                uncompressed[counter++] = (byte) green;
566                                uncompressed[counter++] = (byte) blue;
567                            }
568                        }
569                        if (bitsInCache > 0) {
570                            bitCache <<= (8 - bitsInCache);
571                            uncompressed[counter++] = (byte) bitCache;
572                        }
573                    }
574    
575                    result[i] = uncompressed;
576                }
577    
578            }
579    
580            return result;
581        }

583        protected void writeImageFileHeader(final BinaryOutputStream bos)
584                throws IOException {
585            final int offsetToFirstIFD = TIFF_HEADER_SIZE;
586    
587            writeImageFileHeader(bos, offsetToFirstIFD);
588        }
589    
590        protected void writeImageFileHeader(final BinaryOutputStream bos,
591                final int offsetToFirstIFD) throws IOException {
592            if (byteOrder == ByteOrder.INTEL) {
593                bos.write('I');
594                bos.write('I');
595            } else {
596                bos.write('M');
597                bos.write('M');
598            }
599    
600            bos.write2Bytes(42); // tiffVersion
601    
602            bos.write4Bytes(offsetToFirstIFD);
603        }
