066                final byte[] result, final int offset) {
067            if (byteOrder == ByteOrder.BIG_ENDIAN) {
068                result[offset + 0] = (byte) (value >> 8);
069                result[offset + 1] = (byte) (value >> 0);
070            } else {
071                result[offset + 1] = (byte) (value >> 8);
072                result[offset + 0] = (byte) (value >> 0);
073            }
074        }

173        public static void toBytes(final float value, final ByteOrder byteOrder,
174                final byte[] result, final int offset) {
175            final int bits = Float.floatToRawIntBits(value);
176            if (byteOrder == ByteOrder.LITTLE_ENDIAN) {
177                result[offset + 0] = (byte) (0xff & (bits >> 0));
178                result[offset + 1] = (byte) (0xff & (bits >> 8));
179                result[offset + 2] = (byte) (0xff & (bits >> 16));
180                result[offset + 3] = (byte) (0xff & (bits >> 24));
181            } else {
182                result[offset + 3] = (byte) (0xff & (bits >> 0));
183                result[offset + 2] = (byte) (0xff & (bits >> 8));
184                result[offset + 1] = (byte) (0xff & (bits >> 16));
185                result[offset + 0] = (byte) (0xff & (bits >> 24));
186            }
187        }

230        public static void toBytes(final double value, final ByteOrder byteOrder,
231                final byte[] result, final int offset) {
232            final long bits = Double.doubleToRawLongBits(value);
233            if (byteOrder == ByteOrder.INTEL) {
234                result[offset + 0] = (byte) (0xff & (bits >> 0));
235                result[offset + 1] = (byte) (0xff & (bits >> 8));
236                result[offset + 2] = (byte) (0xff & (bits >> 16));
237                result[offset + 3] = (byte) (0xff & (bits >> 24));
238                result[offset + 4] = (byte) (0xff & (bits >> 32));
239                result[offset + 5] = (byte) (0xff & (bits >> 40));
240                result[offset + 6] = (byte) (0xff & (bits >> 48));
241                result[offset + 7] = (byte) (0xff & (bits >> 56));
242            } else {
243                result[offset + 7] = (byte) (0xff & (bits >> 0));
244                result[offset + 6] = (byte) (0xff & (bits >> 8));
245                result[offset + 5] = (byte) (0xff & (bits >> 16));
246                result[offset + 4] = (byte) (0xff & (bits >> 24));
247                result[offset + 3] = (byte) (0xff & (bits >> 32));
248                result[offset + 2] = (byte) (0xff & (bits >> 40));
249                result[offset + 1] = (byte) (0xff & (bits >> 48));
250                result[offset + 0] = (byte) (0xff & (bits >> 56));
251            }
252        }

295        public static void toBytes(final RationalNumber value, final ByteOrder byteOrder,
296                final byte[] result, final int offset) {
297            if (byteOrder == ByteOrder.MOTOROLA) {
298                result[offset + 0] = (byte) (value.numerator >> 24);
299                result[offset + 1] = (byte) (value.numerator >> 16);
300                result[offset + 2] = (byte) (value.numerator >> 8);
301                result[offset + 3] = (byte) (value.numerator >> 0);
302                result[offset + 4] = (byte) (value.divisor >> 24);
303                result[offset + 5] = (byte) (value.divisor >> 16);
304                result[offset + 6] = (byte) (value.divisor >> 8);
305                result[offset + 7] = (byte) (value.divisor >> 0);
306            } else {
307                result[offset + 3] = (byte) (value.numerator >> 24);
308                result[offset + 2] = (byte) (value.numerator >> 16);
309                result[offset + 1] = (byte) (value.numerator >> 8);
310                result[offset + 0] = (byte) (value.numerator >> 0);
311                result[offset + 7] = (byte) (value.divisor >> 24);
312                result[offset + 6] = (byte) (value.divisor >> 16);
313                result[offset + 5] = (byte) (value.divisor >> 8);
314                result[offset + 4] = (byte) (value.divisor >> 0);
315            }
316        }

322        public static short toShort(final byte[] bytes, final int offset, final ByteOrder byteOrder) {
323            final int byte0 = 0xff & bytes[offset + 0];
324            final int byte1 = 0xff & bytes[offset + 1];
325            if (byteOrder == ByteOrder.MOTOROLA) {
326                return (short) ((byte0 << 8) | byte1);
327            } else {
328                return (short) ((byte1 << 8) | byte0);
329            }
330        }

388        public static float toFloat(final byte[] bytes, final int offset, final ByteOrder byteOrder) {
389            final int byte0 = 0xff & bytes[offset + 0];
390            final int byte1 = 0xff & bytes[offset + 1];
391            final int byte2 = 0xff & bytes[offset + 2];
392            final int byte3 = 0xff & bytes[offset + 3];
393            int bits;
394            if (byteOrder == ByteOrder.MOTOROLA) {
395                bits = (byte0 << 24) | (byte1 << 16) | (byte2 << 8) | (byte3 << 0);
396            } else {
397                bits = (byte3 << 24) | (byte2 << 16) | (byte1 << 8) | (byte0 << 0);
398            }
399            return Float.intBitsToFloat(bits);
400        }

424        public static double toDouble(final byte[] bytes, final int offset, final ByteOrder byteOrder) {
425            final long byte0 = 0xffL & bytes[offset + 0];
426            final long byte1 = 0xffL & bytes[offset + 1];
427            final long byte2 = 0xffL & bytes[offset + 2];
428            final long byte3 = 0xffL & bytes[offset + 3];
429            final long byte4 = 0xffL & bytes[offset + 4];
430            final long byte5 = 0xffL & bytes[offset + 5];
431            final long byte6 = 0xffL & bytes[offset + 6];
432            final long byte7 = 0xffL & bytes[offset + 7];
433            long bits;
434            if (byteOrder == ByteOrder.MOTOROLA) {
435                bits = (byte0 << 56) | (byte1 << 48) | (byte2 << 40)
436                        | (byte3 << 32) | (byte4 << 24) | (byte5 << 16)
437                        | (byte6 << 8) | (byte7 << 0);
438            } else {
439                bits = (byte7 << 56) | (byte6 << 48) | (byte5 << 40)
440                        | (byte4 << 32) | (byte3 << 24) | (byte2 << 16)
441                        | (byte1 << 8) | (byte0 << 0);
442            }
443            return Double.longBitsToDouble(bits);
444        }

468        public static RationalNumber toRational(final byte[] bytes, final int offset,
469                final ByteOrder byteOrder) {
470            final int byte0 = 0xff & bytes[offset + 0];
471            final int byte1 = 0xff & bytes[offset + 1];
472            final int byte2 = 0xff & bytes[offset + 2];
473            final int byte3 = 0xff & bytes[offset + 3];
474            final int byte4 = 0xff & bytes[offset + 4];
475            final int byte5 = 0xff & bytes[offset + 5];
476            final int byte6 = 0xff & bytes[offset + 6];
477            final int byte7 = 0xff & bytes[offset + 7];
478            int numerator;
479            int divisor;
480            if (byteOrder == ByteOrder.MOTOROLA) {
481                numerator = (byte0 << 24) | (byte1 << 16) | (byte2 << 8) | byte3;
482                divisor = (byte4 << 24) | (byte5 << 16) | (byte6 << 8) | byte7;
483            } else {
484                numerator = (byte3 << 24) | (byte2 << 16) | (byte1 << 8) | byte0;
485                divisor = (byte7 << 24) | (byte6 << 16) | (byte5 << 8) | byte4;
486            }
487            return new RationalNumber(numerator, divisor);
488        }
