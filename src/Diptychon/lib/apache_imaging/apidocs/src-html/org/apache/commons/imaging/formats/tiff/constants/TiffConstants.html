017    package org.apache.commons.imaging.formats.tiff.constants;
018    
019    import org.apache.commons.imaging.ImagingConstants;
020    import org.apache.commons.imaging.common.ByteOrder;
021    
022    public interface TiffConstants
023            extends
024                ImagingConstants,
025                TiffFieldTypeConstants,
026                TiffDirectoryConstants,
027                AllTagConstants {
028        public static final ByteOrder DEFAULT_TIFF_BYTE_ORDER = ByteOrder.INTEL;
029    
030        public static final int TIFF_HEADER_SIZE = 8;
031        public static final int TIFF_DIRECTORY_HEADER_LENGTH = 2;
032        public static final int TIFF_DIRECTORY_FOOTER_LENGTH = 4;
033        public static final int TIFF_ENTRY_LENGTH = 12;
034        public static final int TIFF_ENTRY_MAX_VALUE_LENGTH = 4;
035    
036        public static final int TIFF_COMPRESSION_UNCOMPRESSED_1 = 1;
037        public static final int TIFF_COMPRESSION_UNCOMPRESSED = TIFF_COMPRESSION_UNCOMPRESSED_1;
038        public static final int TIFF_COMPRESSION_CCITT_1D = 2;
039        public static final int TIFF_COMPRESSION_CCITT_GROUP_3 = 3;
040        public static final int TIFF_COMPRESSION_CCITT_GROUP_4 = 4;
041        public static final int TIFF_COMPRESSION_LZW = 5;
042        public static final int TIFF_COMPRESSION_JPEG = 6;
043        public static final int TIFF_COMPRESSION_UNCOMPRESSED_2 = 32771;
044        public static final int TIFF_COMPRESSION_PACKBITS = 32773;
045    
046        /**
047         * Parameter key. Used in write operations to indicate the desired
048         * T.4 options to use when using TIFF_COMPRESSION_CCITT_GROUP_3.
049         * <p>
050         * Valid values: any Integer containing a mixture of the
051         * TIFF_FLAG_T4_OPTIONS_2D, TIFF_FLAG_T4_OPTIONS_UNCOMPRESSED_MODE,
052         * and TIFF_FLAG_T4_OPTIONS_FILL flags.
053         */
054        public static final String PARAM_KEY_T4_OPTIONS = "T4_OPTIONS";
055    
056        /**
057         * Parameter key. Used in write operations to indicate the desired
058         * T.6 options to use when using TIFF_COMPRESSION_CCITT_GROUP_4.
059         * <p>
060         * Valid values: any Integer containing either zero or
061         * TIFF_FLAG_T6_OPTIONS_UNCOMPRESSED_MODE.
062         */
063        public static final String PARAM_KEY_T6_OPTIONS = "T6_OPTIONS";
064    
065        public static final int TIFF_FLAG_T4_OPTIONS_2D = 1;
066        public static final int TIFF_FLAG_T4_OPTIONS_UNCOMPRESSED_MODE = 2;
067        public static final int TIFF_FLAG_T4_OPTIONS_FILL = 4;
068        public static final int TIFF_FLAG_T6_OPTIONS_UNCOMPRESSED_MODE = 2;
069    }
