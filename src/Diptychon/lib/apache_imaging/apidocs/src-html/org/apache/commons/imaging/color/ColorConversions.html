029        public static final ColorCieLab convertXYZtoCIELab(final double X, final double Y,
030                final double Z) {
031    
032            double var_X = X / ref_X; // ref_X = 95.047 Observer= 2?, Illuminant=
033                                      // D65
034            double var_Y = Y / ref_Y; // ref_Y = 100.000
035            double var_Z = Z / ref_Z; // ref_Z = 108.883
036    
037            if (var_X > 0.008856) {
038                var_X = Math.pow(var_X, (1 / 3.0));
039            } else {
040                var_X = (7.787 * var_X) + (16 / 116.0);
041            }
042            if (var_Y > 0.008856) {
043                var_Y = Math.pow(var_Y, 1 / 3.0);
044            } else {
045                var_Y = (7.787 * var_Y) + (16 / 116.0);
046            }
047            if (var_Z > 0.008856) {
048                var_Z = Math.pow(var_Z, 1 / 3.0);
049            } else {
050                var_Z = (7.787 * var_Z) + (16 / 116.0);
051            }
052    
053            final double L = (116 * var_Y) - 16;
054            final double a = 500 * (var_X - var_Y);
055            final double b = 200 * (var_Y - var_Z);
056            return new ColorCieLab(L, a, b);
057        }

063        public static final ColorXyz convertCIELabtoXYZ(final double L, final double a, final double b) {
064            double var_Y = (L + 16) / 116.0;
065            double var_X = a / 500 + var_Y;
066            double var_Z = var_Y - b / 200.0;
067    
068            if (Math.pow(var_Y, 3) > 0.008856) {
069                var_Y = Math.pow(var_Y, 3);
070            } else {
071                var_Y = (var_Y - 16 / 116.0) / 7.787;
072            }
073            if (Math.pow(var_X, 3) > 0.008856) {
074                var_X = Math.pow(var_X, 3);
075            } else {
076                var_X = (var_X - 16 / 116.0) / 7.787;
077            }
078            if (Math.pow(var_Z, 3) > 0.008856) {
079                var_Z = Math.pow(var_Z, 3);
080            } else {
081                var_Z = (var_Z - 16 / 116.0) / 7.787;
082            }
083    
084            final double X = ref_X * var_X; // ref_X = 95.047 Observer= 2?, Illuminant=
085                                      // D65
086            final double Y = ref_Y * var_Y; // ref_Y = 100.000
087            final double Z = ref_Z * var_Z; // ref_Z = 108.883
088    
089            return new ColorXyz(X, Y, Z);
090        }

126        public static final int convertXYZtoRGB(final double X, final double Y, final double Z) {
127            // Observer = 2?, Illuminant = D65
128            final double var_X = X / 100.0; // Where X = 0 ? 95.047
129            final double var_Y = Y / 100.0; // Where Y = 0 ? 100.000
130            final double var_Z = Z / 100.0; // Where Z = 0 ? 108.883
131    
132            double var_R = var_X * 3.2406 + var_Y * -1.5372 + var_Z * -0.4986;
133            double var_G = var_X * -0.9689 + var_Y * 1.8758 + var_Z * 0.0415;
134            double var_B = var_X * 0.0557 + var_Y * -0.2040 + var_Z * 1.0570;
135    
136            if (var_R > 0.0031308) {
137                var_R = 1.055 * Math.pow(var_R, (1 / 2.4)) - 0.055;
138            } else {
139                var_R = 12.92 * var_R;
140            }
141            if (var_G > 0.0031308) {
142                var_G = 1.055 * Math.pow(var_G, (1 / 2.4)) - 0.055;
143            } else {
144                var_G = 12.92 * var_G;
145            }
146            if (var_B > 0.0031308) {
147                var_B = 1.055 * Math.pow(var_B, (1 / 2.4)) - 0.055;
148            } else {
149                var_B = 12.92 * var_B;
150            }
151    
152            final double R = (var_R * 255);
153            final double G = (var_G * 255);
154            final double B = (var_B * 255);
155    
156            return convertRGBtoRGB(R, G, B);
157        }

196        public static final ColorCmy convertRGBtoCMY(final int rgb) {
197            final int R = 0xff & (rgb >> 16);
198            final int G = 0xff & (rgb >> 8);
199            final int B = 0xff & (rgb >> 0);
200    
201            // RGB values = 0 ? 255
202            // CMY values = 0 ? 1
203    
204            final double C = 1 - (R / 255.0);
205            final double M = 1 - (G / 255.0);
206            final double Y = 1 - (B / 255.0);
207    
208            return new ColorCmy(C, M, Y);
209        }

211        public static final int convertCMYtoRGB(final ColorCmy cmy) {
212            // From Ghostscript's gdevcdj.c:
213            // * Ghostscript: R = (1.0 - C) * (1.0 - K)
214            // * Adobe: R = 1.0 - min(1.0, C + K)
215            // and similarly for G and B.
216            // This is Ghostscript's formula with K = 0.
217    
218            // CMY values = 0 ? 1
219            // RGB values = 0 ? 255
220    
221            final double R = (1 - cmy.C) * 255.0;
222            final double G = (1 - cmy.M) * 255.0;
223            final double B = (1 - cmy.Y) * 255.0;
224    
225            return convertRGBtoRGB(R, G, B);
226        }

228        public static final ColorCmyk convertCMYtoCMYK(final ColorCmy cmy) {
229            // Where CMYK and CMY values = 0 ? 1
230    
231            double C = cmy.C;
232            double M = cmy.M;
233            double Y = cmy.Y;
234    
235            double var_K = 1.0;
236    
237            if (C < var_K) {
238                var_K = C;
239            }
240            if (M < var_K) {
241                var_K = M;
242            }
243            if (Y < var_K) {
244                var_K = Y;
245            }
246            if (var_K == 1) { // Black
247                C = 0;
248                M = 0;
249                Y = 0;
250            } else {
251                C = (C - var_K) / (1 - var_K);
252                M = (M - var_K) / (1 - var_K);
253                Y = (Y - var_K) / (1 - var_K);
254            }
255            return new ColorCmyk(C, M, Y, var_K);
256        }

284        public static final ColorHsl convertRGBtoHSL(final int rgb) {
285    
286            final int R = 0xff & (rgb >> 16);
287            final int G = 0xff & (rgb >> 8);
288            final int B = 0xff & (rgb >> 0);
289    
290            final double var_R = (R / 255.0); // Where RGB values = 0 ? 255
291            final double var_G = (G / 255.0);
292            final double var_B = (B / 255.0);
293    
294            final double var_Min = Math.min(var_R, Math.min(var_G, var_B)); // Min. value
295                                                                      // of RGB
296            double var_Max;
297            boolean maxIsR = false;
298            boolean maxIsG = false;
299            if (var_R >= var_G && var_R >= var_B) {
300                var_Max = var_R;
301                maxIsR = true;
302            } else if (var_G > var_B) {
303                var_Max = var_G;
304                maxIsG = true;
305            } else {
306                var_Max = var_B;
307            }
308            final double del_Max = var_Max - var_Min; // Delta RGB value
309    
310            final double L = (var_Max + var_Min) / 2.0;
311    
312            double H, S;
313            // Debug.debug("del_Max", del_Max);
314            if (del_Max == 0) {
315                // This is a gray, no chroma...
316            
317                H = 0; // HSL results = 0 ? 1
318                S = 0;
319            } else {
320            // Chromatic data...
321    
322                // Debug.debug("L", L);
323    
324                if (L < 0.5) {
325                    S = del_Max / (var_Max + var_Min);
326                } else {
327                    S = del_Max / (2 - var_Max - var_Min);
328                }
329    
330                // Debug.debug("S", S);
331    
332                final double del_R = (((var_Max - var_R) / 6) + (del_Max / 2)) / del_Max;
333                final double del_G = (((var_Max - var_G) / 6) + (del_Max / 2)) / del_Max;
334                final double del_B = (((var_Max - var_B) / 6) + (del_Max / 2)) / del_Max;
335    
336                if (maxIsR) {
337                    H = del_B - del_G;
338                } else if (maxIsG) {
339                    H = (1 / 3.0) + del_R - del_B;
340                } else {
341                    H = (2 / 3.0) + del_G - del_R;
342                }
343    
344                // Debug.debug("H1", H);
345    
346                if (H < 0) {
347                    H += 1;
348                }
349                if (H > 1) {
350                    H -= 1;
351                }
352    
353                // Debug.debug("H2", H);
354            }
355    
356            return new ColorHsl(H, S, L);
357        }

363        public static int convertHSLtoRGB(final double H, final double S, final double L) {
364            double R, G, B;
365    
366            if (S == 0) {
367                // HSL values = 0 ? 1
368                R = L * 255; // RGB results = 0 ? 255
369                G = L * 255;
370                B = L * 255;
371            } else {
372                double var_2;
373    
374                if (L < 0.5) {
375                    var_2 = L * (1 + S);
376                } else {
377                    var_2 = (L + S) - (S * L);
378                }
379    
380                final double var_1 = 2 * L - var_2;
381    
382                R = 255 * convertHuetoRGB(var_1, var_2, H + (1 / 3.0));
383                G = 255 * convertHuetoRGB(var_1, var_2, H);
384                B = 255 * convertHuetoRGB(var_1, var_2, H - (1 / 3.0));
385            }
386    
387            return convertRGBtoRGB(R, G, B);
388        }

390        private static double convertHuetoRGB(final double v1, final double v2, double vH) {
391            if (vH < 0) {
392                vH += 1;
393            }
394            if (vH > 1) {
395                vH -= 1;
396            }
397            if ((6 * vH) < 1) {
398                return (v1 + (v2 - v1) * 6 * vH);
399            }
400            if ((2 * vH) < 1) {
401                return (v2);
402            }
403            if ((3 * vH) < 2) {
404                return (v1 + (v2 - v1) * ((2 / 3.0) - vH) * 6);
405            }
406            return (v1);
407        }

409        public static final ColorHsv convertRGBtoHSV(final int rgb) {
410            final int R = 0xff & (rgb >> 16);
411            final int G = 0xff & (rgb >> 8);
412            final int B = 0xff & (rgb >> 0);
413    
414            final double var_R = (R / 255.0); // RGB values = 0 ? 255
415            final double var_G = (G / 255.0);
416            final double var_B = (B / 255.0);
417    
418            final double var_Min = Math.min(var_R, Math.min(var_G, var_B)); // Min. value
419                                                                      // of RGB
420            boolean maxIsR = false;
421            boolean maxIsG = false;
422            double var_Max;
423            if (var_R >= var_G && var_R >= var_B) {
424                var_Max = var_R;
425                maxIsR = true;
426            } else if (var_G > var_B) {
427                var_Max = var_G;
428                maxIsG = true;
429            } else {
430                var_Max = var_B;
431            }
432            final double del_Max = var_Max - var_Min; // Delta RGB value
433    
434            final double V = var_Max;
435    
436            double H, S;
437            if (del_Max == 0) {
438                // This is a gray, no chroma...
439                H = 0; // HSV results = 0 ? 1
440                S = 0;
441            } else {
442            // Chromatic data...
443                S = del_Max / var_Max;
444    
445                final double del_R = (((var_Max - var_R) / 6) + (del_Max / 2)) / del_Max;
446                final double del_G = (((var_Max - var_G) / 6) + (del_Max / 2)) / del_Max;
447                final double del_B = (((var_Max - var_B) / 6) + (del_Max / 2)) / del_Max;
448    
449                if (maxIsR) {
450                    H = del_B - del_G;
451                } else if (maxIsG) {
452                    H = (1 / 3.0) + del_R - del_B;
453                } else {
454                    H = (2 / 3.0) + del_G - del_R;
455                }
456    
457                if (H < 0) {
458                    H += 1;
459                }
460                if (H > 1) {
461                    H -= 1;
462                }
463            }
464    
465            return new ColorHsv(H, S, V);
466        }

472        public static int convertHSVtoRGB(final double H, final double S, final double V) {
473            double R, G, B;
474    
475            if (S == 0) {
476                // HSV values = 0 ? 1
477                R = V * 255;
478                G = V * 255;
479                B = V * 255;
480            } else {
481                double var_h = H * 6;
482                if (var_h == 6) {
483                    var_h = 0; // H must be < 1
484                }
485                final double var_i = Math.floor(var_h); // Or ... var_i = floor( var_h )
486                final double var_1 = V * (1 - S);
487                final double var_2 = V * (1 - S * (var_h - var_i));
488                final double var_3 = V * (1 - S * (1 - (var_h - var_i)));
489    
490                double var_r, var_g, var_b;
491    
492                if (var_i == 0) {
493                    var_r = V;
494                    var_g = var_3;
495                    var_b = var_1;
496                } else if (var_i == 1) {
497                    var_r = var_2;
498                    var_g = V;
499                    var_b = var_1;
500                } else if (var_i == 2) {
501                    var_r = var_1;
502                    var_g = V;
503                    var_b = var_3;
504                } else if (var_i == 3) {
505                    var_r = var_1;
506                    var_g = var_2;
507                    var_b = V;
508                } else if (var_i == 4) {
509                    var_r = var_3;
510                    var_g = var_1;
511                    var_b = V;
512                } else {
513                    var_r = V;
514                    var_g = var_1;
515                    var_b = var_2;
516                }
517    
518                R = var_r * 255; // RGB results = 0 ? 255
519                G = var_g * 255;
520                B = var_b * 255;
521            }
522    
523            return convertRGBtoRGB(R, G, B);
524        }

736            if (Math.pow(var_Y, 3) > 0.008856) {
737                var_Y = Math.pow(var_Y, 3);
738            } else {
739                var_Y = (var_Y - 16 / 116) / 7.787;
740            }
741    
742            final double ref_X = 95.047; // Observer= 2?, Illuminant= D65
743            final double ref_Y = 100.000;
744            final double ref_Z = 108.883;
745    
746            final double ref_U = (4 * ref_X) / (ref_X + (15 * ref_Y) + (3 * ref_Z));
