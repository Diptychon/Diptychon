098        public Dimension getImageSize(final ByteSource byteSource, final Map<String,Object> params)
099                throws ImageReadException, IOException {
100            final FormatCompliance formatCompliance = FormatCompliance.getDefault();
101            final TiffContents contents = new TiffReader(isStrict(params))
102                    .readFirstDirectory(byteSource, params, false, formatCompliance);
103            final TiffDirectory directory = contents.directories.get(0);
104    
105            final TiffField widthField = directory.findField(
106                    TiffTagConstants.TIFF_TAG_IMAGE_WIDTH, true);
107            final TiffField heightField = directory.findField(
108                    TiffTagConstants.TIFF_TAG_IMAGE_LENGTH, true);
109    
110            if ((widthField == null) || (heightField == null)) {
111                throw new ImageReadException("TIFF image missing size info.");
112            }
113    
114            final int height = heightField.getIntValue();
115            final int width = widthField.getIntValue();
116    
117            return new Dimension(width, height);
118        }

129        @Override
130        public IImageMetadata getMetadata(final ByteSource byteSource, final Map<String,Object> params)
131                throws ImageReadException, IOException {
132            final FormatCompliance formatCompliance = FormatCompliance.getDefault();
133            final TiffReader tiffReader = new TiffReader(isStrict(params));
134            final TiffContents contents = tiffReader.readContents(byteSource, params,
135                    formatCompliance);
136    
137            final List<TiffDirectory> directories = contents.directories;
138    
139            final TiffImageMetadata result = new TiffImageMetadata(contents);
140    
141            for (int i = 0; i < directories.size(); i++) {
142                final TiffDirectory dir = directories.get(i);
143    
144                final TiffImageMetadata.Directory metadataDirectory = new TiffImageMetadata.Directory(
145                        tiffReader.getByteOrder(), dir);
146    
147                final List<TiffField> entries = dir.getDirectoryEntrys();
148    
149                for (int j = 0; j < entries.size(); j++) {
150                    final TiffField entry = entries.get(j);
151                    metadataDirectory.add(entry);
152                }
153    
154                result.add(metadataDirectory);
155            }
156    
157            return result;
158        }

160        @Override
161        public ImageInfo getImageInfo(final ByteSource byteSource, final Map<String,Object> params)
162                throws ImageReadException, IOException {
163            final FormatCompliance formatCompliance = FormatCompliance.getDefault();
164            final TiffContents contents = new TiffReader(isStrict(params))
165                    .readDirectories(byteSource, false, formatCompliance);
166            final TiffDirectory directory = contents.directories.get(0);
167    
168            final TiffField widthField = directory.findField(
169                    TiffTagConstants.TIFF_TAG_IMAGE_WIDTH, true);
170            final TiffField heightField = directory.findField(
171                    TiffTagConstants.TIFF_TAG_IMAGE_LENGTH, true);
172    
173            if ((widthField == null) || (heightField == null)) {
174                throw new ImageReadException("TIFF image missing size info.");
175            }
176    
177            final int height = heightField.getIntValue();
178            final int width = widthField.getIntValue();
179    
180            // -------------------
181    
182            final TiffField resolutionUnitField = directory
183                    .findField(TiffTagConstants.TIFF_TAG_RESOLUTION_UNIT);
184            int resolutionUnit = 2; // Inch
185            if ((resolutionUnitField != null)
186                    && (resolutionUnitField.getValue() != null)) {
187                resolutionUnit = resolutionUnitField.getIntValue();
188            }
189    
190            double unitsPerInch = -1;
191            switch (resolutionUnit) {
192            case 1:
193                break;
194            case 2: // Inch
195                unitsPerInch = 1.0;
196                break;
197            case 3: // Centimeter
198                unitsPerInch = 2.54;
199                break;
200            default:
201                break;
202    
203            }
204            final TiffField xResolutionField = directory
205                    .findField(TiffTagConstants.TIFF_TAG_XRESOLUTION);
206            final TiffField yResolutionField = directory
207                    .findField(TiffTagConstants.TIFF_TAG_YRESOLUTION);
208    
209            int physicalWidthDpi = -1;
210            float physicalWidthInch = -1;
211            int physicalHeightDpi = -1;
212            float physicalHeightInch = -1;
213    
214            if (unitsPerInch > 0) {
215                if ((xResolutionField != null)
216                        && (xResolutionField.getValue() != null)) {
217                    final double XResolutionPixelsPerUnit = xResolutionField
218                            .getDoubleValue();
219                    physicalWidthDpi = (int) Math
220                            .round((XResolutionPixelsPerUnit * unitsPerInch));
221                    physicalWidthInch = (float) (width / (XResolutionPixelsPerUnit * unitsPerInch));
222                }
223                if ((yResolutionField != null)
224                        && (yResolutionField.getValue() != null)) {
225                    final double YResolutionPixelsPerUnit = yResolutionField
226                            .getDoubleValue();
227                    physicalHeightDpi = (int) Math
228                            .round((YResolutionPixelsPerUnit * unitsPerInch));
229                    physicalHeightInch = (float) (height / (YResolutionPixelsPerUnit * unitsPerInch));
230                }
231            }
232    
233            // -------------------
234    
235            final TiffField bitsPerSampleField = directory
236                    .findField(TiffTagConstants.TIFF_TAG_BITS_PER_SAMPLE);
237    
238            int bitsPerSample = 1;
239            if ((bitsPerSampleField != null)
240                    && (bitsPerSampleField.getValue() != null)) {
241                bitsPerSample = bitsPerSampleField.getIntValueOrArraySum();
242            }
243    
244            final int bitsPerPixel = bitsPerSample; // assume grayscale;
245            // dunno if this handles colormapped images correctly.
246    
247            // -------------------
248    
249            final List<String> comments = new ArrayList<String>();
250            final List<TiffField> entries = directory.entries;
251            for (int i = 0; i < entries.size(); i++) {
252                final TiffField field = entries.get(i);
253                final String comment = field.toString();
254                comments.add(comment);
255            }
256    
257            final ImageFormat format = ImageFormat.IMAGE_FORMAT_TIFF;
258            final String formatName = "TIFF Tag-based Image File Format";
259            final String mimeType = "image/tiff";
260            final int numberOfImages = contents.directories.size();
261            // not accurate ... only reflects first
262            final boolean isProgressive = false;
263            // is TIFF ever interlaced/progressive?
264    
265            final String formatDetails = "Tiff v." + contents.header.tiffVersion;
266    
267            final boolean isTransparent = false; // TODO: wrong
268            boolean usesPalette = false;
269            final TiffField colorMapField = directory
270                    .findField(TiffTagConstants.TIFF_TAG_COLOR_MAP);
271            if (colorMapField != null) {
272                usesPalette = true;
273            }
274    
275            final int colorType = ImageInfo.COLOR_TYPE_RGB;
276    
277            final int compression = 0xffff & directory
278                    .getSingleFieldValue(TiffTagConstants.TIFF_TAG_COMPRESSION);
279            String compressionAlgorithm;
280    
281            switch (compression) {
282            case TIFF_COMPRESSION_UNCOMPRESSED_1:
283                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_NONE;
284                break;
285            case TIFF_COMPRESSION_CCITT_1D:
286                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_CCITT_1D;
287                break;
288            case TIFF_COMPRESSION_CCITT_GROUP_3:
289                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_CCITT_GROUP_3;
290                break;
291            case TIFF_COMPRESSION_CCITT_GROUP_4:
292                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_CCITT_GROUP_4;
293                break;
294            case TIFF_COMPRESSION_LZW:
295                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_LZW;
296                break;
297            case TIFF_COMPRESSION_JPEG:
298                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_JPEG;
299                break;
300            case TIFF_COMPRESSION_UNCOMPRESSED_2:
301                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_NONE;
302                break;
303            case TIFF_COMPRESSION_PACKBITS:
304                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_PACKBITS;
305                break;
306            default:
307                compressionAlgorithm = ImageInfo.COMPRESSION_ALGORITHM_UNKNOWN;
308                break;
309            }
310    
311            final ImageInfo result = new ImageInfo(formatDetails, bitsPerPixel, comments,
312                    format, formatName, height, mimeType, numberOfImages,
313                    physicalHeightDpi, physicalHeightInch, physicalWidthDpi,
314                    physicalWidthInch, width, isProgressive, isTransparent,
315                    usesPalette, colorType, compressionAlgorithm);
316    
317            return result;
318        }

320        @Override
321        public String getXmpXml(final ByteSource byteSource, final Map<String,Object> params)
322                throws ImageReadException, IOException {
323            final FormatCompliance formatCompliance = FormatCompliance.getDefault();
324            final TiffContents contents = new TiffReader(isStrict(params))
325                    .readDirectories(byteSource, false, formatCompliance);
326            final TiffDirectory directory = contents.directories.get(0);
327    
328            final byte bytes[] = directory.getFieldValue(TiffTagConstants.TIFF_TAG_XMP,
329                    false);
330            if (bytes == null) {
331                return null;
332            }
333    
334            try {
335                // segment data is UTF-8 encoded xml.
336                final String xml = new String(bytes, "utf-8");
337                return xml;
338            } catch (final UnsupportedEncodingException e) {
339                throw new ImageReadException("Invalid JPEG XMP Segment.");
340            }
341        }

344        public boolean dumpImageFile(final PrintWriter pw, final ByteSource byteSource)
345                throws ImageReadException, IOException {
346            try {
347                pw.println("tiff.dumpImageFile");
348    
349                {
350                    final ImageInfo imageData = getImageInfo(byteSource);
351                    if (imageData == null) {
352                        return false;
353                    }
354    
355                    imageData.toString(pw, "");
356                }
357    
358                pw.println("");
359    
360                // try
361                {
362                    final FormatCompliance formatCompliance = FormatCompliance
363                            .getDefault();
364                    final Map<String,Object> params = null;
365                    final TiffContents contents = new TiffReader(true).readContents(
366                            byteSource, params, formatCompliance);
367    
368                    final List<TiffDirectory> directories = contents.directories;
369    
370                    if (directories == null) {
371                        return false;
372                    }
373    
374                    for (int d = 0; d < directories.size(); d++) {
375                        final TiffDirectory directory = directories.get(d);
376    
377                        final List<TiffField> entries = directory.entries;
378    
379                        if (entries == null) {
380                            return false;
381                        }
382    
383                        // Debug.debug("directory offset", directory.offset);
384    
385                        for (int i = 0; i < entries.size(); i++) {
386                            final TiffField field = entries.get(i);
387    
388                            field.dump(pw, d + "");
389                        }
390                    }
391    
392                    pw.println("");
393                }
394                // catch (Exception e)
395                // {
396                // Debug.debug(e);
397                // pw.println("");
398                // return false;
399                // }
400    
401                return true;
402            } finally {
403                pw.println("");
404            }

417        public List<byte[]> collectRawImageData(final ByteSource byteSource, final Map<String,Object> params)
418                throws ImageReadException, IOException {
419            final FormatCompliance formatCompliance = FormatCompliance.getDefault();
420            final TiffContents contents = new TiffReader(isStrict(params))
421                    .readDirectories(byteSource, true, formatCompliance);
422    
423            final List<byte[]> result = new ArrayList<byte[]>();
424            for (int i = 0; i < contents.directories.size(); i++) {
425                final TiffDirectory directory = contents.directories.get(i);
426                final List<ImageDataElement> dataElements = directory
427                        .getTiffRawImageDataElements();
428                for (int j = 0; j < dataElements.size(); j++) {
429                    final TiffDirectory.ImageDataElement element = dataElements.get(j);
430                    final byte bytes[] = byteSource.getBlock(element.offset,
431                            element.length);
432                    result.add(bytes);
433                }
434            }
435            return result;
436        }

439        public BufferedImage getBufferedImage(final ByteSource byteSource, final Map<String,Object> params)
440                throws ImageReadException, IOException {
441            final FormatCompliance formatCompliance = FormatCompliance.getDefault();
442            final TiffReader reader = new TiffReader(isStrict(params));
443            final TiffContents contents = reader.readFirstDirectory(byteSource, params,
444                    true, formatCompliance);
445            final ByteOrder byteOrder = reader.getByteOrder();
446            final TiffDirectory directory = contents.directories.get(0);
447            final BufferedImage result = directory.getTiffImage(byteOrder, params);
448            if (null == result) {
449                throw new ImageReadException("TIFF does not contain an image.");
450            }
451            return result;
452        }

473        protected BufferedImage getBufferedImage(final TiffDirectory directory,
474                final ByteOrder byteOrder, final Map<String,Object> params) throws ImageReadException, IOException {
475            final List<TiffField> entries = directory.entries;
476    
477            if (entries == null) {
478                throw new ImageReadException("TIFF missing entries");
479            }
480    
481            final int photometricInterpretation = 0xffff & directory
482                    .getSingleFieldValue(TiffTagConstants.TIFF_TAG_PHOTOMETRIC_INTERPRETATION);
483            final int compression = 0xffff & directory
484                    .getSingleFieldValue(TiffTagConstants.TIFF_TAG_COMPRESSION);
485            final int width = directory.findField(TiffTagConstants.TIFF_TAG_IMAGE_WIDTH,
486                    true).getIntValue();
487            final int height = directory.findField(
488                    TiffTagConstants.TIFF_TAG_IMAGE_LENGTH, true).getIntValue();
489            int samplesPerPixel = 1;
490            final TiffField samplesPerPixelField = directory
491                    .findField(TiffTagConstants.TIFF_TAG_SAMPLES_PER_PIXEL);
492            if (samplesPerPixelField != null) {
493                samplesPerPixel = samplesPerPixelField.getIntValue();
494            }
495            int bitsPerSample[] = { 1 };
496            int bitsPerPixel = samplesPerPixel;
497            final TiffField bitsPerSampleField = directory
498                    .findField(TiffTagConstants.TIFF_TAG_BITS_PER_SAMPLE);
499            if (bitsPerSampleField != null) {
500                bitsPerSample = bitsPerSampleField.getIntArrayValue();
501                bitsPerPixel = bitsPerSampleField.getIntValueOrArraySum();
502            }
503    
504            // int bitsPerPixel = getTagAsValueOrArraySum(entries,
505            // TIFF_TAG_BITS_PER_SAMPLE);
506    
507            int predictor = -1;
508            {
509                // dumpOptionalNumberTag(entries, TIFF_TAG_FILL_ORDER);
510                // dumpOptionalNumberTag(entries, TIFF_TAG_FREE_BYTE_COUNTS);
511                // dumpOptionalNumberTag(entries, TIFF_TAG_FREE_OFFSETS);
512                // dumpOptionalNumberTag(entries, TIFF_TAG_ORIENTATION);
513                // dumpOptionalNumberTag(entries, TIFF_TAG_PLANAR_CONFIGURATION);
514                final TiffField predictorField = directory
515                        .findField(TiffTagConstants.TIFF_TAG_PREDICTOR);
516                if (null != predictorField) {
517                    predictor = predictorField.getIntValueOrArraySum();
518                }
519            }
520    
521            if (samplesPerPixel != bitsPerSample.length) {
522                throw new ImageReadException("Tiff: samplesPerPixel ("
523                        + samplesPerPixel + ")!=fBitsPerSample.length ("
524                        + bitsPerSample.length + ")");
525            }
526    
527            final boolean hasAlpha = false;
528            final ImageBuilder imageBuilder = new ImageBuilder(width, height, hasAlpha);
529    
530            final PhotometricInterpreter photometricInterpreter = getPhotometricInterpreter(
531                    directory, photometricInterpretation, bitsPerPixel,
532                    bitsPerSample, predictor, samplesPerPixel, width, height);
533    
534            final TiffImageData imageData = directory.getTiffImageData();
535    
536            final DataReader dataReader = imageData.getDataReader(directory,
537                    photometricInterpreter, bitsPerPixel, bitsPerSample, predictor,
538                    samplesPerPixel, width, height, compression, byteOrder);
539    
540            dataReader.readImageData(imageBuilder);
541    
542            photometricInterpreter.dumpstats();
543    
544            return imageBuilder.getBufferedImage();
545        }

547        private PhotometricInterpreter getPhotometricInterpreter(
548                final TiffDirectory directory, final int photometricInterpretation,
549                final int bitsPerPixel, final int bitsPerSample[], final int predictor,
550                final int samplesPerPixel, final int width, final int height)
551                throws ImageReadException {
552            switch (photometricInterpretation) {
553            case 0:
554            case 1:
555                final boolean invert = photometricInterpretation == 0;
556    
557                return new PhotometricInterpreterBiLevel(bitsPerPixel,
558                        samplesPerPixel, bitsPerSample, predictor, width, height,
559                        invert);
560            case 3: // Palette
561            {
562                final int colorMap[] = directory.findField(
563                        TiffTagConstants.TIFF_TAG_COLOR_MAP, true)
564                        .getIntArrayValue();
565    
566                final int expected_colormap_size = 3 * (1 << bitsPerPixel);
567    
568                if (colorMap.length != expected_colormap_size) {
569                    throw new ImageReadException("Tiff: fColorMap.length ("
570                            + colorMap.length + ")!=expected_colormap_size ("
571                            + expected_colormap_size + ")");
572                }
573    
574                return new PhotometricInterpreterPalette(samplesPerPixel,
575                        bitsPerSample, predictor, width, height, colorMap);
576            }
577            case 2: // RGB
578                return new PhotometricInterpreterRgb(samplesPerPixel,
579                        bitsPerSample, predictor, width, height);
580            case 5: // CMYK
581                return new PhotometricInterpreterCmyk(samplesPerPixel,
582                        bitsPerSample, predictor, width, height);
583            case 6: //
584            {
585                final double yCbCrCoefficients[] = directory.findField(
586                        TiffTagConstants.TIFF_TAG_YCBCR_COEFFICIENTS, true)
587                        .getDoubleArrayValue();
588    
589                final int yCbCrPositioning[] = directory.findField(
590                        TiffTagConstants.TIFF_TAG_YCBCR_POSITIONING, true)
591                        .getIntArrayValue();
592                final int yCbCrSubSampling[] = directory.findField(
593                        TiffTagConstants.TIFF_TAG_YCBCR_SUB_SAMPLING, true)
594                        .getIntArrayValue();
595    
596                final double referenceBlackWhite[] = directory.findField(
597                        TiffTagConstants.TIFF_TAG_REFERENCE_BLACK_WHITE, true)
598                        .getDoubleArrayValue();
599    
600                return new PhotometricInterpreterYCbCr(yCbCrCoefficients,
601                        yCbCrPositioning, yCbCrSubSampling, referenceBlackWhite,
602                        samplesPerPixel, bitsPerSample, predictor, width, height);
603            }
604    
605            case 8:
606                return new PhotometricInterpreterCieLab(samplesPerPixel,
607                        bitsPerSample, predictor, width, height);
608    
609            case 32844:
610            case 32845: {
611                final boolean yonly = (photometricInterpretation == 32844);
612                return new PhotometricInterpreterLogLuv(samplesPerPixel,
613                        bitsPerSample, predictor, width, height, yonly);
614            }
615    
616            default:
617                throw new ImageReadException(
618                        "TIFF: Unknown fPhotometricInterpretation: "
619                                + photometricInterpretation);
620            }
621        }
