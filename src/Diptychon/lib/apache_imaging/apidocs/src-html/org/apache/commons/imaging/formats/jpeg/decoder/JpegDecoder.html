040    public class JpegDecoder extends BinaryFileParser implements JpegUtils.Visitor,
041            JpegConstants {
042        /*
043         * JPEG is an advanced image format that takes significant computation to
044         * decode. Keep decoding fast: - Don't allocate memory inside loops,
045         * allocate it once and reuse. - Minimize calculations per pixel and per
046         * block (using lookup tables for YCbCr->RGB conversion doubled
047         * performance). - Math.round() is slow, use (int)(x+0.5f) instead for
048         * positive numbers.
049         */
050

052        private final DhtSegment.HuffmanTable[] huffmanDCTables = new DhtSegment.HuffmanTable[4];
053        private final DhtSegment.HuffmanTable[] huffmanACTables = new DhtSegment.HuffmanTable[4];
054        private SofnSegment sofnSegment;

057        private BufferedImage image = null;

065        public void visitSOS(final int marker, final byte markerBytes[], final byte imageData[]) {
066            final ByteArrayInputStream is = new ByteArrayInputStream(imageData);
067            try {
068                final int segmentLength = read2Bytes("segmentLength", is,
069                        "Not a Valid JPEG File");
070                final byte[] sosSegmentBytes = readByteArray("SosSegment",
071                        segmentLength - 2, is, "Not a Valid JPEG File");
072                sosSegment = new SosSegment(marker, sosSegmentBytes);
073    
074                int hMax = 0;
075                int vMax = 0;
076                for (int i = 0; i < sofnSegment.numberOfComponents; i++) {
077                    hMax = Math.max(hMax,
078                            sofnSegment.components[i].horizontalSamplingFactor);
079                    vMax = Math.max(vMax,
080                            sofnSegment.components[i].verticalSamplingFactor);
081                }
082                final int hSize = 8 * hMax;
083                final int vSize = 8 * vMax;
084    
085                final JpegInputStream bitInputStream = new JpegInputStream(is);
086                final int xMCUs = (sofnSegment.width + hSize - 1) / hSize;
087                final int yMCUs = (sofnSegment.height + vSize - 1) / vSize;
088                final Block[] mcu = allocateMCUMemory();
089                final Block[] scaledMCU = new Block[mcu.length];
090                for (int i = 0; i < scaledMCU.length; i++) {
091                    scaledMCU[i] = new Block(hSize, vSize);
092                }
093                final int[] preds = new int[sofnSegment.numberOfComponents];
094                ColorModel colorModel;
095                WritableRaster raster;
096                if (sofnSegment.numberOfComponents == 3) {
097                    colorModel = new DirectColorModel(24, 0x00ff0000, 0x0000ff00,
098                            0x000000ff);
099                    raster = WritableRaster.createPackedRaster(DataBuffer.TYPE_INT,
100                            sofnSegment.width, sofnSegment.height, new int[] {
101                                    0x00ff0000, 0x0000ff00, 0x000000ff }, null);
102                } else if (sofnSegment.numberOfComponents == 1) {
103                    colorModel = new DirectColorModel(24, 0x00ff0000, 0x0000ff00,
104                            0x000000ff);
105                    raster = WritableRaster.createPackedRaster(DataBuffer.TYPE_INT,
106                            sofnSegment.width, sofnSegment.height, new int[] {
107                                    0x00ff0000, 0x0000ff00, 0x000000ff }, null);
108                    // FIXME: why do images come out too bright with CS_GRAY?
109                    // colorModel = new ComponentColorModel(
110                    // ColorSpace.getInstance(ColorSpace.CS_GRAY), false, true,
111                    // Transparency.OPAQUE, DataBuffer.TYPE_BYTE);
112                    // raster = colorModel.createCompatibleWritableRaster(
113                    // sofnSegment.width, sofnSegment.height);
114                } else {
115                    throw new ImageReadException(sofnSegment.numberOfComponents
116                            + " components are invalid or unsupported");
117                }
118                final DataBuffer dataBuffer = raster.getDataBuffer();
119    
120                for (int y1 = 0; y1 < vSize * yMCUs; y1 += vSize) {
121                    for (int x1 = 0; x1 < hSize * xMCUs; x1 += hSize) {
122                        readMCU(bitInputStream, preds, mcu);
123                        rescaleMCU(mcu, hSize, vSize, scaledMCU);
124                        int srcRowOffset = 0;
125                        int dstRowOffset = y1 * sofnSegment.width + x1;
126                        for (int y2 = 0; y2 < vSize && y1 + y2 < sofnSegment.height; y2++) {
127                            for (int x2 = 0; x2 < hSize
128                                    && x1 + x2 < sofnSegment.width; x2++) {
129                                if (scaledMCU.length == 3) {
130                                    final int Y = scaledMCU[0].samples[srcRowOffset + x2];
131                                    final int Cb = scaledMCU[1].samples[srcRowOffset + x2];
132                                    final int Cr = scaledMCU[2].samples[srcRowOffset + x2];
133                                    final int rgb = YCbCrConverter.convertYCbCrToRGB(Y,
134                                            Cb, Cr);
135                                    dataBuffer.setElem(dstRowOffset + x2, rgb);
136                                } else if (mcu.length == 1) {
137                                    final int Y = scaledMCU[0].samples[srcRowOffset + x2];
138                                    dataBuffer.setElem(dstRowOffset + x2, (Y << 16)
139                                            | (Y << 8) | Y);
140                                } else {
141                                    throw new ImageReadException(
142                                            "Unsupported JPEG with " + mcu.length
143                                                    + " components");
144                                }
145                            }
146                            srcRowOffset += hSize;
147                            dstRowOffset += sofnSegment.width;
148                        }
149                    }
150                }
151                image = new BufferedImage(colorModel, raster,
152                        colorModel.isAlphaPremultiplied(), new Properties());
153                // byte[] remainder = super.getStreamBytes(is);
154                // for (int i = 0; i < remainder.length; i++)
155                // {
156                // System.out.println("" + i + " = " +
157                // Integer.toHexString(remainder[i]));
158                // }
159            } catch (final ImageReadException imageReadEx) {
160                imageReadException = imageReadEx;
161            } catch (final IOException ioEx) {
162                ioException = ioEx;
163            } catch (final RuntimeException ex) {
164                // Corrupt images can throw NPE and IOOBE
165                imageReadException = new ImageReadException("Error parsing JPEG",
166                        ex);
167            }
168        }

170        public boolean visitSegment(final int marker, final byte[] markerBytes,
171                final int segmentLength, final byte[] segmentLengthBytes, final byte[] segmentData)
172                throws ImageReadException, IOException {
173            final int[] sofnSegments = { SOF0Marker, SOF1Marker, SOF2Marker,
174                    SOF3Marker, SOF5Marker, SOF6Marker, SOF7Marker, SOF9Marker,
175                    SOF10Marker, SOF11Marker, SOF13Marker, SOF14Marker,
176                    SOF15Marker, };
177    
178            if (Arrays.binarySearch(sofnSegments, marker) >= 0) {
179                if (marker != SOF0Marker) {
180                    throw new ImageReadException("Only sequential, baseline JPEGs "
181                            + "are supported at the moment");
182                }
183                sofnSegment = new SofnSegment(marker, segmentData);
184            } else if (marker == DQTMarker) {
185                final DqtSegment dqtSegment = new DqtSegment(marker, segmentData);
186                for (int i = 0; i < dqtSegment.quantizationTables.size(); i++) {
187                    final DqtSegment.QuantizationTable table = dqtSegment.quantizationTables
188                            .get(i);
189                    if (0 > table.destinationIdentifier
190                            || table.destinationIdentifier >= quantizationTables.length) {
191                        throw new ImageReadException(
192                                "Invalid quantization table identifier "
193                                        + table.destinationIdentifier);
194                    }
195                    quantizationTables[table.destinationIdentifier] = table;
196                    final int[] quantizationMatrixInt = new int[64];
197                    ZigZag.zigZagToBlock(table.elements, quantizationMatrixInt);
198                    final float[] quantizationMatrixFloat = new float[64];
199                    for (int j = 0; j < 64; j++) {
200                        quantizationMatrixFloat[j] = quantizationMatrixInt[j];
201                    }
202                    Dct.scaleDequantizationMatrix(quantizationMatrixFloat);
203                    scaledQuantizationTables[table.destinationIdentifier] = quantizationMatrixFloat;
204                }
205            } else if (marker == DHTMarker) {
206                final DhtSegment dhtSegment = new DhtSegment(marker, segmentData);
207                for (int i = 0; i < dhtSegment.huffmanTables.size(); i++) {
208                    final DhtSegment.HuffmanTable table = dhtSegment.huffmanTables.get(i);
209                    DhtSegment.HuffmanTable[] tables;
210                    if (table.tableClass == 0) {
211                        tables = huffmanDCTables;
212                    } else if (table.tableClass == 1) {
213                        tables = huffmanACTables;
214                    } else {
215                        throw new ImageReadException("Invalid huffman table class "
216                                + table.tableClass);
217                    }
218                    if (0 > table.destinationIdentifier
219                            || table.destinationIdentifier >= tables.length) {
220                        throw new ImageReadException(
221                                "Invalid huffman table identifier "
222                                        + table.destinationIdentifier);
223                    }
224                    tables[table.destinationIdentifier] = table;
225                }
226            }
227            return true;
228        }

230        private void rescaleMCU(final Block[] dataUnits, final int hSize, final int vSize, final Block[] ret) {
231            for (int i = 0; i < dataUnits.length; i++) {
232                final Block block = dataUnits[i];
233                if (block.width == hSize && block.height == vSize) {
234                    System.arraycopy(block.samples, 0, ret[i].samples, 0, hSize
235                            * vSize);
236                } else {
237                    final int hScale = hSize / block.width;
238                    final int vScale = vSize / block.height;
239                    if (hScale == 2 && vScale == 2) {
240                        int srcRowOffset = 0;
241                        int dstRowOffset = 0;
242                        for (int y = 0; y < block.height; y++) {
243                            for (int x = 0; x < hSize; x++) {
244                                final int sample = block.samples[srcRowOffset + (x >> 1)];
245                                ret[i].samples[dstRowOffset + x] = sample;
246                                ret[i].samples[dstRowOffset + hSize + x] = sample;
247                            }
248                            srcRowOffset += block.width;
249                            dstRowOffset += 2 * hSize;
250                        }
251                    } else {
252                        // FIXME: optimize
253                        int dstRowOffset = 0;
254                        for (int y = 0; y < vSize; y++) {
255                            for (int x = 0; x < hSize; x++) {
256                                ret[i].samples[dstRowOffset + x] = block.samples[(y / vScale)
257                                        * block.width + (x / hScale)];
258                            }
259                            dstRowOffset += hSize;
260                        }
261                    }
262                }
263            }
264        }

266        private Block[] allocateMCUMemory() throws ImageReadException {
267            final Block[] mcu = new Block[sosSegment.numberOfComponents];
268            for (int i = 0; i < sosSegment.numberOfComponents; i++) {
269                final SosSegment.Component scanComponent = sosSegment.components[i];
270                SofnSegment.Component frameComponent = null;
271                for (int j = 0; j < sofnSegment.numberOfComponents; j++) {
272                    if (sofnSegment.components[j].componentIdentifier == scanComponent.scanComponentSelector) {
273                        frameComponent = sofnSegment.components[j];
274                        break;
275                    }
276                }
277                if (frameComponent == null) {
278                    throw new ImageReadException("Invalid component");
279                }
280                final Block fullBlock = new Block(
281                        8 * frameComponent.horizontalSamplingFactor,
282                        8 * frameComponent.verticalSamplingFactor);
283                mcu[i] = fullBlock;
284            }
285            return mcu;
286        }

288        private final int[] zz = new int[64];
289        private final int[] blockInt = new int[64];
290        private final float[] block = new float[64];
291    
292        private void readMCU(final JpegInputStream is, final int[] preds, final Block[] mcu)
293                throws IOException, ImageReadException {
294            for (int i = 0; i < sosSegment.numberOfComponents; i++) {
295                final SosSegment.Component scanComponent = sosSegment.components[i];
296                SofnSegment.Component frameComponent = null;
297                for (int j = 0; j < sofnSegment.numberOfComponents; j++) {
298                    if (sofnSegment.components[j].componentIdentifier == scanComponent.scanComponentSelector) {
299                        frameComponent = sofnSegment.components[j];
300                        break;
301                    }
302                }
303                if (frameComponent == null) {
304                    throw new ImageReadException("Invalid component");
305                }
306                final Block fullBlock = mcu[i];
307                for (int y = 0; y < frameComponent.verticalSamplingFactor; y++) {
308                    for (int x = 0; x < frameComponent.horizontalSamplingFactor; x++) {
309                        Arrays.fill(zz, 0);
310                        // page 104 of T.81
311                        final int t = decode(
312                                is,
313                                huffmanDCTables[scanComponent.dcCodingTableSelector]);
314                        int diff = receive(t, is);
315                        diff = extend(diff, t);
316                        zz[0] = preds[i] + diff;
317                        preds[i] = zz[0];
318    
319                        // "Decode_AC_coefficients", figure F.13, page 106 of T.81
320                        int k = 1;
321                        while (true) {
322                            final int rs = decode(
323                                    is,
324                                    huffmanACTables[scanComponent.acCodingTableSelector]);
325                            final int ssss = rs & 0xf;
326                            final int rrrr = rs >> 4;
327                            final int r = rrrr;
328    
329                            if (ssss == 0) {
330                                if (r == 15) {
331                                    k += 16;
332                                } else {
333                                    break;
334                                }
335                            } else {
336                                k += r;
337    
338                                // "Decode_ZZ(k)", figure F.14, page 107 of T.81
339                                zz[k] = receive(ssss, is);
340                                zz[k] = extend(zz[k], ssss);
341    
342                                if (k == 63) {
343                                    break;
344                                } else {
345                                    k++;
346                                }
347                            }
348                        }
349    
350                        final int shift = (1 << (sofnSegment.precision - 1));
351                        final int max = (1 << sofnSegment.precision) - 1;
352    
353                        final float[] scaledQuantizationTable = scaledQuantizationTables[frameComponent.quantTabDestSelector];
354                        ZigZag.zigZagToBlock(zz, blockInt);
355                        for (int j = 0; j < 64; j++) {
356                            block[j] = blockInt[j] * scaledQuantizationTable[j];
357                        }
358                        Dct.inverseDCT8x8(block);
359    
360                        int dstRowOffset = 8 * y * 8
361                                * frameComponent.horizontalSamplingFactor + 8 * x;
362                        int srcNext = 0;
363                        for (int yy = 0; yy < 8; yy++) {
364                            for (int xx = 0; xx < 8; xx++) {
365                                float sample = block[srcNext++];
366                                sample += shift;
367                                int result;
368                                if (sample < 0) {
369                                    result = 0;
370                                } else if (sample > max) {
371                                    result = max;
372                                } else {
373                                    result = fastRound(sample);
374                                }
375                                fullBlock.samples[dstRowOffset + xx] = result;
376                            }
377                            dstRowOffset += 8 * frameComponent.horizontalSamplingFactor;
378                        }
379                    }
380                }
381            }
382        }
383    
384        private static int fastRound(final float x) {
385            return (int) (x + 0.5f);
386        }
387    
388        private int extend(int v, final int t) {
389            // "EXTEND", section F.2.2.1, figure F.12, page 105 of T.81
390            int vt = (1 << (t - 1));
391            while (v < vt) {
392                vt = (-1 << t) + 1;
393                v += vt;
394            }
395            return v;
396        }

398        private int receive(final int ssss, final JpegInputStream is) throws IOException,
399                ImageReadException {
400            // "RECEIVE", section F.2.2.4, figure F.17, page 110 of T.81
401            int i = 0;
402            int v = 0;
403            while (i != ssss) {
404                i++;
405                v = (v << 1) + is.nextBit();
406            }
407            return v;
408        }

410        private int decode(final JpegInputStream is, final DhtSegment.HuffmanTable huffmanTable)
411                throws IOException, ImageReadException {
412            // "DECODE", section F.2.2.3, figure F.16, page 109 of T.81
413            int i = 1;
414            int code = is.nextBit();
415            while (code > huffmanTable.maxCode[i]) {
416                i++;
417                code = (code << 1) | is.nextBit();
418            }
419            int j = huffmanTable.valPtr[i];
420            j += code - huffmanTable.minCode[i];
421            final int value = huffmanTable.huffVal[j];
422            return value;
423        }
424    
425        public BufferedImage decode(final ByteSource byteSource) throws IOException,
426                ImageReadException {
427            final JpegUtils jpegUtils = new JpegUtils();
428            jpegUtils.traverseJFIF(byteSource, this);
429            if (imageReadException != null) {
430                throw imageReadException;
431            }
432            if (ioException != null) {
433                throw ioException;
434            }
435            return image;
436        }
