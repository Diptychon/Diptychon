420        public void removeField(final int tag) {
421            final List<TiffOutputField> matches = new ArrayList<TiffOutputField>();
422            for (int i = 0; i < fields.size(); i++) {
423                final TiffOutputField field = fields.get(i);
424                if (field.tag == tag) {
425                    matches.add(field);
426                }
427            }
428            fields.removeAll(matches);
429        }

435        public TiffOutputField findField(final int tag) {
436            for (int i = 0; i < fields.size(); i++) {
437                final TiffOutputField field = fields.get(i);
438                if (field.tag == tag) {
439                    return field;
440                }
441            }
442            return null;
443        }

445        public void sortFields() {
446            final Comparator<TiffOutputField> comparator = new Comparator<TiffOutputField>() {
447                public int compare(final TiffOutputField e1, final TiffOutputField e2) {
448                    if (e1.tag != e2.tag) {
449                        return e1.tag - e2.tag;
450                    }
451                    return e1.getSortHint() - e2.getSortHint();
452                }
453            };
454            Collections.sort(fields, comparator);
455        }

461        @Override
462        public void writeItem(final BinaryOutputStream bos) throws IOException,
463                ImageWriteException {
464            // Write Directory Field Count
465            bos.write2Bytes(fields.size()); // DirectoryFieldCount
466    
467            // Write Fields
468            for (int i = 0; i < fields.size(); i++) {
469                final TiffOutputField field = fields.get(i);
470                field.writeField(bos);
471    
472                // Debug.debug("\t" + "writing field (" + field.tag + ", 0x" +
473                // Integer.toHexString(field.tag) + ")", field.tagInfo);
474                // if(field.tagInfo.isOffset())
475                // Debug.debug("\t\tOFFSET!", field.bytes);
476            }
477    
478            int nextDirectoryOffset = 0;
479            if (nextDirectory != null) {
480                nextDirectoryOffset = nextDirectory.getOffset();
481            }
482    
483            // Write nextDirectoryOffset
484            if (nextDirectoryOffset == UNDEFINED_VALUE) {
485                bos.write4Bytes(0);
486            } else {
487                bos.write4Bytes(nextDirectoryOffset);
488            }
489        }

531        protected List<TiffOutputItem> getOutputItems(
532                final TiffOutputSummary outputSummary) throws ImageWriteException {
533            // first validate directory fields.
534    
535            removeFieldIfPresent(TiffTagConstants.TIFF_TAG_JPEG_INTERCHANGE_FORMAT);
536            removeFieldIfPresent(TiffTagConstants.TIFF_TAG_JPEG_INTERCHANGE_FORMAT_LENGTH);
537    
538            TiffOutputField jpegOffsetField = null;
539            if (null != jpegImageData) {
540                jpegOffsetField = new TiffOutputField(
541                        TiffTagConstants.TIFF_TAG_JPEG_INTERCHANGE_FORMAT,
542                        FIELD_TYPE_LONG, 1, FieldType.getStubLocalValue());
543                add(jpegOffsetField);
544    
545                final byte lengthValue[] = FIELD_TYPE_LONG.writeData(
546                        new int[] { jpegImageData.length, },
547                        outputSummary.byteOrder);
548    
549                final TiffOutputField jpegLengthField = new TiffOutputField(
550                        TiffTagConstants.TIFF_TAG_JPEG_INTERCHANGE_FORMAT_LENGTH,
551                        FIELD_TYPE_LONG, 1, lengthValue);
552                add(jpegLengthField);
553    
554            }
555    
556            // --------------------------------------------------------------
557    
558            removeFieldIfPresent(TiffTagConstants.TIFF_TAG_STRIP_OFFSETS);
559            removeFieldIfPresent(TiffTagConstants.TIFF_TAG_STRIP_BYTE_COUNTS);
560            removeFieldIfPresent(TiffTagConstants.TIFF_TAG_TILE_OFFSETS);
561            removeFieldIfPresent(TiffTagConstants.TIFF_TAG_TILE_BYTE_COUNTS);
562    
563            TiffOutputField imageDataOffsetField;
564            ImageDataOffsets imageDataInfo = null;
565            if (null != tiffImageData) {
566                final boolean stripsNotTiles = tiffImageData.stripsNotTiles();
567    
568                TagInfo offsetTag;
569                TagInfo byteCountsTag;
570                if (stripsNotTiles) {
571                    offsetTag = TiffTagConstants.TIFF_TAG_STRIP_OFFSETS;
572                    byteCountsTag = TiffTagConstants.TIFF_TAG_STRIP_BYTE_COUNTS;
573                } else {
574                    offsetTag = TiffTagConstants.TIFF_TAG_TILE_OFFSETS;
575                    byteCountsTag = TiffTagConstants.TIFF_TAG_TILE_BYTE_COUNTS;
576                }
577    
578                // --------
579    
580                final TiffElement.DataElement imageData[] = tiffImageData.getImageData();
581    
582                int imageDataOffsets[] = null;
583                int imageDataByteCounts[] = null;
584                // TiffOutputField imageDataOffsetsField = null;
585    
586                imageDataOffsets = new int[imageData.length];
587                imageDataByteCounts = new int[imageData.length];
588                for (int i = 0; i < imageData.length; i++) {
589                    imageDataByteCounts[i] = imageData[i].length;
590                }
591    
592                // --------
593    
594                // Append imageData-related fields to first directory
595                imageDataOffsetField = new TiffOutputField(offsetTag,
596                        FIELD_TYPE_LONG, imageDataOffsets.length,
597                        FIELD_TYPE_LONG.writeData(imageDataOffsets,
598                                outputSummary.byteOrder));
599                add(imageDataOffsetField);
600    
601                // --------
602    
603                final byte data[] = FIELD_TYPE_LONG.writeData(imageDataByteCounts,
604                        outputSummary.byteOrder);
605                final TiffOutputField byteCountsField = new TiffOutputField(
606                        byteCountsTag, FIELD_TYPE_LONG, imageDataByteCounts.length,
607                        data);
608                add(byteCountsField);
609    
610                // --------
611    
612                imageDataInfo = new ImageDataOffsets(imageData, imageDataOffsets,
613                        imageDataOffsetField);
614            }
615    
616            // --------------------------------------------------------------
617    
618            final List<TiffOutputItem> result = new ArrayList<TiffOutputItem>();
619            result.add(this);
620            sortFields();
621    
622            for (int i = 0; i < fields.size(); i++) {
623                final TiffOutputField field = fields.get(i);
624                if (field.isLocalValue()) {
625                    continue;
626                }
627    
628                final TiffOutputItem item = field.getSeperateValue();
629                result.add(item);
630                // outputSummary.add(item, field);
631            }
632    
633            if (null != imageDataInfo) {
634                for (final TiffOutputItem outputItem : imageDataInfo.outputItems) {
635                    result.add(outputItem);
636                }
637    
638                outputSummary.addTiffImageData(imageDataInfo);
639            }
640    
641            if (null != jpegImageData) {
642                final TiffOutputItem item = new TiffOutputItem.Value("JPEG image data",
643                        jpegImageData.data);
644                result.add(item);
645                outputSummary.add(item, jpegOffsetField);
646            }
647    
648            return result;
649        }
