016    package org.apache.commons.imaging.formats.jpeg;
017    
018    public final class Block {
019        public final int[] samples;
020        public final int width;
021        public final int height;
022    
023        public Block(final int width, final int height) {
024            samples = new int[width * height];
025            this.width = width;
026            this.height = height;
027        }
028    }
