032    public class PcxWriter implements PcxConstants {
033        private int encoding;
034        private int bitDepth = -1;
035        private PixelDensity pixelDensity = null;
036    
037        public PcxWriter(Map<String,Object> params) throws ImageWriteException {
038            // make copy of params; we'll clear keys as we consume them.
039            params = (params == null) ? new HashMap<String,Object>() : new HashMap<String,Object>(params);
040    
041            // clear format key.
042            if (params.containsKey(PARAM_KEY_FORMAT)) {
043                params.remove(PARAM_KEY_FORMAT);
044            }
045    
046            // uncompressed PCX files are not even documented in ZSoft's spec,
047            // let alone supported by most image viewers
048            encoding = PcxImageParser.PcxHeader.ENCODING_RLE;
049            if (params.containsKey(PARAM_KEY_PCX_COMPRESSION)) {
050                final Object value = params.remove(PARAM_KEY_PCX_COMPRESSION);
051                if (value != null) {
052                    if (!(value instanceof Number)) {
053                        throw new ImageWriteException(
054                                "Invalid compression parameter: " + value);
055                    }
056                    final int compression = ((Number) value).intValue();
057                    if (compression == PCX_COMPRESSION_UNCOMPRESSED) {
058                        encoding = PcxImageParser.PcxHeader.ENCODING_UNCOMPRESSED;
059                    }
060                }
061            }
062    
063            if (params.containsKey(PARAM_KEY_PCX_BIT_DEPTH)) {
064                final Object value = params.remove(PARAM_KEY_PCX_BIT_DEPTH);
065                if (value != null) {
066                    if (!(value instanceof Number)) {
067                        throw new ImageWriteException(
068                                "Invalid bit depth parameter: " + value);
069                    }
070                    bitDepth = ((Number) value).intValue();
071                }
072            }
073    
074            if (params.containsKey(PARAM_KEY_PIXEL_DENSITY)) {
075                final Object value = params.remove(PARAM_KEY_PIXEL_DENSITY);
076                if (value != null) {
077                    if (!(value instanceof PixelDensity)) {
078                        throw new ImageWriteException(
079                                "Invalid pixel density parameter");
080                    }
081                    pixelDensity = (PixelDensity) value;
082                }
083            }
084            if (pixelDensity == null) {
085                // DPI is mandatory, so we have to invent something
086                pixelDensity = PixelDensity.createFromPixelsPerInch(72, 72);
087            }
088    
089            if (params.size() > 0) {
090                final Object firstKey = params.keySet().iterator().next();
091                throw new ImageWriteException("Unknown parameter: " + firstKey);
092            }
093        }

095        private void writeScanLine(final BinaryOutputStream bos, final byte[] scanline)
096                throws IOException, ImageWriteException {
097            if (encoding == PcxImageParser.PcxHeader.ENCODING_UNCOMPRESSED) {
098                bos.writeByteArray(scanline);
099            } else {
100                if (encoding == PcxImageParser.PcxHeader.ENCODING_RLE) {
101                    int previousByte = -1;
102                    int repeatCount = 0;
103                    for (final byte element : scanline) {
104                        if ((element & 0xff) == previousByte
105                                && repeatCount < 63) {
106                            ++repeatCount;
107                        } else {
108                            if (repeatCount > 0) {
109                                if (repeatCount == 1
110                                        && (previousByte & 0xc0) != 0xc0) {
111                                    bos.write(previousByte);
112                                } else {
113                                    bos.write(0xc0 | repeatCount);
114                                    bos.write(previousByte);
115                                }
116                            }
117                            previousByte = 0xff & element;
118                            repeatCount = 1;
119                        }
120                    }
121                    if (repeatCount > 0) {
122                        if (repeatCount == 1 && (previousByte & 0xc0) != 0xc0) {
123                            bos.write(previousByte);
124                        } else {
125                            bos.write(0xc0 | repeatCount);
126                            bos.write(previousByte);
127                        }
128                    }
129                } else {
130                    throw new ImageWriteException("Invalid PCX encoding "
131                            + encoding);
132                }
133            }
134        }

136        public void writeImage(final BufferedImage src, final OutputStream os)
137                throws ImageWriteException, IOException {
138            final PaletteFactory paletteFactory = new PaletteFactory();
139            final SimplePalette palette = paletteFactory
140                    .makeExactRgbPaletteSimple(src, 256);
141            final BinaryOutputStream bos = new BinaryOutputStream(os,
142                    ByteOrder.INTEL);
143            if (palette == null || bitDepth == 24 || bitDepth == 32) {
144                if (bitDepth == 32) {
145                    write32BppPCX(src, bos);
146                } else {
147                    write24BppPCX(src, bos);
148                }
149            } else if (palette.length() > 16 || bitDepth == 8) {
150                write256ColorPCX(src, palette, bos);
151            } else if (palette.length() > 2 || bitDepth == 4) {
152                write16ColorPCX(src, palette, bos);
153            } else {
154                boolean onlyBlackAndWhite = true;
155                if (palette.length() >= 1) {
156                    final int rgb = palette.getEntry(0);
157                    if (rgb != 0 && rgb != 0xffffff) {
158                        onlyBlackAndWhite = false;
159                    }
160                }
161                if (palette.length() == 2) {
162                    final int rgb = palette.getEntry(1);
163                    if (rgb != 0 && rgb != 0xffffff) {
164                        onlyBlackAndWhite = false;
165                    }
166                }
167                if (onlyBlackAndWhite) {
168                    writeBlackAndWhitePCX(src, palette, bos);
169                } else {
170                    write16ColorPCX(src, palette, bos);
171                }
172            }
173        }

175        private void write32BppPCX(final BufferedImage src, final BinaryOutputStream bos)
176                throws ImageWriteException, IOException {
177            final int bytesPerLine = src.getWidth() % 2 == 0 ? src.getWidth() : src
178                    .getWidth() + 1;
179    
180            // PCX header
181            bos.write(10); // manufacturer
182            bos.write(5); // version
183            bos.write(encoding); // encoding
184            bos.write(32); // bits per pixel
185            bos.write2Bytes(0); // xMin
186            bos.write2Bytes(0); // yMin
187            bos.write2Bytes(src.getWidth() - 1); // xMax
188            bos.write2Bytes(src.getHeight() - 1); // yMax
189            bos.write2Bytes((short) Math.round(pixelDensity
190                    .horizontalDensityInches())); // hDpi
191            bos.write2Bytes((short) Math.round(pixelDensity.verticalDensityInches())); // vDpi
192            bos.writeByteArray(new byte[48]); // 16 color palette
193            bos.write(0); // reserved
194            bos.write(1); // planes
195            bos.write2Bytes(bytesPerLine); // bytes per line
196            bos.write2Bytes(1); // palette info
197            bos.write2Bytes(0); // hScreenSize
198            bos.write2Bytes(0); // vScreenSize
199            bos.writeByteArray(new byte[54]);
200    
201            final int rgbs[] = new int[src.getWidth()];
202            final byte rgbBytes[] = new byte[4 * bytesPerLine];
203            for (int y = 0; y < src.getHeight(); y++) {
204                src.getRGB(0, y, src.getWidth(), 1, rgbs, 0, src.getWidth());
205                for (int x = 0; x < rgbs.length; x++) {
206                    rgbBytes[4 * x + 0] = (byte) (rgbs[x] & 0xff);
207                    rgbBytes[4 * x + 1] = (byte) ((rgbs[x] >> 8) & 0xff);
208                    rgbBytes[4 * x + 2] = (byte) ((rgbs[x] >> 16) & 0xff);
209                    rgbBytes[4 * x + 3] = 0;
210                }
211                writeScanLine(bos, rgbBytes);
212            }
213        }

216                throws ImageWriteException, IOException {

218                    .getWidth() + 1;

220            // PCX header
221            bos.write(10); // manufacturer
222            bos.write(5); // version
223            bos.write(encoding); // encoding
224            bos.write(8); // bits per pixel
225            bos.write2Bytes(0); // xMin
226            bos.write2Bytes(0); // yMin
227            bos.write2Bytes(src.getWidth() - 1); // xMax
228            bos.write2Bytes(src.getHeight() - 1); // yMax
229            bos.write2Bytes((short) Math.round(pixelDensity
230                    .horizontalDensityInches())); // hDpi
231            bos.write2Bytes((short) Math.round(pixelDensity.verticalDensityInches())); // vDpi
232            bos.writeByteArray(new byte[48]); // 16 color palette
233            bos.write(0); // reserved
234            bos.write(3); // planes
235            bos.write2Bytes(bytesPerLine); // bytes per line
236            bos.write2Bytes(1); // palette info
237            bos.write2Bytes(0); // hScreenSize
238            bos.write2Bytes(0); // vScreenSize
239            bos.writeByteArray(new byte[54]);
240    
241            final int rgbs[] = new int[src.getWidth()];
242            final byte rgbBytes[] = new byte[3 * bytesPerLine];
243            for (int y = 0; y < src.getHeight(); y++) {
244                src.getRGB(0, y, src.getWidth(), 1, rgbs, 0, src.getWidth());
245                for (int x = 0; x < rgbs.length; x++) {
246                    rgbBytes[x] = (byte) ((rgbs[x] >> 16) & 0xff);
247                    rgbBytes[bytesPerLine + x] = (byte) ((rgbs[x] >> 8) & 0xff);
248                    rgbBytes[2 * bytesPerLine + x] = (byte) (rgbs[x] & 0xff);
249                }
250                writeScanLine(bos, rgbBytes);
251            }
252        }

254        private void writeBlackAndWhitePCX(final BufferedImage src,
255                final SimplePalette palette, final BinaryOutputStream bos)
256                throws ImageWriteException, IOException {
257            int bytesPerLine = (src.getWidth() + 7) / 8;
258            if (bytesPerLine % 2 != 0) {
259                ++bytesPerLine;
260            }
261    
262            // PCX header
263            bos.write(10); // manufacturer
264            bos.write(3); // version - it seems some apps only open
265            // black and white files if the version is 3...
266            bos.write(encoding); // encoding
267            bos.write(1); // bits per pixel
268            bos.write2Bytes(0); // xMin
269            bos.write2Bytes(0); // yMin
270            bos.write2Bytes(src.getWidth() - 1); // xMax
271            bos.write2Bytes(src.getHeight() - 1); // yMax
272            bos.write2Bytes((short) Math.round(pixelDensity
273                    .horizontalDensityInches())); // hDpi
274            bos.write2Bytes((short) Math.round(pixelDensity.verticalDensityInches())); // vDpi
275            bos.writeByteArray(new byte[48]); // 16 color palette
276            bos.write(0); // reserved
277            bos.write(1); // planes
278            bos.write2Bytes(bytesPerLine); // bytes per line
279            bos.write2Bytes(1); // palette info
280            bos.write2Bytes(0); // hScreenSize
281            bos.write2Bytes(0); // vScreenSize
282            bos.writeByteArray(new byte[54]);
283    
284            final byte[] row = new byte[bytesPerLine];
285            for (int y = 0; y < src.getHeight(); y++) {
286                Arrays.fill(row, (byte) 0);
287                for (int x = 0; x < src.getWidth(); x++) {
288                    final int rgb = 0xffffff & src.getRGB(x, y);
289                    int bit;
290                    if (rgb == 0x000000) {
291                        bit = 0;
292                    } else if (rgb == 0xffffff) {
293                        bit = 1;
294                    } else {
295                        throw new ImageWriteException(
296                                "Pixel neither black nor white");
297                    }
298                    row[x / 8] |= (bit << (7 - (x % 8)));
299                }
300                writeScanLine(bos, row);
301            }
302        }

304        private void write16ColorPCX(final BufferedImage src, final SimplePalette palette,
305                final BinaryOutputStream bos) throws ImageWriteException, IOException {
306            int bytesPerLine = (src.getWidth() + 1) / 2;
307            if (bytesPerLine % 2 != 0) {
308                ++bytesPerLine;
309            }
310    
311            final byte[] palette16 = new byte[16 * 3];
312            for (int i = 0; i < 16; i++) {
313                int rgb;
314                if (i < palette.length()) {
315                    rgb = palette.getEntry(i);
316                } else {
317                    rgb = 0;
318                }
319                palette16[3 * i + 0] = (byte) (0xff & (rgb >> 16));
320                palette16[3 * i + 1] = (byte) (0xff & (rgb >> 8));
321                palette16[3 * i + 2] = (byte) (0xff & rgb);
322            }
323    
324            // PCX header
325            bos.write(10); // manufacturer
326            bos.write(5); // version
327            bos.write(encoding); // encoding
328            bos.write(4); // bits per pixel
329            bos.write2Bytes(0); // xMin
330            bos.write2Bytes(0); // yMin
331            bos.write2Bytes(src.getWidth() - 1); // xMax
332            bos.write2Bytes(src.getHeight() - 1); // yMax
333            bos.write2Bytes((short) Math.round(pixelDensity
334                    .horizontalDensityInches())); // hDpi
335            bos.write2Bytes((short) Math.round(pixelDensity.verticalDensityInches())); // vDpi
336            bos.writeByteArray(palette16); // 16 color palette
337            bos.write(0); // reserved
338            bos.write(1); // planes
339            bos.write2Bytes(bytesPerLine); // bytes per line
340            bos.write2Bytes(1); // palette info
341            bos.write2Bytes(0); // hScreenSize
342            bos.write2Bytes(0); // vScreenSize
343            bos.writeByteArray(new byte[54]);
344    
345            final byte[] indeces = new byte[bytesPerLine];
346            for (int y = 0; y < src.getHeight(); y++) {
347                Arrays.fill(indeces, (byte) 0);
348                for (int x = 0; x < src.getWidth(); x++) {
349                    final int argb = src.getRGB(x, y);
350                    final int index = palette.getPaletteIndex(0xffffff & argb);
351                    indeces[x / 2] |= (index << 4 * (1 - (x % 2)));
352                }
353                writeScanLine(bos, indeces);
354            }
355        }

357        private void write256ColorPCX(final BufferedImage src, final SimplePalette palette,
358                final BinaryOutputStream bos) throws ImageWriteException, IOException {
359            final int bytesPerLine = src.getWidth() % 2 == 0 ? src.getWidth() : src
360                    .getWidth() + 1;
361    
362            // PCX header
363            bos.write(10); // manufacturer
364            bos.write(5); // version
365            bos.write(encoding); // encoding
366            bos.write(8); // bits per pixel
367            bos.write2Bytes(0); // xMin
368            bos.write2Bytes(0); // yMin
369            bos.write2Bytes(src.getWidth() - 1); // xMax
370            bos.write2Bytes(src.getHeight() - 1); // yMax
371            bos.write2Bytes((short) Math.round(pixelDensity
372                    .horizontalDensityInches())); // hDpi
373            bos.write2Bytes((short) Math.round(pixelDensity.verticalDensityInches())); // vDpi
374            bos.writeByteArray(new byte[48]); // 16 color palette
375            bos.write(0); // reserved
376            bos.write(1); // planes
377            bos.write2Bytes(bytesPerLine); // bytes per line
378            bos.write2Bytes(1); // palette info
379            bos.write2Bytes(0); // hScreenSize
380            bos.write2Bytes(0); // vScreenSize
381            bos.writeByteArray(new byte[54]);
382    
383            final byte[] indeces = new byte[bytesPerLine];
384            for (int y = 0; y < src.getHeight(); y++) {
385                for (int x = 0; x < src.getWidth(); x++) {
386                    final int argb = src.getRGB(x, y);
387                    final int index = palette.getPaletteIndex(0xffffff & argb);
388                    indeces[x] = (byte) index;
389                }
390                writeScanLine(bos, indeces);
391            }
392            // palette
393            bos.write(12);
394            for (int i = 0; i < 256; i++) {
395                int rgb;
396                if (i < palette.length()) {
397                    rgb = palette.getEntry(i);
398                } else {
399                    rgb = 0;
400                }
401                bos.write((rgb >> 16) & 0xff);
402                bos.write((rgb >> 8) & 0xff);
403                bos.write(rgb & 0xff);
404            }
405        }
