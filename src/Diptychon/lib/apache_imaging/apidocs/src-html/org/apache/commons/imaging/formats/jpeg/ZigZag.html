016    package org.apache.commons.imaging.formats.jpeg;
017    
018    public class ZigZag {
019        private static final int zigZag[] = {
020             0,  1,  5,  6, 14, 15, 27, 28,
021             2,  4,  7, 13, 16, 26, 29, 42,
022             3,  8, 12, 17, 25, 30, 41, 43,
023             9, 11, 18, 24, 31, 40, 44, 53,
024            10, 19, 23, 32, 39, 45, 52, 54,
025            20, 22, 33, 38, 46, 51, 55, 60,
026            21, 34, 37, 47, 50, 56, 59, 61,
027            35, 36, 48, 49, 57, 58, 62, 63
028        };
029    
030        public static void zigZagToBlock(final int[] zz, final int[] block) {
031            for (int i = 0; i < 64; i++) {
032                block[i] = zz[zigZag[i]];
033            }
034        }
035    
036        public static void blockToZigZag(final int[] block, final int[] zz) {
037            for (int i = 0; i < 64; i++) {
038                zz[zigZag[i]] = block[i];
039            }
040        }
041    }
