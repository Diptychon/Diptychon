017    package org.apache.commons.imaging.formats.jpeg.segments;
018    
019    import java.io.ByteArrayInputStream;
020    import java.io.IOException;
021    import java.io.InputStream;
022    import java.io.UnsupportedEncodingException;
023    
024    import org.apache.commons.imaging.common.BinaryFileParser;
025    
026    /**
027     * http://www.aiim.org/documents/standards/PDF-Ref/References/Adobe/5116.DCT_Filter.pdf
028     */
029    public class App14Segment extends AppnSegment {
030        private static final byte[] adobePrefix;
031        public static final int ADOBE_COLOR_TRANSFORM_UNKNOWN = 0;
032        public static final int ADOBE_COLOR_TRANSFORM_YCbCr = 1;
033        public static final int ADOBE_COLOR_TRANSFORM_YCCK = 2;
034        
035        static {
036            byte[] adobe = null;
037            try {
038                adobe = "Adobe".getBytes("US-ASCII");
039            } catch (final UnsupportedEncodingException cannotHappen) {
040            }
041            adobePrefix = adobe;
042        }
043    
044        public App14Segment(final int marker, final byte segmentData[])
045                throws IOException {
046            this(marker, segmentData.length, new ByteArrayInputStream(
047                    segmentData));
048        }
049        
050        public App14Segment(final int marker, final int marker_length, final InputStream is)
051                throws IOException {
052            super(marker, marker_length, is);
053        }
054    
055        public boolean isAdobeJpegSegment() {
056            return BinaryFileParser.byteArrayHasPrefix(bytes, adobePrefix);
057        }
058    
059        public int getAdobeColorTransform() {
060            return 0xff & bytes[11];
061        }
062    }
