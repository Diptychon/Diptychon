017    package org.apache.commons.imaging.formats.tiff.photometricinterpreters;
018    
019    import java.io.IOException;
020    
021    import org.apache.commons.imaging.ImageReadException;
022    import org.apache.commons.imaging.common.ImageBuilder;
023    
024    public abstract class PhotometricInterpreter {
025        protected final int samplesPerPixel;
026        protected final int bitsPerSample[];
027        protected final int predictor;
028        protected final int width;
029        protected final int height;
030    
031        public PhotometricInterpreter(final int fSamplesPerPixel, final int fBitsPerSample[],
032                final int Predictor, final int width, final int height) {
033            this.samplesPerPixel = fSamplesPerPixel;
034            this.bitsPerSample = fBitsPerSample;
035            this.predictor = Predictor;
036            this.width = width;
037            this.height = height;
038        }
039    
040        public abstract void interpretPixel(ImageBuilder imageBuilder,
041                int samples[], int x, int y) throws ImageReadException, IOException;
042    
043        /**
044         * @throws ImageReadException
045         *             may be thrown by sub-classes
046         * @throws IOException
047         *             may be thrown by sub-classes
048         */
049        public void dumpstats() throws ImageReadException, IOException {
050    
051        }
052    
053    }
