045        public boolean isPhotoshopJpegSegment(final byte segmentData[]) {
046            if (!BinaryFileParser.byteArrayHasPrefix(segmentData,
047                    PHOTOSHOP_IDENTIFICATION_STRING)) {
048                return false;
049            }
050    
051            final int index = PHOTOSHOP_IDENTIFICATION_STRING.size();
052            if (index + CONST_8BIM.size() > segmentData.length) {
053                return false;
054            }
055    
056            if (!CONST_8BIM.equals(segmentData, index, CONST_8BIM.size())) {
057                return false;
058            }
059    
060            return true;
061        }

063        /*
064         * In practice, App13 segments are only used for Photoshop/IPTC metadata.
065         * However, we should not treat App13 signatures without Photoshop's
066         * signature as Photoshop/IPTC segments.
067         * 
068         * A Photoshop/IPTC App13 segment begins with the Photoshop Identification
069         * string.
070         * 
071         * There follows 0-N blocks (Photoshop calls them "Image Resource Blocks").
072         * 
073         * Each block has the following structure:
074         * 
075         * 1. 4-byte type. This is always "8BIM" for blocks in a Photoshop App13
076         * segment. 2. 2-byte id. IPTC data is stored in blocks with id 0x0404, aka.
077         * IPTC_NAA_RECORD_IMAGE_RESOURCE_ID 3. Block name as a Pascal String. This
078         * is padded to have an even length. 4. 4-byte size (in bytes). 5. Block
079         * data. This is also padded to have an even length.
080         * 
081         * The block data consists of a 0-N records. A record has the following
082         * structure:
083         * 
084         * 1. 2-byte prefix. The value is always 0x1C02 2. 1-byte record type. The
085         * record types are documented by the IPTC. See IptcConstants. 3. 2-byte
086         * record size (in bytes). 4. Record data, "record size" bytes long.
087         * 
088         * Record data (unlike block data) is NOT padded to have an even length.
089         * 
090         * Record data, for IPTC record, should always be ISO-8859-1. But according
091         * to SANSELAN-33, this isn't always the case.
092         * 
093         * The exception is the first record in the block, which must always be a
094         * record version record, whose value is a two-byte number; the value is
095         * 0x02.
096         * 
097         * Some IPTC blocks are missing this first "record version" record, so we
098         * don't require it.
099         */
100        public PhotoshopApp13Data parsePhotoshopSegment(final byte bytes[], final Map<String,Object> params)
101                throws ImageReadException, IOException {
102            final boolean strict = ParamMap.getParamBoolean(params,
103                    ImagingConstants.PARAM_KEY_STRICT, false);
104            final boolean verbose = ParamMap.getParamBoolean(params,
105                    ImagingConstants.PARAM_KEY_VERBOSE, false);
106    
107            return parsePhotoshopSegment(bytes, verbose, strict);
108        }

110        public PhotoshopApp13Data parsePhotoshopSegment(final byte bytes[],
111                final boolean verbose, final boolean strict) throws ImageReadException,
112                IOException {
113            final List<IptcRecord> records = new ArrayList<IptcRecord>();
114    
115            final List<IptcBlock> allBlocks = parseAllBlocks(bytes, verbose, strict);
116    
117            for (int i = 0; i < allBlocks.size(); i++) {
118                final IptcBlock block = allBlocks.get(i);
119    
120                // Ignore everything but IPTC data.
121                if (!block.isIPTCBlock()) {
122                    continue;
123                }
124    
125                records.addAll(parseIPTCBlock(block.blockData, verbose));
126            }
127    
128            return new PhotoshopApp13Data(records, allBlocks);
129        }

131        protected List<IptcRecord> parseIPTCBlock(final byte bytes[], final boolean verbose)
132                throws ImageReadException, IOException {
133            final List<IptcRecord> elements = new ArrayList<IptcRecord>();
134    
135            int index = 0;
136            // Integer recordVersion = null;
137            while (index + 1 < bytes.length) {
138                final int tagMarker = 0xff & bytes[index++];
139                if (verbose) {
140                    Debug.debug("tagMarker",
141                            tagMarker + " (0x" + Integer.toHexString(tagMarker)
142                                    + ")");
143                }
144    
145                if (tagMarker != IPTC_RECORD_TAG_MARKER) {
146                    if (verbose) {
147                        System.out.println("Unexpected record tag marker in IPTC data.");
148                    }
149                    return elements;
150                }
151    
152                final int recordNumber = 0xff & bytes[index++];
153                if (verbose) {
154                    Debug.debug(
155                            "recordNumber",
156                            recordNumber + " (0x"
157                                    + Integer.toHexString(recordNumber) + ")");
158                }
159    
160                // int recordPrefix = convertByteArrayToShort("recordPrefix", index,
161                // bytes);
162                // if (verbose)
163                // Debug.debug("recordPrefix", recordPrefix + " (0x"
164                // + Integer.toHexString(recordPrefix) + ")");
165                // index += 2;
166                //
167                // if (recordPrefix != IPTC_RECORD_PREFIX)
168                // {
169                // if (verbose)
170                // System.out
171                // .println("Unexpected record prefix in IPTC data!");
172                // return elements;
173                // }
174    
175                // throw new ImageReadException(
176                // "Unexpected record prefix in IPTC data.");
177    
178                final int recordType = 0xff & bytes[index];
179                if (verbose) {
180                    Debug.debug("recordType",
181                            recordType + " (0x" + Integer.toHexString(recordType)
182                                    + ")");
183                }
184                index++;
185    
186                final int recordSize = convertByteArrayToShort("recordSize", index, bytes);
187                index += 2;
188    
189                final boolean extendedDataset = recordSize > IPTC_NON_EXTENDED_RECORD_MAXIMUM_SIZE;
190                final int dataFieldCountLength = recordSize & 0x7fff;
191                if (extendedDataset && verbose) {
192                    Debug.debug("extendedDataset. dataFieldCountLength: "
193                            + dataFieldCountLength);
194                }
195                if (extendedDataset) {
196                    // ignore extended dataset and everything after.
197                    return elements;
198                }
199    
200                final byte recordData[] = readBytearray("recordData", bytes, index,
201                        recordSize);
202                index += recordSize;
203    
204                // Debug.debug("recordSize", recordSize + " (0x"
205                // + Integer.toHexString(recordSize) + ")");
206    
207                if (recordNumber != IPTC_APPLICATION_2_RECORD_NUMBER) {
208                    continue;
209                }
210    
211                if (recordType == 0) {
212                    if (verbose) {
213                        System.out.println("ignore record version record! "
214                                + elements.size());
215                    }
216                    // ignore "record version" record;
217                    continue;
218                }
219                // if (recordVersion == null)
220                // {
221                // // The first record in a JPEG/Photoshop IPTC block must be
222                // // the record version.
223                // if (recordType != 0)
224                // throw new ImageReadException("Missing record version: "
225                // + recordType);
226                // recordVersion = new Integer(convertByteArrayToShort(
227                // "recordNumber", recordData));
228                //
229                // if (recordSize != 2)
230                // throw new ImageReadException(
231                // "Invalid record version record size: " + recordSize);
232                //
233                // // JPEG/Photoshop IPTC metadata is always in Record version
234                // // 2
235                // if (recordVersion.intValue() != 2)
236                // throw new ImageReadException(
237                // "Invalid IPTC record version: " + recordVersion);
238                //
239                // // Debug.debug("recordVersion", recordVersion);
240                // continue;
241                // }
242    
243                final String value = new String(recordData, "ISO-8859-1");
244    
245                final IptcType iptcType = IptcTypeLookup.getIptcType(recordType);
246    
247                // Debug.debug("iptcType", iptcType);
248                // debugByteArray("iptcData", iptcData);
249                // Debug.debug();
250    
251                // if (recordType == IPTC_TYPE_CREDIT.type
252                // || recordType == IPTC_TYPE_OBJECT_NAME.type)
253                // {
254                // this.debugByteArray("recordData", recordData);
255                // Debug.debug("index", IPTC_TYPE_CREDIT.name);
256                // }
257    
258                final IptcRecord element = new IptcRecord(iptcType, recordData, value);
259                elements.add(element);
260            }
261    
262            return elements;
263        }

265        protected List<IptcBlock> parseAllBlocks(final byte bytes[], final boolean verbose,
266                final boolean strict) throws ImageReadException, IOException {
267            final List<IptcBlock> blocks = new ArrayList<IptcBlock>();
268    
269            BinaryInputStream bis = null;
270            try {
271                bis = new BinaryInputStream(bytes, APP13_BYTE_ORDER);
272    
273                // Note that these are unsigned quantities. Name is always an even
274                // number of bytes (including the 1st byte, which is the size.)
275        
276                final byte[] idString = bis.readByteArray(
277                        PHOTOSHOP_IDENTIFICATION_STRING.size(),
278                        "App13 Segment missing identification string");
279                if (!PHOTOSHOP_IDENTIFICATION_STRING.equals(idString)) {
280                    throw new ImageReadException("Not a Photoshop App13 Segment");
281                }
282        
283                // int index = PHOTOSHOP_IDENTIFICATION_STRING.length;
284        
285                while (true) {
286                    final byte[] imageResourceBlockSignature = bis
287                            .readByteArray(CONST_8BIM.size(),
288                                    "App13 Segment missing identification string",
289                                    false, false);
290                    if (null == imageResourceBlockSignature) {
291                        break;
292                    }
293                    if (!CONST_8BIM.equals(imageResourceBlockSignature)) {
294                        throw new ImageReadException(
295                                "Invalid Image Resource Block Signature");
296                    }
297        
298                    final int blockType = bis
299                            .read2ByteInteger("Image Resource Block missing type");
300                    if (verbose) {
301                        Debug.debug("blockType",
302                                blockType + " (0x" + Integer.toHexString(blockType)
303                                        + ")");
304                    }
305        
306                    final int blockNameLength = bis
307                            .read1ByteInteger("Image Resource Block missing name length");
308                    if (verbose && blockNameLength > 0) {
309                        Debug.debug("blockNameLength", blockNameLength + " (0x"
310                                + Integer.toHexString(blockNameLength) + ")");
311                    }
312                    byte[] blockNameBytes;
313                    if (blockNameLength == 0) {
314                        bis.read1ByteInteger("Image Resource Block has invalid name");
315                        blockNameBytes = new byte[0];
316                    } else {
317                        blockNameBytes = bis.readByteArray(blockNameLength,
318                                "Invalid Image Resource Block name", verbose, strict);
319                        if (null == blockNameBytes) {
320                            break;
321                        }
322        
323                        if (blockNameLength % 2 == 0) {
324                            bis.read1ByteInteger("Image Resource Block missing padding byte");
325                        }
326                    }
327        
328                    final int blockSize = bis
329                            .read4ByteInteger("Image Resource Block missing size");
330                    if (verbose) {
331                        Debug.debug("blockSize",
332                                blockSize + " (0x" + Integer.toHexString(blockSize)
333                                        + ")");
334                    }
335        
336                    /*
337                     * doesn't catch cases where blocksize is invalid but is still less
338                     * than bytes.length but will at least prevent OutOfMemory errors
339                     */
340                    if (blockSize > bytes.length) {
341                        throw new ImageReadException("Invalid Block Size : "
342                                + blockSize + " > " + bytes.length);
343                    }
344        
345                    final byte[] blockData = bis.readByteArray(blockSize,
346                            "Invalid Image Resource Block data", verbose, strict);
347                    if (null == blockData) {
348                        break;
349                    }
350        
351                    blocks.add(new IptcBlock(blockType, blockNameBytes, blockData));
352        
353                    if ((blockSize % 2) != 0) {
354                        bis.read1ByteInteger("Image Resource Block missing padding byte");
355                    }
356                }
357        
358                return blocks;
359            } finally {
360                if (bis != null) {
361                    bis.close();
362                }
363            }
364        }

368        public byte[] writePhotoshopApp13Segment(final PhotoshopApp13Data data)
369                throws IOException, ImageWriteException {
370            final ByteArrayOutputStream os = new ByteArrayOutputStream();
371            final BinaryOutputStream bos = new BinaryOutputStream(os);
372    
373            PHOTOSHOP_IDENTIFICATION_STRING.writeTo(bos);
374    
375            final List<IptcBlock> blocks = data.getRawBlocks();
376            for (int i = 0; i < blocks.size(); i++) {
377                final IptcBlock block = blocks.get(i);
378    
379                CONST_8BIM.writeTo(bos);
380    
381                if (block.blockType < 0 || block.blockType > 0xffff) {
382                    throw new ImageWriteException("Invalid IPTC block type.");
383                }
384                bos.write2ByteInteger(block.blockType);
385    
386                if (block.blockNameBytes.length > 255) {
387                    throw new ImageWriteException("IPTC block name is too long: "
388                            + block.blockNameBytes.length);
389                }
390                bos.write(block.blockNameBytes.length);
391                bos.write(block.blockNameBytes);
392                if (block.blockNameBytes.length % 2 == 0) {
393                    bos.write(0); // pad to even size, including length byte.
394                }
395    
396                if (block.blockData.length > IPTC_NON_EXTENDED_RECORD_MAXIMUM_SIZE) {
397                    throw new ImageWriteException("IPTC block data is too long: "
398                            + block.blockData.length);
399                }
400                bos.write4ByteInteger(block.blockData.length);
401                bos.write(block.blockData);
402                if (block.blockData.length % 2 == 1) {
403                    bos.write(0); // pad to even size
404                }
405    
406            }
407    
408            bos.flush();
409            return os.toByteArray();
410        }

412        public byte[] writeIPTCBlock(List<IptcRecord> elements)
413                throws ImageWriteException, IOException {
414            byte blockData[];
415            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
416            final BinaryOutputStream bos = new BinaryOutputStream(baos,
417                    getByteOrder());
418    
419            // first, right record version record
420            bos.write(IPTC_RECORD_TAG_MARKER);
421            bos.write(IPTC_APPLICATION_2_RECORD_NUMBER);
422            bos.write(IptcTypes.RECORD_VERSION.type); // record version record
423                                                      // type.
424            bos.write2Bytes(2); // record version record size
425            bos.write2Bytes(2); // record version value
426    
427            // make a copy of the list.
428            elements = new ArrayList<IptcRecord>(elements);
429    
430            // sort the list. Records must be in numerical order.
431            final Comparator<IptcRecord> comparator = new Comparator<IptcRecord>() {
432                public int compare(final IptcRecord e1, final IptcRecord e2) {
433                    return e2.iptcType.getType() - e1.iptcType.getType();
434                }
435            };
436            Collections.sort(elements, comparator);
437            // TODO: make sure order right
438    
439            // write the list.
440            for (int i = 0; i < elements.size(); i++) {
441                final IptcRecord element = elements.get(i);
442    
443                if (element.iptcType == IptcTypes.RECORD_VERSION) {
444                    continue; // ignore
445                }
446    
447                bos.write(IPTC_RECORD_TAG_MARKER);
448                bos.write(IPTC_APPLICATION_2_RECORD_NUMBER);
449                if (element.iptcType.getType() < 0
450                        || element.iptcType.getType() > 0xff) {
451                    throw new ImageWriteException("Invalid record type: "
452                            + element.iptcType.getType());
453                }
454                bos.write(element.iptcType.getType());
455    
456                final byte recordData[] = element.value.getBytes("ISO-8859-1");
457                if (!new String(recordData, "ISO-8859-1").equals(element.value)) {
458                    throw new ImageWriteException(
459                            "Invalid record value, not ISO-8859-1");
460                }
461    
462                bos.write2Bytes(recordData.length);
463                bos.write(recordData);
464            }
465    
466            blockData = baos.toByteArray();
467    
468            return blockData;
469        }
