017    package org.apache.commons.imaging.formats.bmp.pixelparsers;
018    
019    import java.io.ByteArrayInputStream;
020    import java.io.IOException;
021    
022    import org.apache.commons.imaging.ImageReadException;
023    import org.apache.commons.imaging.common.BinaryFileParser;
024    import org.apache.commons.imaging.common.ByteOrder;
025    import org.apache.commons.imaging.common.ImageBuilder;
026    import org.apache.commons.imaging.formats.bmp.BmpHeaderInfo;
027    
028    public abstract class PixelParser {
029        public final BmpHeaderInfo bhi;
030        public final byte colorTable[];
031        public final byte imageData[];
032    
033        protected final BinaryFileParser bfp;
034        protected final ByteArrayInputStream is;
035    
036        public PixelParser(final BmpHeaderInfo bhi, final byte ColorTable[], final byte ImageData[]) {
037            this.bhi = bhi;
038            this.colorTable = ColorTable;
039            this.imageData = ImageData;
040    
041            bfp = new BinaryFileParser(ByteOrder.INTEL);
042            is = new ByteArrayInputStream(ImageData);
043        }
044    
045        public abstract void processImage(ImageBuilder imageBuilder)
046                throws ImageReadException, IOException;
047    
048        protected int getColorTableRGB(int index) {
049            index *= 4;
050            final int blue = 0xff & colorTable[index + 0];
051            final int green = 0xff & colorTable[index + 1];
052            final int red = 0xff & colorTable[index + 2];
053            final int alpha = 0xff;
054    
055            final int rgb = (alpha << 24) | (red << 16) | (green << 8) | (blue << 0);
056            return rgb;
057        }
058    
059    }
